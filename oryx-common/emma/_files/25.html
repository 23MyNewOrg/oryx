419        // apply Householder transforms to solve Q.y = b
420        for (int minor = 0; minor < FastMath.min(m, n); minor++) {
421          final double[] qrtMinor = qrt[minor];
422          final double factor     = 1.0 / (rDiag[minor] * qrtMinor[minor]);
423 
424          Arrays.fill(alpha, 0, kWidth, 0.0);
425          for (int row = minor; row < m; ++row) {
426            final double   d    = qrtMinor[row];
427            final double[] yRow = y[row];
428            for (int k = 0; k < kWidth; ++k) {
429              alpha[k] += d * yRow[k];
430            }
431          }
432          for (int k = 0; k < kWidth; ++k) {
433            alpha[k] *= factor;
434          }
435 
436          for (int row = minor; row < m; ++row) {
437            final double   d    = qrtMinor[row];
438            final double[] yRow = y[row];
439            for (int k = 0; k < kWidth; ++k) {
440              yRow[k] += alpha[k] * d;
441            }
442          }
443        }
444 
445        // solve triangular system R.x = y
446        for (int j = rDiag.length - 1; j >= 0; --j) {
447          final int      jBlock = j / blockSize;
448          final int      jStart = jBlock * blockSize;
449          final double   factor = 1.0 / rDiag[j];
450          final double[] yJ     = y[j];
451          final double[] xBlock = xBlocks[jBlock * cBlocks + kBlock];
452          int index = (j - jStart) * kWidth;
453          for (int k = 0; k < kWidth; ++k) {
454            yJ[k]          *= factor;
455            xBlock[index++] = yJ[k];
456          }
457 
458          final double[] qrtJ = qrt[j];
459          for (int i = 0; i < j; ++i) {
460            final double rIJ  = qrtJ[i];
461            final double[] yI = y[i];
462            for (int k = 0; k < kWidth; ++k) {
463              yI[k] -= yJ[k] * rIJ;
464            }
465          }
466        }
467      }
468 
469      return new BlockRealMatrix(n, columns, xBlocks, false);
470    }

473     * {@inheritDoc}
474     * @throws SingularMatrixException if the decomposed matrix is singular.
475     */

477      return solve(MatrixUtils.createRealIdentityMatrix(qrt[0].length));
478    }
