34    public static RealMatrix multiplyXYT(LongObjectMap<float[]> X, LongObjectMap<float[]> Y) {
35      int Ysize = Y.size();
36      int Xsize = X.size();
37      RealMatrix result = new Array2DRowRealMatrix(Xsize, Ysize);
38      for (int row = 0; row < Xsize; row++) {
39        for (int col = 0; col < Ysize; col++) {
40          result.setEntry(row, col, SimpleVectorMath.dot(X.get(row), Y.get(col)));
41        }
42      }
43      return result;
44    }

75    @Test
76    public void testTransposeTimesSelf() {
77      LongObjectMap<float[]> M = new LongObjectMap<float[]>();
78      M.put(1L, new float[] {4.0f, -1.0f, -5.0f});
79      M.put(2L, new float[] {2.0f, 0.0f, 3.0f});
80      RealMatrix MTM = MatrixUtils.transposeTimesSelf(M);
81      assertArrayEquals(new double[]{20.0, -4.0, -14.0}, MTM.getRow(0));
82      assertArrayEquals(new double[]{-4.0, 1.0, 5.0}, MTM.getRow(1));
83      assertArrayEquals(new double[]{-14.0, 5.0, 34.0}, MTM.getRow(2));
84    }

87    public void testMultiplyXYT() {
88      LongObjectMap<float[]> X = new LongObjectMap<float[]>();
89      X.put(0L, new float[] {2.0f, 3.5f});
90      X.put(1L, new float[] {-1.0f, 0.0f});
91      LongObjectMap<float[]> Y = new LongObjectMap<float[]>();
92      Y.put(0L, new float[] {0.0f, -3.0f});
93      Y.put(1L, new float[] {-0.5f, 1.5f});
94      RealMatrix product = multiplyXYT(X, Y);
95      assertEquals(-10.5, product.getEntry(0, 0));
96      assertEquals(4.25, product.getEntry(0, 1));
