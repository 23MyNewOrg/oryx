29  public final class DelimitedDataUtilsTest extends OryxTest {
30  
31    @Test
32    public void testEncode() {
33      assertEquals("", DelimitedDataUtils.encode());
34      assertEquals("foo", DelimitedDataUtils.encode("foo"));
35      assertEquals("foo,bar", DelimitedDataUtils.encode("foo", "bar"));
36    }
37  
38    @Test
39    public void testEncodeDelimiters() {
40      assertEquals("\"foo,bing\",bar", DelimitedDataUtils.encode("foo,bing", "bar"));
41    }
42  
43    @Test
44    public void testEncodeQuote() {
45      assertEquals("foo,\"bar\"\"bing\"", DelimitedDataUtils.encode("foo", "bar\"bing"));
46    }
47  
48    @Test
49    public void testDecode() {
50      assertArrayEquals(new String[0], DelimitedDataUtils.decode(""));
51      assertArrayEquals(new String[] {"foo"}, DelimitedDataUtils.decode("foo"));
52      assertArrayEquals(new String[] {"foo","bar"}, DelimitedDataUtils.decode("foo,bar"));
53    }
54  
55    @Test
56    public void testDecodeEmpty() {
57      assertArrayEquals(new String[0], DelimitedDataUtils.decode(""));
58      assertArrayEquals(new String[] {"foo", "bar", ""}, DelimitedDataUtils.decode("foo,bar,"));
59      assertArrayEquals(new String[] {"", "foo", "bar"}, DelimitedDataUtils.decode(",foo,bar"));
60      assertArrayEquals(new String[] {"foo", "", "bar"}, DelimitedDataUtils.decode("foo,,bar"));
61    }
62  
63    @Test
64    public void testDecodeDelimiters() {
65      assertArrayEquals(new String[] {"foo,bing","bar"}, DelimitedDataUtils.decode("\"foo,bing\",bar"));
66    }
67  
68    @Test
69    public void testDecodeQuote() {
70      assertArrayEquals(new String[] {"foo", "bar\"bing"}, DelimitedDataUtils.decode("foo,\"bar\"\"bing\""));
71    }
72  
73    @Test
74    public void testVsIterable() {
75      assertEquals(DelimitedDataUtils.encode("foo", "bar"), DelimitedDataUtils.encode(Arrays.asList("foo", "bar")));
76    }
77  
78    @Test
79    public void testVsObject() {
80      assertEquals(DelimitedDataUtils.encode("3.0", "-2.0"), DelimitedDataUtils.encode(Arrays.asList(3.0, -2.0)));
81    }
