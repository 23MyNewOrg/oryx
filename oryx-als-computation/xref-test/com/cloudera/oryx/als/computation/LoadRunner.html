45  /**
46   * Runs a mixed, concurrent load against a given recommender instance.
47   * 
48   * @author Sean Owen
49   */
50  public final class LoadRunner implements Callable<Object> {
51    
52    private static final Logger log = LoggerFactory.getLogger(LoadRunner.class);
53    
54    private final OryxRecommender client;
55    private final String[] uniqueUserIDs;
56    private final String[] uniqueItemIDs;
57    private final int steps;
58  
59    /**
60     * @param client recommender to load
61     * @param dataDirectory a directory containing data files from which user and item IDs should be read
62     * @param steps number of load steps to run
63     */
64    public LoadRunner(OryxRecommender client, File dataDirectory, int steps) throws IOException {
65      Preconditions.checkNotNull(client);
66      Preconditions.checkNotNull(dataDirectory);
67      Preconditions.checkArgument(steps > 0);  
68      
69      log.info("Reading IDs...");    
70      Set<String> userIDsSet = Sets.newHashSet();
71      Set<String> itemIDsSet = Sets.newHashSet();
72      for (File f : dataDirectory.listFiles(IOUtils.CSV_COMPRESSED_FILTER)) {
73        if (!f.getName().contains("oryx-append")) {
74          for (CharSequence line : new FileLineIterable(f)) {
75            String[] columns = DelimitedDataUtils.decode(line);
76            userIDsSet.add(columns[0]);
77            itemIDsSet.add(columns[1]);
78          }
79        }
80      }
81      
82      this.client = client;    
83      this.uniqueUserIDs = userIDsSet.toArray(new String[userIDsSet.size()]);
84      this.uniqueItemIDs = itemIDsSet.toArray(new String[itemIDsSet.size()]);
85      this.steps = steps;
86    }
87  
88    public int getSteps() {
89      return steps;
90    }
91  
92    @Override
93    public Void call() throws Exception {
94      runLoad();
95      return null;
96    }
97    
98    public void runLoad() throws InterruptedException {
99  
100     final StorelessUnivariateStatistic recommendedBecause = new Mean();
101     final StorelessUnivariateStatistic setPreference = new Mean();
102     final StorelessUnivariateStatistic removePreference = new Mean();
103     final StorelessUnivariateStatistic ingest = new Mean();
104     final StorelessUnivariateStatistic refresh = new Mean();
105     final StorelessUnivariateStatistic estimatePreference = new Mean();
106     final StorelessUnivariateStatistic mostSimilarItems = new Mean();
107     final StorelessUnivariateStatistic similarityToItem = new Mean();
108     final StorelessUnivariateStatistic mostPopularItems = new Mean();
109     final StorelessUnivariateStatistic recommendToMany = new Mean();
110     final StorelessUnivariateStatistic recommend = new Mean();
111     final RandomGenerator random = RandomManager.getRandom();
112 
113     int numCores = Runtime.getRuntime().availableProcessors();
114     final int stepsPerWorker = steps / numCores;
115     Collection<Callable<Object>> workers = Lists.newArrayListWithCapacity(numCores);
116     for (int i = 0; i < numCores; i++) {
117       workers.add(new Callable<Object>() {
118         @Override
119         public Void call() throws Exception {
120           for (int i = 0; i < stepsPerWorker; i++) {
121             double r;
122             String userID;
123             String itemID;
124             String itemID2;
125             float value;
126             synchronized (random) {
127               r = random.nextDouble();
128               userID = uniqueUserIDs[random.nextInt(uniqueUserIDs.length)];
129               itemID = uniqueItemIDs[random.nextInt(uniqueItemIDs.length)];
130               itemID2 = uniqueItemIDs[random.nextInt(uniqueItemIDs.length)];
131               value = random.nextInt(10);
132             }
133             long stepStart = System.currentTimeMillis();
134             if (r < 0.05) {
135               client.recommendedBecause(userID, itemID, 10);
136               recommendedBecause.increment(System.currentTimeMillis() - stepStart);
137             } else if (r < 0.07) {
138               client.setPreference(userID, itemID);
139               setPreference.increment(System.currentTimeMillis() - stepStart);
140             } else if (r < 0.08) {
141               client.setPreference(userID, itemID, value);
142               setPreference.increment(System.currentTimeMillis() - stepStart);
143             } else if (r < 0.11) {
144               client.removePreference(userID, itemID);
145               removePreference.increment(System.currentTimeMillis() - stepStart);
146             } else if (r < 0.12) {
147               Reader reader = new StringReader(DelimitedDataUtils.encode(userID, itemID, Float.toString(value)) + '\n');
148               client.ingest(reader);
149               ingest.increment(System.currentTimeMillis() - stepStart);
150             } else if (r < 0.13) {
151               client.refresh();
152               refresh.increment(System.currentTimeMillis() - stepStart);
153             } else if (r < 0.14) {
154               client.similarityToItem(itemID, itemID2);
155               similarityToItem.increment(System.currentTimeMillis() - stepStart);
156             } else if (r < 0.15) {
157               client.mostPopularItems(10);
158               mostPopularItems.increment(System.currentTimeMillis() - stepStart);
159             } else if (r < 0.19) {
160               client.estimatePreference(userID, itemID);
161               estimatePreference.increment(System.currentTimeMillis() - stepStart);
162             } else if (r < 0.20) {
163               client.estimateForAnonymous(itemID, new String[]{itemID2});
164               estimatePreference.increment(System.currentTimeMillis() - stepStart);
165             } else if (r < 0.25) {
166               client.mostSimilarItems(new String[]{itemID}, 10);
167               mostSimilarItems.increment(System.currentTimeMillis() - stepStart);
168             } else if (r < 0.30) {
169               client.recommendToMany(new String[]{userID, userID}, 10, true, null);
170               recommendToMany.increment(System.currentTimeMillis() - stepStart);
171             } else {
172               client.recommend(userID, 10);
173               recommend.increment(System.currentTimeMillis() - stepStart);
174             }
175           }
176           return null;
177         }
178       });
179     }
180 
181     log.info("Starting load test...");
182 
183     long start = System.currentTimeMillis();
184     ExecutorService executor = Executors.newFixedThreadPool(numCores);
185     Iterable<Future<Object>> futures;
186     try {
187       futures = executor.invokeAll(workers);
188     } finally {
189       ExecutorUtils.shutdownNowAndAwait(executor);
190     }
191     long end = System.currentTimeMillis();
192 
193     ExecutorUtils.checkExceptions(futures);
194 
195     log.info("Finished {} steps in {}ms", steps, end - start);
196 
197     log.info("recommendedBecause: {}", recommendedBecause.getResult());
198     log.info("setPreference: {}", setPreference.getResult());
199     log.info("removePreference: {}", removePreference.getResult());
200     log.info("ingest: {}", ingest.getResult());
201     log.info("refresh: {}", refresh.getResult());
202     log.info("estimatePreference: {}", estimatePreference.getResult());
203     log.info("mostSimilarItems: {}", mostSimilarItems.getResult());
204     log.info("similarityToItem: {}", similarityToItem.getResult());
205     log.info("mostPopularItems: {}", mostPopularItems.getResult());        
206     log.info("recommendToMany: {}", recommendToMany.getResult());
207     log.info("recommend: {}", recommend.getResult());    
208   }
