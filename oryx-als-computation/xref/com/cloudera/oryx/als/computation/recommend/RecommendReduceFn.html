25  import com.google.common.base.Preconditions;
26  import org.apache.crunch.CrunchRuntimeException;
27  import org.apache.crunch.Emitter;
28  import org.apache.crunch.Pair;
29  import org.apache.hadoop.conf.Configuration;
30  
31  import java.io.IOException;
32  
33  public final class RecommendReduceFn extends OryxReduceDoFn<Integer,
34      Iterable<Pair<Long, Pair<float[], LongSet>>>,
35      Pair<Long,NumericIDValue>> {
36  
37    private int numRecs;
38    private LongObjectMap<float[]> partialY;
39  
40    @Override
41    public void initialize() {
42      super.initialize();
43      Configuration conf = getConfiguration();
44      String yKey = conf.get(RecommendStep.Y_KEY_KEY);
45      try {
46        partialY = ComputationDataUtils.loadPartialY(getPartition(), getNumPartitions(), yKey, conf);
47      } catch (IOException e) {
48        throw new CrunchRuntimeException(e);
49      }
50      numRecs = ConfigUtils.getDefaultConfig().getInt("model.recommend.how-many");
51      Preconditions.checkArgument(numRecs > 0, "# recs must be positive: %s", numRecs);
52    }
53  
54    @Override
55    public void process(Pair<Integer, Iterable<Pair<Long, Pair<float[], LongSet>>>> input,
56                        Emitter<Pair<Long, NumericIDValue>> emitter) {
57      Preconditions.checkState(input.first() == getPartition(),
58          "Key must match partition: %s != %s", input.first(), getPartition());
59      for (Pair<Long, Pair<float[], LongSet>> value : input.second()) {
60        long userID = value.first();
61        float[] userFeatures = value.second().first();
62        LongSet knownItemIDs = value.second().second();
63        Iterable<NumericIDValue> recs = TopN.selectTopN(
64            new RecommendIterator(userFeatures, partialY.entrySet().iterator(), knownItemIDs),
65            numRecs);
66        for (NumericIDValue rec : recs) {
67          emitter.emit(Pair.of(userID, rec));
68        }
69      }
70    }
71  }
