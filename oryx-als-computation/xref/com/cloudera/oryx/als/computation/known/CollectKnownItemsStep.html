18  import java.io.IOException;
19  
20  import com.cloudera.oryx.als.computation.ALSJobStep;
21  import com.cloudera.oryx.als.computation.types.ALSTypes;
22  import com.cloudera.oryx.common.servcomp.Namespaces;
23  import com.cloudera.oryx.computation.common.JobStepConfig;
24  
25  import org.apache.crunch.impl.mr.MRPipeline;
26  import org.apache.crunch.types.avro.Avros;
27  
28  /**
29   * @author Sean Owen
30   */
31  public final class CollectKnownItemsStep extends ALSJobStep {
32  
33    @Override
34    protected MRPipeline createPipeline() throws IOException {
35  
36      JobStepConfig config = getConfig();
37      String instanceDir = config.getInstanceDir();
38      long generationID = config.getGenerationID();
39  
40      String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "knownItems/";
41  
42      if (!validOutputPath(outputKey)) {
43        return null;
44      }
45  
46      MRPipeline p = createBasicPipeline(CollectKnownItemsFn.class);
47      // Really should read in and exclude tag IDs but doesn't really hurt much
48      p.read(input(Namespaces.getTempPrefix(instanceDir, generationID) + "userVectors/", ALSTypes.SPARSE_ROW_MATRIX))
49          .parallelDo("collectKnownItems", new CollectKnownItemsFn(), Avros.strings())
50          .write(compressedTextOutput(p.getConfiguration(), outputKey));
51      return p;
52    }
53  
54    public static void main(String[] args) throws Exception {
55      run(new CollectKnownItemsStep(), args);
56    }
57  
58  }
