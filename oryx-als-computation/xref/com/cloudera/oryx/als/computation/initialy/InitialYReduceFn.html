16  package com.cloudera.oryx.als.computation.initialy;
17  
18  import com.cloudera.oryx.als.computation.types.MatrixRow;
19  import com.cloudera.oryx.common.math.SimpleVectorMath;
20  import com.cloudera.oryx.common.random.RandomManager;
21  import com.cloudera.oryx.common.random.RandomUtils;
22  import com.cloudera.oryx.common.settings.ConfigUtils;
23  import com.cloudera.oryx.computation.common.fn.OryxReduceMapFn;
24  import com.google.common.base.Preconditions;
25  import com.google.common.collect.Lists;
26  import com.typesafe.config.Config;
27  import org.apache.commons.math3.random.RandomGenerator;
28  import org.apache.crunch.Pair;
29  
30  import java.util.List;
31  
32  public final class InitialYReduceFn extends OryxReduceMapFn<Long, float[], MatrixRow> {
33  
34    private static final int MAX_FAR_FROM_VECTORS = 100000;
35  
36    private RandomGenerator random;
37    private int features;
38    private List<float[]> farFrom;
39  
40    @Override
41    public void initialize() {
42      super.initialize();
43      random = RandomManager.getRandom();
44      Config config = ConfigUtils.getDefaultConfig();
45      features = config.getInt("model.features");
46      Preconditions.checkArgument(features > 0, "Number of features must be positive");
47      farFrom = Lists.newArrayList();
48    }
49  
50    @Override
51    public MatrixRow map(Pair<Long, float[]> input) {
52      long itemID = input.first();
53      // Did we see any previous feature vector from previous Y?
54      float[] featureVector = null;
55      float[] maybeFeatureVector = input.second();
56      int maybeLength = maybeFeatureVector.length;
57      if (maybeLength > 0) {
58        if (maybeLength == features) {
59          // Only use this vector if not empty and has the right number of features
60          featureVector = maybeFeatureVector;
61        } else if (maybeLength > features) {
62          // Copy part of the existing vector
63          featureVector = new float[features];
64          System.arraycopy(maybeFeatureVector, 0, featureVector, 0, featureVector.length);
65          SimpleVectorMath.normalize(featureVector);
66        } else if (maybeLength < features) {
67          featureVector = new float[features];
68          System.arraycopy(maybeFeatureVector, 0, featureVector, 0, maybeLength);
69          for (int i = maybeLength; i < featureVector.length; i++) {
70            featureVector[i] = (float) random.nextGaussian();
71          }
72          SimpleVectorMath.normalize(featureVector);
73        }
74      }
75  
76      if (featureVector == null) {
77        // No suitable prior vector, build a new one
78        featureVector = RandomUtils.randomUnitVectorFarFrom(features, farFrom, random);
79      }
80  
81      if (farFrom.size() < MAX_FAR_FROM_VECTORS) { // Simple cap to keep from getting too big
82        farFrom.add(featureVector);
83      }
84  
85      return new MatrixRow(itemID, featureVector);
86    }
87  }
