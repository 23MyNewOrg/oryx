37  /**
38   * Generates a weighted random sample of N items from a distributed data set using the reservoir
39   * algorithm described in <a href="http://utopia.duth.gr/~pefraimi/research/data/2007EncOfAlg.pdf">Efraimidis
40   * and Spirakis (2005)</a>.
41   * 
42   */
43  public final class ReservoirSampling {
44  
45    private ReservoirSampling() {
46    }
47  
48    public static <T> PCollection<T> sample(
49        PCollection<T> input,
50        int sampleSize) {
51      return sample(input, sampleSize, null);
52    }

54    public static <T> PCollection<T> sample(
55        PCollection<T> input,
56        int sampleSize,
57        RandomGenerator random) {
58      PTypeFamily ptf = input.getTypeFamily();
59      PType<Pair<T, Integer>> ptype = ptf.pairs(input.getPType(), ptf.ints());
60      return weightedSample(
61          input.parallelDo(new MapFn<T, Pair<T, Integer>>() {
62            @Override
63            public Pair<T, Integer> map(T t) { return Pair.of(t, 1); }
64          }, ptype),
65          sampleSize,
66          random);
67    }

69    public static <T, N extends Number> PCollection<T> weightedSample(
70        PCollection<Pair<T, N>> input,
71        int sampleSize) {
72      return weightedSample(input, sampleSize, null);
73    }

75    public static <T, N extends Number> PCollection<T> weightedSample(
76        PCollection<Pair<T, N>> input,
77        int sampleSize,
78        RandomGenerator random) {
79      PTypeFamily ptf = input.getTypeFamily();
80      PTable<Integer, Pair<T, N>> groupedIn = input.parallelDo(
81          new MapFn<Pair<T, N>, Pair<Integer, Pair<T, N>>>() {
82            @Override
83            public Pair<Integer, Pair<T, N>> map(Pair<T, N> p) {
84              return Pair.of(0, p);
85            }
86          }, ptf.tableOf(ptf.ints(), input.getPType()));
87      return groupedWeightedSample(groupedIn, sampleSize, random).values();
88    }

91        PTable<K, Pair<T, N>> input,
92        int sampleSize) {
93      return groupedWeightedSample(input, sampleSize, null);
94    }

96    public static <K, T, N extends Number> PTable<K, T> groupedWeightedSample(
97        PTable<K, Pair<T, N>> input,
98        int sampleSize,
99        RandomGenerator random) {
100     PTypeFamily ptf = input.getTypeFamily();
101     PType<K> keyType = input.getPTableType().getKeyType();
102     @SuppressWarnings("unchecked")
103     PType<T> ttype = (PType<T>) input.getPTableType().getValueType().getSubTypes().get(0);
104     PTableType<K, Pair<Double, T>> ptt = ptf.tableOf(keyType, ptf.pairs(ptf.doubles(), ttype));
105 
106     // fill reservoirs by mapping over the vectors and re-emiting them; each map task emits at most sampleSize
107     // vectors per fold; the combiner/reducer will combine the outputs and pare down to sampleSize vectors total
108     PTable<K, Pair<Double, T>> samples = input.parallelDo("reservoirSampling",
109         new SampleFn<K, T, N>(sampleSize, random, ttype), ptt);
110 
111     // pare down to just a single reservoir with sampleSize vectors
112     PTable<K, Pair<Double, T>> reservoir = samples.groupByKey(1).combineValues(new WRSCombineFn<K, T>(sampleSize, ttype));
113 
114     // strip the weights off the final sampled reservoir and return
115     return reservoir.parallelDo("strippingSamplingWeights", new MapFn<Pair<K, Pair<Double, T>>, Pair<K, T>>() {
116       @Override
117       public Pair<K, T> map(Pair<K, Pair<Double, T>> p) {
118         return Pair.of(p.first(), p.second().second());
119       }
120     }, ptf.tableOf(keyType, ttype));
121   }

123   private static final class SampleFn<K, T, N extends Number>
124       extends DoFn<Pair<K, Pair<T, N>>, Pair<K, Pair<Double, T>>> {
125   
126     private static final int PUT_LIMIT = 100000;
127     
128     private final int sampleSize;
129     private final PType<T> ptype;
130     private transient Map<K, SortedMap<Double, T>> current;
131     private RandomGenerator random;
132     private int puts;
133     
134     private SampleFn(int sampleSize, RandomGenerator random, PType<T> ptype) {
135       this.sampleSize = sampleSize;
136       this.ptype = ptype;
137       this.random = random;
138     }
139     
140     @Override
141     public float scaleFactor() {
142       // Indicate that the output of sampling will be on the small side
143       return 0.05f;
144     }
145     
146     @Override
147     public void initialize() {
148       ptype.initialize(getConfiguration());
149       if (current == null) {
150         this.current = Maps.newHashMap();
151       } else {
152         current.clear();
153       }
154       if (random == null) {
155         this.random = RandomManager.getRandom();
156       }
157     }
158     
159     @Override
160     public void process(Pair<K, Pair<T, N>> input,
161         Emitter<Pair<K, Pair<Double, T>>> emitter) {
162       K id = input.first();
163       Pair<T, N> p = input.second();
164       double weight = p.second().doubleValue();
165       if (weight > 0.0) {
166         double score = Math.log(random.nextDouble()) / weight;
167         SortedMap<Double, T> reservoir = current.get(id);
168         if (reservoir == null) {
169           reservoir = Maps.newTreeMap();
170           current.put(id, reservoir);
171         }
172         if (reservoir.size() < sampleSize) { 
173           reservoir.put(score, ptype.getDetachedValue(p.first()));
174           puts++;
175         } else if (score > reservoir.firstKey()) {
176           reservoir.remove(reservoir.firstKey());
177           reservoir.put(score, ptype.getDetachedValue(p.first()));
178         }
179         if (puts > PUT_LIMIT) {
180           // On the off-chance this gets huge, cleanup
181           // Can occur if sampleSize > PUT_LIMIT
182           cleanup(emitter);
183         }
184       }
185     }
186     
187     @Override
188     public void cleanup(Emitter<Pair<K, Pair<Double, T>>> emitter) {
189       for (Map.Entry<K,SortedMap<Double,T>> entry : current.entrySet()) {
190         Map<Double, T> reservoir = entry.getValue();
191         for (Map.Entry<Double,T> e : reservoir.entrySet()) {
192           emitter.emit(Pair.of(entry.getKey(), Pair.of(e.getKey(), e.getValue())));
193         }
194       }
195       current.clear();
196       puts = 0;
197     }
198   }

200   private static final class WRSCombineFn<K, T> extends CombineFn<K, Pair<Double, T>> {
201 
202     private final int sampleSize;
203     private final PType<T> ptype;
204     
205     private WRSCombineFn(int sampleSize, PType<T> ptype) {
206       this.sampleSize = sampleSize;
207       this.ptype = ptype;
208     }
209 
210     @Override
211     public void initialize() {
212       ptype.initialize(getConfiguration());
213     }
214     
215     @Override
216     public void process(Pair<K, Iterable<Pair<Double, T>>> input,
217         Emitter<Pair<K, Pair<Double, T>>> emitter) {
218       SortedMap<Double, T> reservoir = Maps.newTreeMap();
219       for (Pair<Double, T> p : input.second()) {
220         if (reservoir.size() < sampleSize) { 
221           reservoir.put(p.first(), ptype.getDetachedValue(p.second()));        
222         } else if (p.first() > reservoir.firstKey()) {
223           reservoir.remove(reservoir.firstKey());
224           reservoir.put(p.first(), ptype.getDetachedValue(p.second()));  
225         }
226       }
227       for (Map.Entry<Double, T> e : reservoir.entrySet()) {
228         emitter.emit(Pair.of(input.first(), Pair.of(e.getKey(), e.getValue())));
229       }
230     }
231   }
