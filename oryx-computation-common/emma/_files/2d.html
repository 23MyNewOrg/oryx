337  private static boolean isUploadInProgress(String uploadingGenerationPrefix) throws IOException {
338    Store store = Store.get();
339    long now = System.currentTimeMillis();
340    for (String fileName : store.list(uploadingGenerationPrefix, true)) {
341      long lastModified = store.getLastModified(fileName);
342 
343      if (fileName.endsWith(".inprogress") || fileName.endsWith("_COPYING_")) {
344        // .inprogress is our marker for uploading files; _COPYING_ is Hadoop's
345        if (lastModified > now - 3 * GENERATION_WAIT) {
346          log.info("At least one upload is in progress ({})", fileName);
347          return true;
348        }
349        // Else, orphaned upload -- waited a long, long time with no progress
350        log.warn("Stale upload to {}? Deleting and continuing", fileName);
351        try {
352          store.delete(fileName);
353        } catch (IOException ioe) {
354          log.info("Could not delete {}", fileName, ioe);
355        }
356      } else {
357        // All other (data) files. Look for recent modification, but only quite recent.
358        // This accounts for possible side-loads of data from other processes.
359        if (lastModified > now - GENERATION_WAIT) {
360          log.info("At least one upload is in progress ({})", fileName);
361          return true;
362        }
363      }
364    }
