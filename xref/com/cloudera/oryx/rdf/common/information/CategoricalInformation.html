28  /**
29   * Utility methods for computing information, measured in nats, over a categorical feature. This
30   * includes information gain induced by a {@link Decision} over that feature. Here, information is
31   * evaluated as Shannon entropy over the distribution of category values in a set of feature values.
32   *
33   * @author Sean Owen
34   * @see NumericalInformation
35   */
36  final class CategoricalInformation {
37    
38    private CategoricalInformation() {
39    }
40  
41    static Pair<Decision,Double> bestGain(Iterable<Decision> decisions, ExampleSet examples) {
42      int numCategories = examples.getTargetCategoryCount();
43      int[] countsNegative = new int[numCategories];
44      // Start with everything considered a negative example:
45      List<Example> exampleList = examples.getExamples();
46      int numExamples = exampleList.size();
47      for (Example example : exampleList) {
48        int category = ((CategoricalFeature) example.getTarget()).getValueID();
49        countsNegative[category]++;
50      }
51      int numNegative = numExamples;
52  
53      double entropyAll = Information.entropy(countsNegative);
54  
55      BitSet notYetPositiveExamples = new BitSet(numExamples);
56      notYetPositiveExamples.set(0, numExamples);
57      Decision bestDecision = null;
58      double bestGain = Double.NEGATIVE_INFINITY;
59      int[] countsPositive = new int[numCategories];
60      int numPositive = 0;
61  
62      for (Decision decision : decisions) {
63        int nextNotYetPositive = -1;
64        while ((nextNotYetPositive = notYetPositiveExamples.nextSetBit(nextNotYetPositive + 1)) >= 0) {
65          Example example = exampleList.get(nextNotYetPositive);
66          if (decision.isPositive(example)) {
67            int category = ((CategoricalFeature) example.getTarget()).getValueID();
68            countsNegative[category]--;
69            numNegative--;
70            countsPositive[category]++;
71            numPositive++;
72            notYetPositiveExamples.clear(nextNotYetPositive);
73          }
74        }
75        double entropyNegative = Information.entropy(countsNegative);
76        double entropyPositive = Information.entropy(countsPositive);
77        double gain = entropyAll -
78            (numNegative * entropyNegative + numPositive * entropyPositive) / (numNegative + numPositive);
79        if (gain > bestGain) {
80          bestGain = gain;
81          bestDecision = decision;
82        }
83      }
84  
85      return bestDecision == null ? null : new Pair<Decision,Double>(bestDecision, bestGain);
86    }
