16  package com.cloudera.oryx.rdf.common.information;
17  
18  import org.apache.commons.math3.util.Pair;
19  
20  import com.cloudera.oryx.rdf.common.example.ExampleSet;
21  import com.cloudera.oryx.rdf.common.rule.Decision;
22  
23  /**
24   * Contains utility methods for working with entropy and information, measured in nats.
25   *
26   * @author Sean Owen
27   * @see NumericalInformation
28   * @see CategoricalInformation
29   */
30  public final class Information {
31  
32    private Information() {
33    }
34  
35    /**
36     * @param decisions list of possible {@link Decision}s to consider to split the examples in {@link ExampleSet}
37     * @param examples {@link ExampleSet} of examples to evaluate {@link Decision}s over
38     * @return the best {@link Decision} from {@code decisions} -- the one that maximizes information gain --
39     *  along with the information gain it achieves, in nats
40     */
41    public static Pair<Decision,Double> bestGain(Iterable<Decision> decisions, ExampleSet examples) {
42      switch (examples.getTargetType()) {
43        case NUMERIC:
44          return NumericalInformation.bestGain(decisions, examples);
45        case CATEGORICAL:
46          return CategoricalInformation.bestGain(decisions, examples);
47        default:
48          throw new IllegalStateException();
49      }
50    }
51  
52    /**
53     * @param counts counts of distinct categories in a sample
54     * @return entropy, in nats, of those counts
55     */
56    public static double entropy(int[] counts) {
57      // Entropy is really, over all counts:
58      //  sum (-p_i * ln p_i)
59      // where p_i = count_i / total
60      // terms where count is 0 or 1 are 0, so can go away
61      // Here we actually compute total as we go and account for it later, which avoids
62      // a second loop and avoids some divisions.
63      // This means we divide by total at the end to account for the missing denominator in the first p_i term,
64      // and end up adding (subtracting negative) ln total at the end to account for ln p_i.
65      double entropy = 0.0;
66      int total = 0;
67      for (int count : counts) {
68        // if count = 0 or count = 1 then count*ln(count) = 0
69        if (count > 1) {
70          // This is in nats to match differential entropy -- base e, not 2
71          entropy -= count * Math.log(count);
72        }
73        total += count;
74      }
75      return entropy / total + Math.log(total);
76    }
77  
78  }
