16  package com.cloudera.oryx.rdf.common.eval;
17  
18  import java.util.List;
19  
20  import com.cloudera.oryx.common.stats.DoubleWeightedMean;
21  import com.cloudera.oryx.rdf.common.rule.CategoricalPrediction;
22  import com.cloudera.oryx.rdf.common.rule.NumericPrediction;
23  import com.cloudera.oryx.rdf.common.rule.Prediction;
24  import com.google.common.base.Preconditions;
25  
26  /**
27   * A utility class with methods for combining the results of many
28   * {@link com.cloudera.oryx.rdf.common.tree.TreeBasedClassifier}s into one prediction -- in particular, it is used by
29   * {@link com.cloudera.oryx.rdf.common.tree.DecisionForest} to combine the  results of many
30   * {@link com.cloudera.oryx.rdf.common.tree.DecisionTree}s. {@link Prediction}s from many
31   * {@link com.cloudera.oryx.rdf.common.tree.TreeBasedClassifier}s are combined and weighted according to the accuracy of
32   * the individuals.
33   *
34   * @author Sean Owen
35   */
36  public final class WeightedPrediction {
37  
38    private WeightedPrediction() {
39    }
40  
41    /**
42     * @param predictions {@link Prediction}s from individuals
43     * @param weights weights that should be applied to
44     * @return a single {@link Prediction} represented a weighted combination of the inputs
45     */
46    public static <T extends Prediction> Prediction voteOnFeature(List<T> predictions, double[] weights) {
47      Preconditions.checkArgument(!predictions.isEmpty(), "No predictions");
48      Preconditions.checkArgument(predictions.size() == weights.length,
49                                  "%s predictions but %s weights?", predictions.size(), weights.length);
50      switch (predictions.get(0).getFeatureType()) {
51        case NUMERIC:
52          @SuppressWarnings("unchecked")
53          List<NumericPrediction> numericVotes = (List<NumericPrediction>) predictions;
54          return voteOnNumericFeature(numericVotes, weights);
55        case CATEGORICAL:
56          @SuppressWarnings("unchecked")
57          List<CategoricalPrediction> categoricalVotes = (List<CategoricalPrediction>) predictions;
58          return voteOnCategoricalFeature(categoricalVotes, weights);
59        default:
60          throw new IllegalStateException();
61      }
62    }
63    
64    private static Prediction voteOnCategoricalFeature(List<CategoricalPrediction> predictions, double[] weights) {
65      float[] weightedProbabilities = null;
66      double totalWeight = 0.0;
67      for (int i = 0; i < predictions.size(); i++) {
68        CategoricalPrediction vote = predictions.get(i);
69        double weight = weights[i];
70        totalWeight += weight;
71        float[] categoryProbabilities = vote.getCategoryProbabilities();
72        if (weightedProbabilities == null) {
73          weightedProbabilities = new float[categoryProbabilities.length];
74        }
75        for (int j = 0; j < weightedProbabilities.length; j++) {
76          weightedProbabilities[j] += categoryProbabilities[j] * weight;
77        }
78      }
79      for (int j = 0; j < weightedProbabilities.length; j++) {
80        weightedProbabilities[j] /= totalWeight;
81      }
82      return new CategoricalPrediction(weightedProbabilities);
83    }
84    
85    private static Prediction voteOnNumericFeature(List<NumericPrediction> predictions, double[] weights) {
86      DoubleWeightedMean mean = new DoubleWeightedMean();
87      for (int i = 0; i < predictions.size(); i++) {
88        mean.increment(predictions.get(i).getPrediction(), weights[i]);
89      }
90      return new NumericPrediction((float) mean.getResult(), (int) mean.getN());
91    }
92  
93  }
