38   */
39  public abstract class AbstractRDFServlet extends AbstractOryxServlet {
40  
41    private static final String KEY_PREFIX = AbstractRDFServlet.class.getName();
42    public static final String GENERATION_MANAGER_KEY = KEY_PREFIX + ".GENERATION_MANAGER";
43  
44    private RDFGenerationManager generationManager;
45    private InboundSettings inboundSettings;
46    private int totalColumns;
47  
48    @Override
49    public void init(ServletConfig config) throws ServletException {
50      super.init(config);
51      ServletContext context = config.getServletContext();
52      generationManager = (RDFGenerationManager) context.getAttribute(GENERATION_MANAGER_KEY);
53  
54      inboundSettings = InboundSettings.create(ConfigUtils.getDefaultConfig());
55      totalColumns = inboundSettings.getColumnNames().size();
56    }
57  
58    final RDFGenerationManager getGenerationManager() {
59      return generationManager;
60    }
61  
62    final InboundSettings getInboundSettings() {
63      return inboundSettings;
64    }
65  
66    final int getTotalColumns() {
67      return totalColumns;
68    }
69  
70    final Feature buildFeature(int columnNumber,
71                               String token,
72                               Map<Integer, BiMap<String, Integer>> columnToCategoryNameToIDMapping) {
73      if (token == null || token.isEmpty()) {
74        return null;
75      }
76      InboundSettings inboundSettings = getInboundSettings();
77      if (inboundSettings.isNumeric(columnNumber)) {
78        return NumericFeature.forValue(Float.parseFloat(token));
79      }
80      if (inboundSettings.isCategorical(columnNumber)) {
81        return CategoricalFeature.forValue(categoricalFromString(columnNumber, token, columnToCategoryNameToIDMapping));
82      }
83      return IgnoredFeature.INSTANCE;
84    }
85  
86    private static int categoricalFromString(int columnNumber,
87                                             String value,
88                                             Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping) {
89      BiMap<String,Integer> categoryNameToID = columnToCategoryNameToIDMapping.get(columnNumber);
90      if (categoryNameToID == null) {
91        categoryNameToID = HashBiMap.create();
92        columnToCategoryNameToIDMapping.put(columnNumber, categoryNameToID);
93      }
94      Integer mapped = categoryNameToID.get(value);
95      if (mapped != null) {
96        return mapped;
97      }
98      int newCategory = categoryNameToID.size();
99      categoryNameToID.put(value, newCategory);
100     return newCategory;
101   }
