16  package com.cloudera.oryx.kmeans.common;
17  
18  import org.apache.commons.math3.random.RandomGenerator;
19  
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import com.cloudera.oryx.common.random.RandomManager;
24  
25  public final class WeightedSampler<T, W extends Weighted<T>> {
26  
27    private final double[] cumulativeSum;
28    private final List<W> things;
29    private final RandomGenerator random;
30    
31    public WeightedSampler(List<W> things, RandomGenerator random) {
32      this.things = things;
33      this.cumulativeSum = new double[things.size() + 1];
34      for (int i = 0; i < things.size(); i++) {
35        cumulativeSum[i + 1] = cumulativeSum[i] + things.get(i).weight(); 
36      }
37      this.random = (random == null) ? RandomManager.getRandom() : random;
38    }
39   
40    public T sample() {
41      double offset = random.nextDouble() * cumulativeSum[cumulativeSum.length - 1];
42      int next = Arrays.binarySearch(cumulativeSum, offset);
43      Weighted<T> item = (next >= 0) ? things.get(next - 1) : things.get(-2 - next);
44      return item.thing();
45    }
46  }
