39    public static ClusterValidityStatistics create(Iterable<WeightedRealVector> points,
40                                                   Centers test,
41                                                   Centers train,
42                                                   int replicaId) {
43      Preconditions.checkArgument(test.size() == train.size());
44      int k = test.size();
45      double n = 0.0;
46      double testCost = 0.0;
47      double trainCost = 0.0;
48      RealMatrix contingencyMatrix = new Array2DRowRealMatrix(k, k);
49      double[] rowSums = new double[k];
50      double[] colSums = new double[k];
51      for (WeightedRealVector wv : points) {
52        Distance d1 = test.getDistance(wv.thing());
53        Distance d2 = train.getDistance(wv.thing());
54        contingencyMatrix.addToEntry(d1.getClosestCenterId(), d2.getClosestCenterId(), wv.weight());
55        rowSums[d1.getClosestCenterId()] += wv.weight();
56        colSums[d2.getClosestCenterId()] += wv.weight();
57        testCost += d1.getSquaredDistance() * wv.weight();
58        trainCost += d2.getSquaredDistance() * wv.weight();
59        n += wv.weight();
60      }
61  
62      return new ClusterValidityStatistics(k, replicaId, testCost, trainCost,
63          normVarInformation(contingencyMatrix, rowSums, colSums, n),
64          normVanDongen(contingencyMatrix, rowSums, colSums, n));
65    }

67    public static ClusterValidityStatistics parse(String str) {
68      String[] pieces = DelimitedDataUtils.decode(str);
69      Preconditions.checkArgument(pieces.length == 6, "Invalid delimited string: " + str);
70      return new ClusterValidityStatistics(
71          Integer.valueOf(pieces[0]),
72          Integer.valueOf(pieces[1]),
73          Double.valueOf(pieces[2]),
74          Double.valueOf(pieces[3]),
75          Double.valueOf(pieces[4]),

122   /**
123    * Calculates the normalized van Dongen criterion for the contingency contingencyMatrix.
124    *
125    * @return the normalized van Dongen criterion for the contingency contingencyMatrix
126    */
127   private static double normVanDongen(RealMatrix contingencyMatrix, double[] rowSums, double[] colSums, double n) {
128     double rs = 0.0;
129     double cs = 0.0;
130     double rmax = 0.0;
131     double cmax = 0.0;
132     for (int i = 0; i < rowSums.length; i++) {
133       rs += contingencyMatrix.getRowVector(i).getLInfNorm();
134       cs += contingencyMatrix.getColumnVector(i).getLInfNorm();
135       rmax = Math.max(rmax, rowSums[i]);
136       cmax = Math.max(cmax, colSums[i]);
137     }
138     double den = 2 * n - rmax - cmax;
139     if (den == 0.0) {
140       return Double.NaN;
141     }
142     return (2 * n - rs - cs) / den;
143   }

145   /**
146    * Calculates the normalized variation-of-information for the contingency contingencyMatrix.
147    *
148    * @return the normalized variation-of-information for the contingency contingencyMatrix
149    */
150   private static double normVarInformation(
151       RealMatrix contingencyMatrix,
152       final double[] rowSums,
153       final double[] colSums,
154       final double n) {
155     double den = n * (entropy(rowSums, n) + entropy(colSums, n));
156     if (den == 0) {
157       return Double.NaN;
158     }
159 
160     double num = contingencyMatrix.walkInOptimizedOrder(new RealMatrixPreservingVisitor() {
161       private double sum = 0.0;
162       @Override
163       public void start(int rows, int columns, int startRow, int endRow, int startColumn, int endColumn) {
164         sum = 0.0;
165       }
166 
167       @Override
168       public void visit(int row, int column, double value) {
169         if (value > 0.0) {
170           sum += value * (Math.log(value * n) - Math.log(rowSums[row]) - Math.log(colSums[column]));
171         }
172       }
173 
174       @Override
175       public double end() {
176         return sum;
177       }
178     });
179 
180     return 1.0 + 2.0 * (num / den);
181   }
182 
183   private static double entropy(double[] sums, double n) {
184     double e = 0.0;
185     double logn = Math.log(n);
186     for (double sum : sums) {
187       if (sum > 0.0) {
188         e += sum * (Math.log(sum) - logn);
189       }
190     }
191     return e / n;
192   }
