21  import org.apache.commons.math3.linear.RealVector;

23  import com.google.common.base.Function;
24  import com.google.common.collect.Lists;
25  import org.apache.commons.math3.random.RandomGenerator;

27  /**
28   * The strategies used to choose the initial {@link Centers} used for a k-means algorithm
29   * run prior to running Lloyd's algorithm.
30   */
31  public enum KMeansInitStrategy {
32  
33    /**
34     * Uses the classic random selection strategy to create the initial {@code Centers}. The
35     * algorithm will randomly choose K points from the input points, favoring points with
36     * higher weights.
37     */
38    RANDOM {
39      @Override
40      public <W extends Weighted<RealVector>> Centers apply(List<W> points,
41                                                            int numClusters,
42                                                            RandomGenerator random) {
43        return new Centers(
44            Lists.transform(Weighted.sample(points, numClusters, random), new Function<W, RealVector>() {
45              @Override
46              public RealVector apply(W wt) {
47                return wt.thing();
48              }
49            }));
50      }
51    },
52    
53    /**
54     * Uses the <i>k-means++</i> strategy described in Arthur and Vassilvitskii (2007).
55     * See <a href="http://en.wikipedia.org/wiki/K-means%2B%2B">the Wikipedia page</a>
56     * for details.
57     */
58    PLUS_PLUS {
59      @Override
60      public <W extends Weighted<RealVector>> Centers apply(List<W> points,
61                                                  int numClusters,
62                                                  RandomGenerator random) {
63        Centers centers = RANDOM.apply(points, 1, random);
64        double[] cumulativeScores = new double[points.size() + 1];
65        for (int i = 1; i < numClusters; i++) {
66          cumulativeScores[0] = 0;
67          for (int j = 0; j < points.size(); j++) {
68            W wv = points.get(j);
69            double score = centers.getDistance(wv.thing()).getSquaredDistance() * wv.weight();
70            cumulativeScores[j + 1] = cumulativeScores[j] + score;
71          }
72          double r = cumulativeScores[points.size()] * random.nextDouble();
73          int next = Arrays.binarySearch(cumulativeScores, r);
74          int index = (next > 0) ? next - 1 : -2 - next;
75          while (index > 0 && centers.contains(points.get(index).thing())) {
76            index--;
77          }
78          centers = centers.extendWith(points.get(index).thing());
79        }
80        return centers;
81      }
82    };
83  
84    /**
85     * Use this instance to create the initial {@code Centers} from the given parameters.
86     * 
87     * @param points The candidate {@code WeightedVec} instances for the cluster
88     * @param numClusters The number of points in the center (i.e., the "k" in "k-means")
89     * @param randomGenerator The {@code RandomGenerator} instance to use
90     * @return A new {@code Centers} instance created using this instance
91     */
92    public abstract <W extends Weighted<RealVector>> Centers apply(List<W> points, int numClusters,
93                                                         RandomGenerator randomGenerator);
94  }
