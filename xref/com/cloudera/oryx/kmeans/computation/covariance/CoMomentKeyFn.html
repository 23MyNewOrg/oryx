15  package com.cloudera.oryx.kmeans.computation.covariance;
16  
17  import com.cloudera.oryx.computation.common.fn.OryxDoFn;
18  import com.google.common.collect.Maps;
19  import org.apache.commons.math3.linear.RealVector;
20  import org.apache.crunch.Emitter;
21  import org.apache.crunch.Pair;
22  import org.apache.crunch.types.PType;
23  
24  import java.util.Map;
25  
26  final class CoMomentKeyFn<K> extends OryxDoFn<Pair<K, Pair<RealVector, Double>>, Pair<Pair<K, Index>, CoMoment>> {
27    private final Map<K, CoMomentTracker> trackers = Maps.newHashMap();
28    private final PType<K> keyType;
29  
30    CoMomentKeyFn(PType<K> keyType) {
31      this.keyType = keyType;
32    }
33  
34    @Override
35    public void initialize() {
36      super.initialize();
37      trackers.clear();
38      keyType.initialize(getConfiguration());
39    }
40  
41    @Override
42    public void process(Pair<K, Pair<RealVector, Double>> input, Emitter<Pair<Pair<K, Index>, CoMoment>> emitter) {
43      CoMomentTracker tracker = trackers.get(input.first());
44      if (tracker == null) {
45        tracker = new CoMomentTracker();
46        trackers.put(keyType.getDetachedValue(input.first()), tracker);
47      }
48      tracker.update(input.second().first());
49    }
50  
51    @Override
52    public void cleanup(Emitter<Pair<Pair<K, Index>, CoMoment>> emitter) {
53      for (Map.Entry<K, CoMomentTracker> e : trackers.entrySet()) {
54        for (Map.Entry<Index, CoMoment> f : e.getValue().entrySet()) {
55          emitter.emit(Pair.of(Pair.of(e.getKey(), f.getKey()), f.getValue()));
56        }
57      }
58    }
59  }
