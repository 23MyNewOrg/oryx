37  public class KMeansJobStep extends JobStep {
38  
39    @Override
40    protected final KMeansJobStepConfig getConfig() {
41      return (KMeansJobStepConfig) super.getConfig();
42    }
43  
44    @Override
45    protected final JobStepConfig parseConfig(String[] args) {
46      Preconditions.checkNotNull(args);
47      Preconditions.checkArgument(args.length >= 4);
48      return new KMeansJobStepConfig(args[0],
49          Long.parseLong(args[1]),
50          Long.parseLong(args[2]),
51          Integer.parseInt(args[3]));
52    }
53  
54    protected final <V extends RealVector> PCollection<Pair<Integer, V>> inputPairs(
55        Pipeline p,
56        String inputKey,
57        PType<V> ptype) {
58      PType<Pair<Integer, V>> inputType = Avros.pairs(Avros.ints(), ptype);
59      return p.read(avroInput(inputKey, inputType));
60    }
61  
62    protected final <V extends RealVector> PCollection<V> inputVectors(Pipeline p, String inputKey, PType<V> ptype) {
63      return PTables.asPTable(inputPairs(p, inputKey, ptype)).values();
64    }
65  
66    protected final KSketchIndex getCentersIndex(String prefix) throws IOException {
67      return (new CentersIndexLoader(ClusterSettings.create(ConfigUtils.getDefaultConfig()))).load(prefix);
68    }
69  
70    protected static PCollection<Record> toRecords(PCollection<String> lines) {
71      return StringSplitFn.apply(lines);
72    }
73  }
