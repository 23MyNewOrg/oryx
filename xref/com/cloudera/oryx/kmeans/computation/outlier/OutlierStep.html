35  import java.io.IOException;

37  public final class OutlierStep extends KMeansJobStep {
38    @Override
39    protected MRPipeline createPipeline() throws IOException {
40      JobStepConfig stepConfig = getConfig();
41  
42      OutlierSettings settings = OutlierSettings.create();
43      String instanceDir = stepConfig.getInstanceDir();
44      int generationID = stepConfig.getGenerationID();
45      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
46      String outputKey = prefix + "outliers/";
47      if (!validOutputPath(outputKey)) {
48        return null;
49      }
50  
51      String covarianceKey = prefix + "covariance/";
52      if (!validOutputPath(covarianceKey)) {
53        covarianceKey = null;
54      }
55  
56      // For the output data
57      RecordSpec.Builder rsb = RecordSpec.builder()
58          .add("vector_id", DataType.STRING)
59          .add("k", DataType.INT)
60          .add("closest_center_id", DataType.INT)
61          .add("center_distance", DataType.DOUBLE);
62      if (covarianceKey != null) {
63        rsb.add("euclidean_distance", DataType.DOUBLE);
64        rsb.add("outlier_distance", DataType.DOUBLE);
65      }
66      Spec spec = rsb.build();
67  
68      KSketchIndex index = getCentersIndex(prefix);
69      int dim = index.getDimension();
70  
71      String inputKey = prefix + "normalized/";
72      MRPipeline p = createBasicPipeline(OutlierScoreFn.class);
73      inputVectors(p, inputKey, MLAvros.namedVector())
74          .parallelDo(
75              "assign",
76               new AssignFn<NamedRealVector>(index, settings.useApprox()),
77               Avros.tableOf(Avros.reflects(ClusterKey.class), Avros.pairs(MLAvros.namedVector(), Avros.doubles())))
78          .parallelDo(
79              "toRecords",
80              new OutlierScoreFn(spec, covarianceKey, dim),
81              MLRecords.csvRecord(AvroTypeFamily.getInstance(), String.valueOf(DelimitedDataUtils.DELIMITER)))
82          .write(compressedTextOutput(p.getConfiguration(), outputKey));
83  
84      return p;
85    }
86  }
