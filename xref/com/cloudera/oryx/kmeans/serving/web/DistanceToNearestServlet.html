30  /**
31   * <p>Responsds to a GET request to {@code /distanceToNearest/[datum]}. The input is one data point to cluster,
32   * delimited, like "1,-4,3.0". The response body contains the distance to the nearest cluster, on one line.</p>
33   *
34   * @author Sean Owen
35   */
36  public final class DistanceToNearestServlet extends AbstractKMeansServlet {
37  
38    @Override
39    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
40      CharSequence pathInfo = request.getPathInfo();
41      if (pathInfo == null) {
42        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No path");
43        return;
44      }
45      String line = pathInfo.subSequence(1, pathInfo.length()).toString();
46  
47      Generation generation = getGenerationManager().getCurrentGeneration();
48      if (generation == null) {
49        response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE);
50        return;
51      }
52  
53      String[] tokens = DelimitedDataUtils.decode(line);
54      RealVector vector = generation.toVector(tokens);
55      if (vector == null) {
56        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Wrong column count");
57        return;
58      }
59      double distance = findClosest(generation, vector).getSquaredDistance();
60      response.getWriter().write(Double.toString(distance));
61    }
62  
63    static Distance findClosest(Generation generation, RealVector parsedVector) {
64      Centers centroids = generation.getCentroids();
65      Preconditions.checkState(centroids.size() > 0);
66      return centroids.getDistance(parsedVector);
67    }
68  }
