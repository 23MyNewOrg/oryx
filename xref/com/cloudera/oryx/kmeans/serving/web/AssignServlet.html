16  package com.cloudera.oryx.kmeans.serving.web;

18  import javax.servlet.http.HttpServletRequest;
19  import javax.servlet.http.HttpServletResponse;
20  import java.io.IOException;
21  
22  import com.cloudera.oryx.common.io.DelimitedDataUtils;
23  import com.cloudera.oryx.kmeans.serving.generation.Generation;
24  import org.apache.commons.math3.linear.RealVector;
25  
26  /**
27   * <p>Responsds to a GET request to {@code /assign/[datum]}. The input is one data point to cluster,
28   * delimited, like "1,-4,3.0". The response body contains the ID of the nearest cluster, on one line.</p>
29   *
30   * @author Sean Owen
31   */
32  public final class AssignServlet extends AbstractKMeansServlet {
33  
34    @Override
35    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
36      CharSequence pathInfo = request.getPathInfo();
37      //System.out.println("PATH INFO = " + pathInfo);
38      if (pathInfo == null) {
39        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No path");
40        return;
41      }
42      String line = pathInfo.subSequence(1, pathInfo.length()).toString();
43      //System.out.println("LINE = " + line);
44      Generation generation = getGenerationManager().getCurrentGeneration();
45      if (generation == null) {
46        response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE);
47        return;
48      }
49  
50      RealVector vec = generation.toVector(DelimitedDataUtils.decode(line));
51      if (vec == null) {
52        //System.out.println("Bad column count");
53        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Wrong column count");
54        return;
55      }
56  
57      int assignment = DistanceToNearestServlet.findClosest(generation, vec).getClosestCenterId();
58      response.getWriter().write(Integer.toString(assignment));
59    }
60  
61  }
