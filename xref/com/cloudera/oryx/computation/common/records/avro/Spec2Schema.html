16  package com.cloudera.oryx.computation.common.records.avro;
17  
18  import java.util.List;
19  
20  import com.cloudera.oryx.computation.common.records.Spec;
21  import org.apache.avro.Schema;
22  
23  import com.cloudera.oryx.computation.common.records.FieldSpec;
24  import com.google.common.collect.Lists;
25  
26  public final class Spec2Schema {
27  
28    public static Schema create(Spec spec) {
29      switch (spec.getDataType()) {
30        case INT:
31          return Schema.create(Schema.Type.INT);
32        case LONG:
33          return Schema.create(Schema.Type.LONG);
34        case DOUBLE:
35          return Schema.create(Schema.Type.DOUBLE);
36        case BOOLEAN:
37          return Schema.create(Schema.Type.BOOLEAN);
38        case STRING:
39          return Schema.create(Schema.Type.STRING);
40        case RECORD:
41          List<Schema.Field> fields = Lists.newArrayList();
42          for (int i = 0; i < spec.size(); i++) {
43            FieldSpec f = spec.getField(i);
44            fields.add(new Schema.Field(f.name(), create(f.spec()), "", null));
45          }
46          Schema s = Schema.createRecord("R" + spec.hashCode(), "", "", false);
47          s.setFields(fields);
48          return s;
49      }
50      return null;
51    }
52    
53    private Spec2Schema() {}
54  }
