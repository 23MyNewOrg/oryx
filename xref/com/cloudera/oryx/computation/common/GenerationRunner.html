85    /**
86     * @return current state of execution at a {@link GenerationRunnerState}, or {@code null} if no job has been
87     *  started
88     */
89    public final GenerationRunnerState getState() throws IOException, InterruptedException {
90      if (getGenerationID() < 0) {
91        return null;
92      }
93      List<StepState> stepStates = Lists.newArrayList();
94      for (HasState state : stateSources) {
95        stepStates.addAll(state.getStepStates());
96      }
97      return new GenerationRunnerState(getGenerationID(), stepStates, isRunning, startTime, endTime);
98    }

100   /**
101    * Overall entry point -- handles initial concerns like establishing state and checking for running jobs.
102    */
103   @Override
104   public final Void call() throws IOException, JobException, InterruptedException {
105     isRunning = true;
106     startTime = new Date();
107     try {
108       waitForJobAlreadyRunning(instanceDir);
109       log.info("Starting run for instance {}", instanceDir);
110       runGeneration();
111       return null;
112     } finally {
113       isRunning = false;
114       endTime = new Date();
115     }
116   }

123   /**
124    * Does the work of actually running one generation.
125    */
126   private void runGeneration() throws IOException, JobException, InterruptedException {
127 
128     List<String> generationStrings = StoreUtils.listGenerationsForInstance(instanceDir);
129 
130     Store store = Store.get();
131     int count = generationStrings.size();
132 
133     Config config = ConfigUtils.getDefaultConfig();
134     int maxGenerationsToKeep = config.getInt("model.generations.keep");
135     if (count > maxGenerationsToKeep) {
136       Iterable<String> generationsToDelete = generationStrings.subList(0, count - maxGenerationsToKeep);
137       log.info("Deleting old generations: {}", generationsToDelete);
138       for (String generationPrefix : generationsToDelete) {
139         store.recursiveDelete(generationPrefix);
140       }
141     }
142 
143     long lastDoneGeneration = -1;
144     int lastDoneGenerationIndex = -1;
145     for (int i = count - 1; i >= 0; i--) {
146       long generationID = StoreUtils.parseGenerationFromPrefix(generationStrings.get(i));
147       if (store.exists(Namespaces.getGenerationDoneKey(instanceDir, generationID), true)) {
148         lastDoneGeneration = generationID;
149         lastDoneGenerationIndex = i;
150         break;
151       }
152     }
153 
154     lastGenerationID = lastDoneGeneration;
155 
156     long generationToMake;
157     long generationToWaitFor;
158     long generationToRun;
159 
160     if (lastDoneGeneration >= 0) {
161 
162       log.info("Last complete generation is {}", lastDoneGeneration);
163       if (lastDoneGenerationIndex == count - 1) {
164         generationToMake = lastDoneGeneration + 1;
165         generationToRun = -1;
166         generationToWaitFor = -1;
167       } else if (lastDoneGenerationIndex == count - 2) {
168         generationToRun = StoreUtils.parseGenerationFromPrefix(generationStrings.get(lastDoneGenerationIndex + 1));
169         generationToMake = generationToRun + 1;
170         generationToWaitFor = generationToMake;
171       } else {
172         generationToRun = StoreUtils.parseGenerationFromPrefix(generationStrings.get(lastDoneGenerationIndex + 1));
173         generationToMake = -1;
174         generationToWaitFor = StoreUtils.parseGenerationFromPrefix(generationStrings.get(lastDoneGenerationIndex + 2));
175       }
176 
177     } else {
178 
179       log.info("No complete generations");
180       // If nothing is done,
181       if (generationStrings.isEmpty()) {
182         // and no generations exist, make one
183         generationToRun = -1;
184         generationToMake = 0;
185         generationToWaitFor = -1;
186       } else {
187         if (count >= 2) {
188           // Run current one and open a next generation
189           generationToRun = StoreUtils.parseGenerationFromPrefix(generationStrings.get(0));
190           generationToMake = -1;
191           generationToWaitFor = StoreUtils.parseGenerationFromPrefix(generationStrings.get(1));
192         } else {
193           generationToRun = StoreUtils.parseGenerationFromPrefix(generationStrings.get(0));
194           generationToMake = generationToRun + 1;
195           generationToWaitFor = generationToMake;
196         }
197       }
198 
199     }
200 
201     Preconditions.checkState((generationToWaitFor < 0) == (generationToRun < 0),
202                              "There must either be both a generation to wait for and generation " +
203                                  "to run, or neither");
204 
205     if (generationToRun >= 0 && !isAnyInputInGeneration(generationToRun)) {
206       log.info("No data in generation {}, so not running", generationToRun);
207       generationToRun = -1;
208       generationToWaitFor = -1;
209       generationToMake = -1;
210     }
211 
212     if (generationToMake < 0) {
213       log.info("No need to make a new generation");
214     } else {
215       log.info("Making new generation {}", generationToMake);
216       store.mkdir(Namespaces.getInstanceGenerationPrefix(instanceDir, generationToMake) + "inbound/");
217     }
218 
219     maybeWaitToRun(instanceDir, generationToWaitFor, generationToRun);
220 
221     // Check again: maybe an upload was in progress but failed!
222     if (generationToRun >= 0 && !isAnyInputInGeneration(generationToRun)) {
223       log.info("No data in generation {}, so not running", generationToRun);
224       generationToRun = -1;
225     }
226 
227     if (generationToRun < 0) {
228       log.info("No generation to run");
229     } else {
230       generationID = generationToRun;
231       log.info("Running generation {}", generationID);
232       startSize = store.getSizeRecursive(Namespaces.getInstanceGenerationPrefix(instanceDir, generationID));
233       storeConfig(config);
234       runSteps();
235       log.info("Signaling completion of generation {}", generationID);
236       store.touch(Namespaces.getGenerationDoneKey(instanceDir, generationID));
237       store.recursiveDelete(Namespaces.getTempPrefix(instanceDir, getGenerationID()));
238       log.info("Dumping some stats on generation {}", generationID);
239       endSize = store.getSizeRecursive(Namespaces.getInstanceGenerationPrefix(instanceDir, generationID));
240       dumpStats();
241     }
242   }

244     private void storeConfig(Config config) throws IOException {
245     String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "computation.conf";
246     String configString = config.root().render(ConfigRenderOptions.concise());
247     Writer writer = new OutputStreamWriter(Store.get().streamTo(outputKey), Charsets.UTF_8);
248     try {
249       writer.write(configString);
250     } finally {
251       writer.close();
252     }

255   private void dumpStats() throws IOException {
256     // First compute the base stats for all GenerationRunners
257     Map<String, Object> stats = Maps.newHashMap();
258     stats.put("preRunBytes", startSize);
259     stats.put("postRunBytes", endSize);
260 
261     // Now compute some subclass-specific stats if impelmented
262     Map<String, ?> moreStats = collectStats();
263     if (moreStats != null) {
264       stats.putAll(moreStats);
265     }
266 
267     // Write to disk as JSON
268     ObjectMapper mapper = JacksonUtils.getObjectMapper();
269     String statsString = mapper.writeValueAsString(stats);
270     String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "stats.json";
271     Writer writer = new OutputStreamWriter(Store.get().streamTo(outputKey), Charsets.UTF_8);
272     try {
273       writer.write(statsString);
274     } finally {
275       writer.close();
276     }
277   }

284   private static void maybeWaitToRun(String instanceDir,
285                                      long generationToWaitFor,
286                                      long generationToRun) throws IOException, InterruptedException {
287     if (generationToWaitFor < 0) {
288       log.info("No need to wait for a generation");
289       return;
290     }
291 
292     Store store = Store.get();
293 
294     if (generationToRun == 0L) {
295       // Special case: let generation 0 run immediately
296       log.info("Generation 0 may run immediately", generationToWaitFor);
297     } else {
298       String nextGenerationPrefix =
299           Namespaces.getInstanceGenerationPrefix(instanceDir, generationToWaitFor) + "inbound/";
300       long lastModified = store.getLastModified(nextGenerationPrefix);
301       long now = System.currentTimeMillis();
302       if (now > lastModified + GENERATION_WAIT) {
303         log.info("Generation {} is old enough to proceed", generationToWaitFor);
304       } else {
305         // Not long enough, wait
306         long toSleepMS = lastModified + GENERATION_WAIT - now;
307         log.info("Waiting {}s for data to start uploading to generation {} and then move to {}...",
308                  TimeUnit.SECONDS.convert(toSleepMS, TimeUnit.MILLISECONDS),
309                  generationToRun,
310                  generationToWaitFor);
311         Thread.sleep(toSleepMS);
312       }
313     }
314 
315     String uploadingGenerationPrefix =
316         Namespaces.getInstanceGenerationPrefix(instanceDir, generationToRun) + "inbound/";
317 
318     boolean anyInProgress = true;
319     while (anyInProgress) {
320       log.info("Waiting for uploads to finish in {}...", uploadingGenerationPrefix);
321       anyInProgress = false;
322       for (String fileName : store.list(uploadingGenerationPrefix, true)) {
323         // .inprogress is our marker for uploading files; _COPYING_ is Hadoop's
324         if (fileName.endsWith(".inprogress") || fileName.endsWith("_COPYING_")) {
325           long inProgressLastModified = store.getLastModified(fileName);
326           if (System.currentTimeMillis() < inProgressLastModified + 5 * GENERATION_WAIT) {
327             log.info("At least one upload is in progress ({})", fileName);
328             anyInProgress = true;
329             break;
330           } else {
331             log.warn("Stale upload to {}? Deleting and continuing", fileName);
332             try {
333               store.delete(fileName);
334             } catch (IOException ioe) {
335               log.info("Could not delete {}", fileName, ioe);
336             }
337           }
338         }
339       }
340       if (anyInProgress) {
341         Thread.sleep(TimeUnit.MILLISECONDS.convert(1, TimeUnit.MINUTES));
342       }
343     }
344

347   private boolean isAnyInputInGeneration(long generationID) throws IOException {
348     Store store = Store.get();
349     for (String filePrefix :
350          store.list(Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "inbound/", true)) {
351       if (store.getSize(filePrefix) > 0) {
352         return true;
353       }
354     }
355     return false;
356   }

358   protected final int readLatestIterationInProgress() throws IOException {
359     String iterationsPrefix = Namespaces.getIterationsPrefix(instanceDir,  getGenerationID());
360     List<String> iterationPaths = Store.get().list(iterationsPrefix, false);
361     if (iterationPaths == null || iterationPaths.isEmpty()) {
362       return 1;
363     }
364     String iterationString = StoreUtils.lastNonEmptyDelimited(iterationPaths.get(iterationPaths.size() - 1));
365     int iteration = Integer.parseInt(iterationString);
366     if (iteration == 0) {
367       // Iteration 0 is a fake iteration with initial Y; means we're really on 1
368       iteration = 1;
369     }
370     return iteration;
371   }
