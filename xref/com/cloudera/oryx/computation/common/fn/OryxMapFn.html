16  package com.cloudera.oryx.computation.common.fn;
17  
18  import com.cloudera.oryx.common.servcomp.OryxConfiguration;
19  import com.cloudera.oryx.common.settings.ConfigUtils;
20  import com.cloudera.oryx.computation.common.JobStep;
21  
22  import org.apache.crunch.MapFn;
23  import org.apache.hadoop.conf.Configuration;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  public abstract class OryxMapFn<S, T> extends MapFn<S, T> {
28    private static final Logger log = LoggerFactory.getLogger(OryxMapFn.class);
29  
30    private Configuration configuration;
31  
32    @Override
33    public final Configuration getConfiguration() {
34      return configuration;
35    }
36  
37    @Override
38    public void initialize() {
39      super.initialize();
40      Configuration rawConfiguration = super.getConfiguration();
41      log.info("Setup of {} with config {}", this, rawConfiguration);
42      this.configuration = OryxConfiguration.get(rawConfiguration);
43      ConfigUtils.overlayConfigOnDefault(configuration.get(JobStep.CONFIG_SERIALIZATION_KEY));
44    }
45  
46    @Override
47    public final String toString() {
48      return getClass().getSimpleName();
49    }
50  }
