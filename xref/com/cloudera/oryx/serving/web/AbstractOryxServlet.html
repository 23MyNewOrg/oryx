16  package com.cloudera.oryx.serving.web;
17  
18  import java.io.IOException;
19  import java.util.Map;
20  import javax.servlet.ServletConfig;
21  import javax.servlet.ServletContext;
22  import javax.servlet.ServletException;
23  import javax.servlet.http.HttpServlet;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import com.google.common.collect.Maps;
28  
29  import com.cloudera.oryx.serving.stats.ServletStats;
30  
31  /**
32   * Superclass of {@link HttpServlet}s used in the application. All API methods return the following 
33   * HTTP statuses in certain situations:
34   *
35   * <ul>
36   *  <li>{@code 400 Bad Request} if the arguments are invalid</li>
37   *  <li>{@code 401 Unauthorized} if a username/password is required, but not supplied correctly 
38   *  in the request via HTTP DIGEST</li>
39   *  <li>{@code 405 Method Not Allowed} if an incorrect HTTP method is used, like {@code GET} 
40   *  where {@code POST} is required</li>
41   *  <li>{@code 500 Internal Server Error} if an unexpected server-side exception occurs</li>
42   *  <li>{@code 503 Service Unavailable} if not yet available to serve requests</li>
43   * </ul>
44   *
45   * @author Sean Owen
46   */
47  public abstract class AbstractOryxServlet extends HttpServlet {
48  
49    private static final String KEY_PREFIX = AbstractOryxServlet.class.getName();
50    public static final String TIMINGS_KEY = KEY_PREFIX + ".TIMINGS";
51  
52    private ServletStats timing;
53  
54    @Override
55    public void init(ServletConfig config) throws ServletException {
56      super.init(config);
57  
58      ServletContext context = config.getServletContext();
59  
60      Map<String,ServletStats> timings;
61      synchronized (context) {
62        @SuppressWarnings("unchecked")
63        Map<String,ServletStats> temp = (Map<String,ServletStats>) context.getAttribute(TIMINGS_KEY);
64        timings = temp;
65        if (timings == null) {
66          timings = Maps.newTreeMap();
67          context.setAttribute(TIMINGS_KEY, timings);
68        }
69      }
70  
71      String key = getClass().getSimpleName();
72      ServletStats theTiming = timings.get(key);
73      if (theTiming == null) {
74        theTiming = new ServletStats();
75        timings.put(key, theTiming);
76      }
77      timing = theTiming;
78    }
79  
80    @Override
81    protected final void service(HttpServletRequest request, HttpServletResponse response)
82        throws ServletException, IOException {
83  
84      long start = System.nanoTime();
85      super.service(request, response);
86      timing.addTimingNanosec(System.nanoTime() - start);
87  
88      int status = response.getStatus();
89      if (status >= 400) {
90        if (status >= 500) {
91          timing.incrementServerErrors();
92        } else {
93          timing.incrementClientErrors();
94        }
95      }
96    }
97  
98  }
