39  /**
40   * Superclass of {@link HttpServlet}s used in the application. All API methods return the following
41   * HTTP statuses in certain situations:
42   *
43   * <ul>
44   *  <li>{@code 400 Bad Request} if the arguments are invalid</li>
45   *  <li>{@code 401 Unauthorized} if a username/password is required, but not supplied correctly 
46   *  in the request via HTTP DIGEST</li>
47   *  <li>{@code 405 Method Not Allowed} if an incorrect HTTP method is used, like {@code GET} 
48   *  where {@code POST} is required</li>
49   *  <li>{@code 500 Internal Server Error} if an unexpected server-side exception occurs</li>
50   *  <li>{@code 503 Service Unavailable} if not yet available to serve requests</li>
51   * </ul>
52   *
53   * @author Sean Owen
54   */
55  public abstract class AbstractOryxServlet extends HttpServlet {
56  
57    private static final String KEY_PREFIX = AbstractOryxServlet.class.getName();
58    public static final String TIMINGS_KEY = KEY_PREFIX + ".TIMINGS";
59  
60    private static final Pattern ESCAPED_SLASH = Pattern.compile("%2F", Pattern.CASE_INSENSITIVE);
61    protected static final Splitter SLASH = Splitter.on('/').omitEmptyStrings();
62    private static final Splitter COMMA = Splitter.on(',').omitEmptyStrings().trimResults();
63  
64    private ServletStats timing;
65    private ConcurrentMap<String,ResponseContentType> responseTypeCache;
66  
67    @Override
68    public void init(ServletConfig config) throws ServletException {
69      super.init(config);
70  
71      ServletContext context = config.getServletContext();
72  
73      Map<String,ServletStats> timings;
74      synchronized (context) {
75        @SuppressWarnings("unchecked")
76        Map<String,ServletStats> temp = (Map<String,ServletStats>) context.getAttribute(TIMINGS_KEY);
77        timings = temp;
78        if (timings == null) {
79          timings = Maps.newTreeMap();
80          context.setAttribute(TIMINGS_KEY, timings);
81        }
82      }
83  
84      String key = getClass().getSimpleName();
85      ServletStats theTiming = timings.get(key);
86      if (theTiming == null) {
87        theTiming = new ServletStats();
88        timings.put(key, theTiming);
89      }
90      timing = theTiming;
91  
92      responseTypeCache = Maps.newConcurrentMap();
93    }
94  
95    @Override
96    protected final void service(HttpServletRequest request, HttpServletResponse response)
97        throws ServletException, IOException {
98  
99      // Default content type is just text
100     response.setContentType("text/plain; charset=UTF-8");
101 
102     long start = System.nanoTime();
103     super.service(request, response);
104     timing.addTimingNanosec(System.nanoTime() - start);
105 
106     int status = response.getStatus();
107     if (status >= 400) {
108       if (status >= 500) {
109         timing.incrementServerErrors();
110       } else {
111         timing.incrementClientErrors();
112       }
113     }
114   }

116   /**
117    * Hack: we have to double-escape forward-slash so that Tomcat won't read it as a path delimiter,
118    * then un-escape again on the other end.
119    *
120    * @param pathElement path element potentially containing %2F
121    * @return argument, with %2F converted to /
122    */
123   protected static String unescapeSlashHack(CharSequence pathElement) {
124     return ESCAPED_SLASH.matcher(pathElement).replaceAll("/");
125   }
126 
127   protected static void unescapeSlashHack(String[] pathElements) {
128     for (int i = 0; i < pathElements.length; i++) {
129       pathElements[i] = unescapeSlashHack(pathElements[i]);
130     }
131   }

133   protected final void output(HttpServletRequest request,
134                               ServletResponse response,
135                               float... values) throws IOException {
136     Writer writer = response.getWriter();
137     switch (determineResponseType(request)) {
138       case JSON:
139         response.setContentType("application/json");
140         // Single value written alone
141         if (values.length == 1) {
142           writer.write(Float.toString(values[0]));
143         } else {
144           // Many values written as array
145           writer.write('[');
146           boolean first = true;
147           for (float value : values) {
148             if (first) {
149               first = false;
150             } else {
151               writer.write(',');
152             }
153             writer.write(Float.toString(value));
154           }
155           writer.write("]\n");
156         }
157         break;
158       case DELIMITED:
159         // Leave content type at default
160         for (float value : values) {
161           writer.write(Float.toString(value));
162           writer.write('\n');
163         }
164         break;
165       default:
166         throw new IllegalStateException("Unknown response type");
167     }
168   }
169 
170   protected final void output(HttpServletRequest request,
171                               ServletResponse response,
172                               double... values) throws IOException {
173     Writer writer = response.getWriter();
174     switch (determineResponseType(request)) {
175       case JSON:
176         // Single value written alone
177         if (values.length == 1) {
178           writer.write(Double.toString(values[0]));
179         } else {
180           // Many values written as array
181           writer.write('[');
182           boolean first = true;
183           for (double value : values) {
184             if (first) {
185               first = false;
186             } else {
187               writer.write(',');
188             }
189             writer.write(Double.toString(value));
190           }
191           writer.write("]\n");
192         }
193         break;
194       case DELIMITED:
195         for (double value : values) {
196           writer.write(Double.toString(value));
197           writer.write('\n');
198         }
199         break;
200       default:
201         throw new IllegalStateException("Unknown response type");
202     }
203   }

205   /**
206    * Determines the appropriate content type for the response based on request headers. At the moment these
207    * are chosen from the values in {@link ResponseContentType}.
208    */
209   protected final ResponseContentType determineResponseType(HttpServletRequest request) {
210 
211     String acceptHeader = request.getHeader(HttpHeaders.ACCEPT);
212     if (acceptHeader == null) {
213       return ResponseContentType.DELIMITED;
214     }
215     ResponseContentType cached = responseTypeCache.get(acceptHeader);
216     if (cached != null) {
217       return cached;
218     }
219 
220     SortedMap<Double,ResponseContentType> types = Maps.newTreeMap();
221     for (String accept : COMMA.split(acceptHeader)) {
222       double preference;
223       String type;
224       int semiColon = accept.indexOf(';');
225       if (semiColon < 0) {
226         preference = 1.0;
227         type = accept;
228       } else {
229         String valueString = accept.substring(semiColon + 1).trim();
230         if (valueString.startsWith("q=")) {
231           valueString = valueString.substring(2);
232         }
233         try {
234           preference = LangUtils.parseDouble(valueString);
235         } catch (IllegalArgumentException ignored) {
236           preference = 1.0;
237         }
238         type = accept.substring(semiColon);
239       }
240       ResponseContentType parsedType = null;
241       if ("text/csv".equals(type) || "text/plain".equals(type)) {
242         parsedType = ResponseContentType.DELIMITED;
243       } else if ("application/json".equals(type)) {
244         parsedType = ResponseContentType.JSON;
245       }
246       if (parsedType != null) {
247         types.put(preference, parsedType);
248       }
249     }
250 
251     ResponseContentType finalType;
252     if (types.isEmpty()) {
253       finalType = ResponseContentType.DELIMITED;
254     } else {
255       finalType = types.values().iterator().next();
256     }
257 
258     responseTypeCache.putIfAbsent(acceptHeader, finalType);
259     return finalType;
260   }
261 
262   /**
263    * Available content types / formats for response bodies.
264    */
265   protected enum ResponseContentType {
266     JSON,
267     DELIMITED,
268   }
