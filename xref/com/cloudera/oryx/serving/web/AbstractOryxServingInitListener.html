41  /**
42   * Superclass of init listeners for Serving Layer APIs. Subclasses are where algorithm-specific initialization
43   * for servlets for a particular type of API occurs.
44   */
45  public abstract class AbstractOryxServingInitListener implements ServletContextListener {
46  
47    private static final Logger log = LoggerFactory.getLogger(AbstractOryxServingInitListener.class);
48  
49    public static final String READ_ONLY_KEY = "READ_ONLY";
50    private static final String LOCAL_INPUT_DIR_KEY = "LOCAL_INPUT_DIR";
51  
52    private File localInputDir;
53  
54    @Override
55    public void contextInitialized(ServletContextEvent event) {
56      ServletContext context = event.getServletContext();
57      configureLogging(context);
58      configureTempDir(context);
59      configureLocalInputDir();
60      Config config = ConfigUtils.getDefaultConfig();
61      context.setAttribute(READ_ONLY_KEY, config.getBoolean("serving-layer.api.read-only"));
62      context.setAttribute(LOCAL_INPUT_DIR_KEY, getLocalInputDir());
63  
64      APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
65      log.info("Serving Layer console available at {}", apiSettings.getConsoleURI());
66    }
67  
68    @Override
69    public void contextDestroyed(ServletContextEvent event) {
70      // No-op default
71    }
72  
73    protected final File getLocalInputDir() {
74      return localInputDir;
75    }
76  
77    private static void configureLogging(ServletContext context) {
78      MemoryHandler.setSensibleLogFormat();
79      Handler logHandler = null;
80      for (Handler handler : java.util.logging.Logger.getLogger("").getHandlers()) {
81        if (handler instanceof MemoryHandler) {
82          logHandler = handler;
83          break;
84        }
85      }
86      if (logHandler == null) {
87        // Not previously configured by command line, make a new one
88        logHandler = new MemoryHandler();
89        java.util.logging.Logger.getLogger("").addHandler(logHandler);
90      }
91      context.setAttribute(LogServlet.LOG_HANDLER, logHandler);
92    }
93  
94    /**
95     * This is a possible workaround for Tomcat on Windows, not creating the temp dir it allocates?
96     */
97    private static void configureTempDir(ServletContext context) {
98      File tempDir = (File) context.getAttribute(ServletContext.TEMPDIR);
99      Preconditions.checkNotNull(tempDir, "Servlet container didn't set %s", ServletContext.TEMPDIR);
100     Preconditions.checkArgument(!tempDir.isFile());
101     try {
102       IOUtils.mkdirs(tempDir);
103     } catch (IOException e) {
104       log.warn("Can't make {}", tempDir);
105     }
106   }
107 
108   private void configureLocalInputDir() {
109     Config config = ConfigUtils.getDefaultConfig();
110     String localInputDirName = config.getString("serving-layer.local-input-dir");
111     localInputDir = new File(localInputDirName);
112     try {
113       IOUtils.mkdirs(localInputDir);
114     } catch (IOException e) {
115       log.warn("Can't make {}", localInputDir);
116     }
117   }
118 
119   public static Wrapper addServlet(Context context, Servlet servlet, String path) {
120     String name = servlet.getClass().getSimpleName();
121     Wrapper servletWrapper = Tomcat.addServlet(context, name, servlet);
122     servletWrapper.setLoadOnStartup(1);
123     context.addServletMapping(path, name);
124     return servletWrapper;
125   }
126 
127   public abstract HttpServlet getApiJSPXFile();
128 
129   public abstract void addServlets(Context context);
130 
131 }
