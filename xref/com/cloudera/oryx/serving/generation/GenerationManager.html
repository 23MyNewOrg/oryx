42  /**
43   * An implementation of {@link GenerationManager} is responsible for interacting with successive generations of the
44   * underlying model. It sends updates to the component responsible for computing the model,
45   * and manages switching in new models when they become available.
46   *
47   * @author Sean Owen
48   */
49  public abstract class GenerationManager implements Closeable {
50  
51    private static final Logger log = LoggerFactory.getLogger(GenerationManager.class);
52  
53    protected static final int NO_GENERATION = Integer.MIN_VALUE;
54  
55    private final String instanceDir;
56    private final ScheduledExecutorService executorService;
57    private int writeGeneration;
58    private Writer appender;
59    private final File appendTempDir;
60    private File appenderTempFile;
61    private final long writesBetweenUpload;
62    private long countdownToUpload;
63    private final Semaphore refreshSemaphore;
64  
65    protected GenerationManager(File appendTempDir) throws IOException {
66  
67      Config config = ConfigUtils.getDefaultConfig();
68  
69      this.instanceDir = config.getString("model.instance-dir");
70      this.appendTempDir = appendTempDir;
71      writesBetweenUpload = config.getLong("model.writes-between-upload");
72      countdownToUpload = writesBetweenUpload;
73  
74      writeGeneration = NO_GENERATION;
75  
76      executorService = Executors.newScheduledThreadPool(3, new ThreadFactoryBuilder().setDaemon(true).build());
77      refreshSemaphore = new Semaphore(1);
78  
79      executorService.scheduleWithFixedDelay(new Runnable() {
80        @Override
81        public void run() {
82          if (!executorService.isShutdown()) {
83            try {
84              maybeRollAppender();
85            } catch (Throwable t) {
86              log.warn("Exception while maybe rolling appender", t);
87            }
88          }
89        }
90      }, 2, 2, TimeUnit.MINUTES);
91      maybeRollAppender();
92  
93      executorService.scheduleWithFixedDelay(new Runnable() {
94        @Override
95        public void run() {
96          if (!executorService.isShutdown()) {
97            try {
98              refresh();
99            } catch (Throwable t) {
100             log.warn("Exception while refreshing", t);
101           }
102         }
103       }
104     }, 0, 7, TimeUnit.MINUTES); // Should be mutually prime with delay set above
105   }
106 
107   protected final synchronized Writer getAppender() {
108     return appender;
109   }

111   private synchronized void closeAppender() {
112     if (appender != null) {
113 
114       if (writeGeneration < 0L) {
115         log.warn("No write generation to upload file to; aborting upload and waiting for later");
116         return;
117       }
118 
119       try {
120         appender.close();
121       } catch (IOException ioe) {
122         log.warn("Unable to close {} ({}); aborting and deleting file", appenderTempFile, ioe.getMessage());
123         appender = null;
124         try {
125           IOUtils.delete(appenderTempFile);
126         } catch (IOException e) {
127           log.warn("Failed to delete {}", appenderTempFile);
128         }
129         appenderTempFile = null;
130         return;
131       }
132 
133       appender = null;
134 
135       final File fileToUpload = appenderTempFile;
136       appenderTempFile = null;
137 
138       boolean fileToUploadHasData;
139       try {
140         fileToUploadHasData = fileToUpload.exists() && !IOUtils.isGZIPFileEmpty(fileToUpload);
141       } catch (IOException ioe) {
142         log.warn("Unexpected error checking {} for data; deleting", fileToUpload, ioe);
143         try {
144           IOUtils.delete(fileToUpload);
145         } catch (IOException e) {
146           log.warn("Failed to delete {}", fileToUpload);
147         }
148         fileToUploadHasData = false;
149       }
150 
151       if (fileToUploadHasData) {
152 
153         final String appendKey =
154             Namespaces.getInstanceGenerationPrefix(instanceDir, writeGeneration) + "inbound/" + fileToUpload.getName();
155 
156         Callable<?> uploadCallable = new Callable<Object>() {
157           @Override
158           public Void call() {
159             log.info("Uploading {} to {}", fileToUpload, appendKey);
160             String appendProgressKey = appendKey + ".inprogress";
161             try {
162               Store store = Store.get();
163               store.touch(appendProgressKey);
164               store.upload(appendKey, fileToUpload, false);
165               log.info("Uploaded to {}", appendKey);
166             } catch (Throwable t) {
167               log.warn("Unable to upload {} Continuing...", fileToUpload);
168               log.warn("Exception was:", t);
169             } finally {
170               try {
171                 Store.get().delete(appendProgressKey);
172                 IOUtils.delete(fileToUpload);
173               } catch (IOException e) {
174                 log.warn("Could not delete {}", appendProgressKey, e);
175               }
176             }
177             return null;
178           }
179         };
180 
181         if (executorService.isShutdown()) {
182           // Occurring during shutdown, so can't handle exceptions or use the executor
183           try {
184             uploadCallable.call();
185           } catch (Exception e) {
186             log.warn("Unexpected error while trying to upload file during shutdown", e);
187           }
188         } else {
189           Preconditions.checkNotNull(executorService.submit(uploadCallable));
190         }
191 
192       } else {
193         // Just delete right away
194         log.info("File appears to have no data, deleting: {}", fileToUpload);
195         try {
196           IOUtils.delete(fileToUpload);
197         } catch (IOException e) {
198           log.warn("Failed to delete {}", appenderTempFile);
199         }
200       }
201     }
202   }
203 
204   protected final synchronized void decrementCountdownToUpload() {
205     countdownToUpload--;
206   }
207 
208   private synchronized void maybeRollAppender() throws IOException {
209     int newMostRecentGeneration = getMostRecentGeneration();
210     if (newMostRecentGeneration > writeGeneration || countdownToUpload <= 0) {
211       countdownToUpload = writesBetweenUpload;
212       // Close and write into *current* write generation first -- but only if it exists
213       if (writeGeneration >= 0L) {
214         closeAppender();
215         writeGeneration = newMostRecentGeneration;
216       } else {
217         writeGeneration = newMostRecentGeneration;
218         closeAppender();
219       }
220       appenderTempFile = File.createTempFile("oryx-append-", ".csv.gz", appendTempDir);
221       // A small buffer is needed here, but GZIPOutputStream already provides a substantial native buffer
222       appender = IOUtils.buildGZIPWriter(appenderTempFile);
223     }
224   }
225 
226   private int getMostRecentGeneration() throws IOException {
227     List<String> recentGenerationPathStrings = StoreUtils.listGenerationsForInstance(instanceDir);
228     if (recentGenerationPathStrings.isEmpty()) {
229       log.warn("No generation found at all at {}; wrong path?", Namespaces.getInstancePrefix(instanceDir));
230       return -1;
231     }
232     return StoreUtils.parseGenerationFromPrefix(
233         recentGenerationPathStrings.get(recentGenerationPathStrings.size() - 1));
234   }
235 
236   @Override
237   public final synchronized void close() {
238     ExecutorUtils.shutdownAndAwait(executorService); // Let others complete
239     closeAppender();
240   }
241 
242   /**
243    * Triggers a refresh of the object's internal state, which particularly includes rebuilding or reloading
244    * a matrix model.
245    */
246   public final synchronized void refresh() {
247     try {
248       if (appender != null) {
249         appender.flush();
250       }
251     } catch (IOException e) {
252       log.warn("Exception while flushing", e);
253     }
254 
255     if (refreshSemaphore.tryAcquire()) {
256       Preconditions.checkNotNull(executorService.submit(new RefreshCallable()));
257     } else {
258       log.info("Refresh is already in progress");
259     }
260   }
261 
262   protected abstract void loadRecentModel(int mostRecentModelGeneration) throws IOException;
263 
264   private final class RefreshCallable implements Callable<Object> {
265 
266     @Override
267     public Void call() {
268       try {
269         maybeRollAppender();
270         int mostRecentModelGeneration = getMostRecentModelGeneration();
271         if (mostRecentModelGeneration >= 0) {
272           loadRecentModel(mostRecentModelGeneration);
273         } else {
274           log.info("No available generation, nothing to do");
275         }
276       } catch (Throwable t) {
277         log.warn("Unexpected exception while refreshing", t);
278       } finally {
279         refreshSemaphore.release();
280       }
281       return null;
282     }
283 
284     private int getMostRecentModelGeneration() throws IOException {
285       List<String> recentGenerationPathStrings = StoreUtils.listGenerationsForInstance(instanceDir);
286       Store store = Store.get();
287       for (int i = recentGenerationPathStrings.size() - 1; i >= 0; i--) {
288         int generationID = StoreUtils.parseGenerationFromPrefix(recentGenerationPathStrings.get(i));
289         String instanceGenerationPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
290         if (store.exists(Namespaces.getGenerationDoneKey(instanceDir, generationID), true) &&
291             store.exists(instanceGenerationPrefix + "model.pmml.gz", true)) {
292           return generationID;
293         }
294       }
295       return -1;
296     }

298   }
