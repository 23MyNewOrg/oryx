33  package com.cloudera.oryx.common.io;

35  import java.io.DataInput;
36  import java.io.DataOutput;
37  import java.io.IOException;
38  
39  import com.google.common.base.Preconditions;

41  /**
42   * <p>Encodes signed and unsigned values using a common variable-length
43   * scheme, found for example in
44   * <a href="http://code.google.com/apis/protocolbuffers/docs/encoding.html">
45   * Google's Protocol Buffers</a>. It uses fewer bytes to encode smaller values,
46   * but will use slightly more bytes to encode large values.</p>
47   *
48   * <p>Signed values are further encoded using so-called zig-zag encoding
49   * in order to make them "compatible" with variable-length encoding.</p>
50   *
51   * @author Sean Owen
52   * @author Mahout
53   */
54  public final class Varint {
55  
56    private Varint() {
57    }
58  
59    /**
60     * Encodes a value using the variable-length encoding from
61     * <a href="http://code.google.com/apis/protocolbuffers/docs/encoding.html">
62     * Google Protocol Buffers</a>. It uses zig-zag encoding to efficiently
63     * encode signed values. If values are known to be nonnegative,
64     * {@link #writeUnsignedVarLong(long, DataOutput)} should be used.
65     *
66     * @param value value to encode
67     * @param out to write bytes to
68     * @throws IOException if {@link DataOutput} throws {@link IOException}
69     */
70    public static void writeSignedVarLong(long value, DataOutput out) throws IOException {
71      // Great trick from http://code.google.com/apis/protocolbuffers/docs/encoding.html#types
72      writeUnsignedVarLong((value << 1) ^ (value >> 63), out);
73    }
74  
75    /**
76     * Encodes a value using the variable-length encoding from
77     * <a href="http://code.google.com/apis/protocolbuffers/docs/encoding.html">
78     * Google Protocol Buffers</a>. Zig-zag is not used, so input must not be negative.
79     * If values can be negative, use {@link #writeSignedVarLong(long, DataOutput)}
80     * instead. This method treats negative input as like a large unsigned value.
81     *
82     * @param value value to encode
83     * @param out to write bytes to
84     * @throws IOException if {@link DataOutput} throws {@link IOException}
85     */
86    public static void writeUnsignedVarLong(long value, DataOutput out) throws IOException {
87      while ((value & 0xFFFFFFFFFFFFFF80L) != 0L) {
88        out.writeByte(((int) value & 0x7F) | 0x80);
89        value >>>= 7;
90      }
91      out.writeByte((int) value & 0x7F);
92    }
93  
94    /**
95     * @see #writeSignedVarLong(long, DataOutput)
96     */
97    public static void writeSignedVarInt(int value, DataOutput out) throws IOException {
98      // Great trick from http://code.google.com/apis/protocolbuffers/docs/encoding.html#types
99      writeUnsignedVarInt((value << 1) ^ (value >> 31), out);
100   }
101 
102   /**
103    * @see #writeUnsignedVarLong(long, DataOutput)
104    */
105   public static void writeUnsignedVarInt(int value, DataOutput out) throws IOException {
106     while ((value & 0xFFFFFF80) != 0L) {
107       out.writeByte((value & 0x7F) | 0x80);
108       value >>>= 7;
109     }
110     out.writeByte(value & 0x7F);
111   }

113   /**
114    * @param in to read bytes from
115    * @return decode value
116    * @throws IOException if {@link DataInput} throws {@link IOException}
117    * @throws IllegalArgumentException if variable-length value does not terminate
118    *  after 9 bytes have been read
119    * @see #writeSignedVarLong(long, DataOutput)
120    */
121   public static long readSignedVarLong(DataInput in) throws IOException {
122     long raw = readUnsignedVarLong(in);
123     // This undoes the trick in writeSignedVarLong()
124     long temp = (((raw << 63) >> 63) ^ raw) >> 1;
125     // This extra step lets us deal with the largest signed values by treating
126     // negative results from read unsigned methods as like unsigned values
127     // Must re-flip the top bit if the original read value had it set.
128     return temp ^ (raw & (1L << 63));
129   }
130 
131   /**
132    * @param in to read bytes from
133    * @return decode value
134    * @throws IOException if {@link DataInput} throws {@link IOException}
135    * @throws IllegalArgumentException if variable-length value does not terminate
136    *  after 9 bytes have been read
137    * @see #writeUnsignedVarLong(long, DataOutput)
138    */
139   public static long readUnsignedVarLong(DataInput in) throws IOException {
140     long value = 0L;
141     int i = 0;
142     long b;
143     while (((b = in.readByte()) & 0x80L) != 0) {
144       value |= (b & 0x7F) << i;
145       i += 7;
146       Preconditions.checkArgument(i <= 63, "Variable length quantity is too long");
147     }
148     return value | (b << i);
149   }
150 
151   /**
152    * Like {@link #readUnsignedVarLong(DataInput)} but operates on a byte array.
153    */
154   public static long readUnsignedVarLong(byte[] bytes, int start, int length) {
155     long value = 0L;
156     int i = 0;
157     int maxMinus1 = start + length - 1;
158     for (int offset = start; offset < maxMinus1; offset++) {
159       long b = bytes[offset];
160       value |= (b & 0x7F) << i;
161       i += 7;
162     }
163     long b = bytes[maxMinus1];
164     return value | (b << i);
165   }
166 
167   /**
168    * @throws IllegalArgumentException if variable-length value does not terminate
169    *  after 5 bytes have been read
170    * @throws IOException if {@link DataInput} throws {@link IOException}
171    * @see #readSignedVarLong(DataInput)
172    */
173   public static int readSignedVarInt(DataInput in) throws IOException {
174     int raw = readUnsignedVarInt(in);
175     // This undoes the trick in writeSignedVarInt()
176     int temp = (((raw << 31) >> 31) ^ raw) >> 1;
177     // This extra step lets us deal with the largest signed values by treating
178     // negative results from read unsigned methods as like unsigned values.
179     // Must re-flip the top bit if the original read value had it set.
180     return temp ^ (raw & (1 << 31));
181   }
182 
183   /**
184    * @throws IllegalArgumentException if variable-length value does not terminate
185    *  after 5 bytes have been read
186    * @throws IOException if {@link DataInput} throws {@link IOException}
187    * @see #readUnsignedVarLong(DataInput)
188    */
189   public static int readUnsignedVarInt(DataInput in) throws IOException {
190     int value = 0;
191     int i = 0;
192     int b;
193     while (((b = in.readByte()) & 0x80) != 0) {
194       value |= (b & 0x7F) << i;
195       i += 7;
196       Preconditions.checkArgument(i <= 35, "Variable length quantity is too long");
197     }
198     return value | (b << i);
199   }

201   /**
202    * Like {@link #readUnsignedVarInt(java.io.DataInput)} but operates on a byte array.
203    */
204   public static int readUnsignedVarInt(byte[] bytes, int start, int length) {
205     int value = 0;
206     int i = 0;
207     int maxMinus1 = start + length - 1;
208     for (int offset = start; offset < maxMinus1; offset++) {
209       int b = bytes[offset];
