61    private Store() {
62      Config config = ConfigUtils.getDefaultConfig();
63      boolean localData;
64      if (config.hasPath("model.local")) {
65        log.warn("model.local is deprecated; use model.local-data");
66        localData = config.getBoolean("model.local");
67      } else {
68        localData = config.getBoolean("model.local-data");
69      }
70      try {
71        Configuration conf = OryxConfiguration.get();
72        if (localData) {
73          fs = FileSystem.getLocal(conf);
74        } else {
75          UserGroupInformation.setConfiguration(conf);
76          fs = FileSystem.get(URI.create(Namespaces.get().getPrefix()), conf);
77        }
78      } catch (IOException ioe) {
79        log.error("Unable to configure Store", ioe);
80        throw new IllegalStateException(ioe);
81      }
82    }

117   /**
118    * Gets the total size of all files in all subdirectories of a path.
119    *
120    * @param key path to compute size of
121    * @return total number of bytes at the requested path in bytes
122    */
123   public long getSizeRecursive(String key) throws IOException {
124     Preconditions.checkNotNull(key);
125     Path path = Namespaces.toPath(key);
126 
127     if (!fs.exists(path)) {
128       return 0L;
129     }
130 
131     long size = 0L;
132     RemoteIterator<? extends FileStatus> it = fs.listFiles(path, true);
133     while (it.hasNext()) {
134       FileStatus status = it.next();
135       // listFiles() should only show files
136       Preconditions.checkState(status.isFile());
137       size += status.getLen();
138     }
139 
140     return size;
141   }

177   private void makeParentDirs(Path path) throws IOException {
178     Preconditions.checkNotNull(path);
179     Path parent = path.getParent();
180     if (fs.exists(parent)) {
181       // Can't be a file
182       return;
183     }
184     boolean success;
185     try {
186       success = fs.mkdirs(parent);
187     } catch (AccessControlException ace) {
188       log.error("Permissions problem; is {} writable in HDFS?", parent);
189       throw ace;
190     }
191     if (!success && !fs.exists(parent)) {
192       throw new IOException("Can't make " + parent);
193     }
194   }

198    * @param file local {@link File} to store data into -- can be directory or explicit file

206     fs.copyToLocalFile(false, path, filePath);

209   /**
210    * @param dirKey location of directory whose contents will be downloaded
211    * @param dir local {@link File} to store files/directories under
212    * @throws IOException if an error occurs while downloading
213    */
214   public void downloadDirectory(String dirKey, File dir) throws IOException {
215     Preconditions.checkNotNull(dirKey);
216     Preconditions.checkNotNull(dir);
217     Preconditions.checkArgument(dir.exists() && dir.isDirectory(), "Not a directory: %s", dir);
218 
219     Path dirPath = Namespaces.toPath(dirKey);
220     if (!fs.exists(dirPath)) {
221       return;
222     }
223     Preconditions.checkArgument(fs.getFileStatus(dirPath).isDirectory(), "Not a directory: %s", dirPath);
224 
225     boolean complete = false;
226     try {
227       for (FileStatus status : fs.listStatus(dirPath)) {
228         String name = status.getPath().getName();
229         String fromKey = dirKey + '/' + name;
230         File toLocal = new File(dir, name);
231         if (status.isFile()) {
232           download(fromKey, toLocal);
233         } else {
234           boolean success = toLocal.mkdir();
235           if (!success && !toLocal.exists()) {
236             throw new IOException("Can't make " + toLocal);
237           }
238           downloadDirectory(fromKey, toLocal);
239         }
240       }
241       complete = true;
242     } finally {
243       if (!complete) {
244         log.warn("Failed to download {} so deleting {}", dirKey, dir);
245         IOUtils.deleteRecursively(dir);
246       }
247     }
248   }

250   /**
251    * Uploads a local file to a remote file.
252    *
253    * @param key file to write to
254    * @param file file bytes to upload
255    * @param overwrite if true, overwrite the existing file data if exists already
256    * @throws IOException if the data can't be written, or file exists and overwrite is false
257    */
258   public void upload(String key, File file, boolean overwrite) throws IOException {
259     Preconditions.checkNotNull(key);
260     Preconditions.checkNotNull(file);
261     Preconditions.checkArgument(file.exists(), "Doesn't exist: %s", file);
262     Preconditions.checkArgument(file.isFile(), "Not a file: %s", file);
263     if (!overwrite && exists(key, true)) {
264       throw new IOException(key + " already exists");
265     }
266 
267     Path path = Namespaces.toPath(key);
268     makeParentDirs(path);
269     Path filePath = new Path(file.getAbsolutePath());
270     try {
271       fs.copyFromLocalFile(false, overwrite, filePath, path);
272     } catch (AccessControlException ace) {
273       log.error("Permissions problem; is {} writable in HDFS?", path);
274       throw ace;
275     }
276     if (!fs.exists(path)) {
277       throw new IOException("Couldn't upload " + filePath + " to " + path);
278     }
279   }

281   /**
282    * Uploads a directory recurisvely
283    *
284    * @param dirKey location under which to store the contents found under {@code dir}
285    * @param dir directory whose <em>contents</em> will be uploaded
286    * @param overwrite if true, overwrite existing files
287    * @throws IOException if the data can't be written, or file exists and overwrite is false
288    */
289   public void uploadDirectory(String dirKey, File dir, boolean overwrite) throws IOException {
290     Preconditions.checkNotNull(dirKey);
291     Preconditions.checkNotNull(dir);
292     Preconditions.checkArgument(dir.exists() && dir.isDirectory(), "Not a directory: %s", dir);
293     File[] contents = dir.listFiles(IOUtils.NOT_HIDDEN);
294     if (contents != null) {
295       boolean complete = false;
296       try {
297         for (File content : contents) {
298           String toKey = dirKey + '/' + content.getName();
299           if (content.isFile()) {
300             upload(toKey, content, overwrite);
301           } else {
302             uploadDirectory(toKey, content, overwrite);
303           }
304         }
305         complete = true;
306       } finally {
307         if (!complete) {
308           log.warn("Failed to upload {} so deleting {}", dir, dirKey);
309           recursiveDelete(dirKey);
310         }
311       }
312     }
313   }

336   /**
337    * Makes a directory. If the file system doesn't have an idea of directories, it makes a 0-length file.
338    *
339    * @param key directory to create
340    */
341   public void mkdir(String key) throws IOException {
342     Preconditions.checkNotNull(key);
343     Path path = Namespaces.toPath(key);
344     if (fs.exists(path)) {
345       if (fs.isDirectory(path)) {
346         // OK
347         return;
348       }
349       throw new IOException("Already exists as file: " + path);
350     }
351     boolean success;
352     try {
353       success = fs.mkdirs(path);
354     } catch (AccessControlException ace) {
355       log.error("Permissions problem; is {} writable in HDFS?", path);
356       throw ace;
357     }
358     if (!success && !fs.exists(path)) {
359       throw new IOException("Can't mkdirs for " + path);
360     }
361   }

368   public void delete(String key) throws IOException {
369     Preconditions.checkNotNull(key);
370     Path path = Namespaces.toPath(key);
371     if (!fs.isFile(path)) {
372       throw new IOException("Not a file: " + path);
373     }
374     boolean success;
375     try {
376       success = fs.delete(path, false);
377     } catch (AccessControlException ace) {
378       log.error("Permissions problem; is {} writable in HDFS?", path);
379       throw ace;
380     }
381     if (!success) {
382       throw new IOException("Can't delete " + path);

386   /**
387    * Recursively deletes a file/directory. If the file system does not have a notion of directories, this deletes
388    * all keys that begin with the given prefix.
389    *
390    * @param keyPrefix file/directory ("prefix") to delete
391    */
392   public void recursiveDelete(String keyPrefix) throws IOException {
393     Preconditions.checkNotNull(keyPrefix);
394     Path path = Namespaces.toPath(keyPrefix);
395     if (!fs.exists(path)) {
396       return;
397     }
398 
399     boolean success;
400     try {
401       log.info("Deleting recursively: {}", path);
402       success = fs.delete(path, true);
403     } catch (AccessControlException ace) {
404       log.error("Permissions problem; is {} writable in HDFS?", path);
405       throw ace;
406     }
407     if (!success) {
408       throw new IOException("Can't delete " + path);
409     }
410   }

412   /**
413    * Lists contents of a directory. For file systems without a notion of directory, this lists prefixes that
414    * have the same prefix as the given prefix, but excludes "directories" (keys with same prefix, but followed
415    * by more path elements). Results are returned in lexicographically sorted order.
416    *
417    * @param prefix directory to list
418    * @param files if true, only list files, not directories
419    * @return list of keys representing directory contents
420    */
421   public List<String> list(String prefix, boolean files) throws IOException {
422     Preconditions.checkNotNull(prefix);
423     Path path = Namespaces.toPath(prefix);
424     if (!fs.exists(path)) {
425       return Collections.emptyList();
426     }
427 
428     Preconditions.checkArgument(fs.getFileStatus(path).isDirectory(), "Not a directory: %s", path);
429     FileStatus[] statuses = fs.listStatus(path, new FilesOrDirsPathFilter(fs, files));
430     String prefixString = Namespaces.get().getPrefix();
431 
432     List<String> result = Lists.newArrayListWithCapacity(statuses.length);
433     for (FileStatus fileStatus : statuses) {
434       String listPath = fileStatus.getPath().toString();
435       Preconditions.checkState(listPath.startsWith(prefixString),
436                                "%s doesn't start with %s", listPath, prefixString);
437       if (!listPath.endsWith("_SUCCESS")) {
438         listPath = listPath.substring(prefixString.length());
439         if (fileStatus.isDirectory() && !listPath.endsWith("/")) {
440           listPath += "/";
441         }
442         result.add(listPath);
443       }
444     }
445     Collections.sort(result);
446     return result;
447   }
