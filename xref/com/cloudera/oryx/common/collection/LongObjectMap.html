62    private static final double LOAD_FACTOR = 1.25;
63    private static final int MAX_SIZE = (int) (RandomUtils.MAX_INT_SMALLER_TWIN_PRIME / LOAD_FACTOR);

65    /** Dummy object used to represent a key that has been removed. */
66    private static final long REMOVED = Long.MAX_VALUE;
67    private static final long NULL = Long.MIN_VALUE;

69    // For faster access
70    long[] keys;
71    V[] values;
72    private int numEntries;
73    private int numSlotsUsed;

75    /** Creates a new  with default capacity. */
76    public LongObjectMap() {
77      this(2);
78    }
79    /**
80     * Creates a new  whose capacity can accommodate the given number of entries without
81     * rehash.
82     * 
83     * @param size desired capacity
84     * @throws IllegalArgumentException if size is less than 0, maxSize is less than 1
85     *  or at least half of {@link RandomUtils#MAX_INT_SMALLER_TWIN_PRIME}, or
86     *  loadFactor is less than 1
87     */
88    public LongObjectMap(int size) {
89      Preconditions.checkArgument(size >= 0, "size must be at least 0");
90      Preconditions.checkArgument(size < MAX_SIZE, "size must be less than " + MAX_SIZE);
91      int hashSize = RandomUtils.nextTwinPrime((int) (LOAD_FACTOR * size) + 1);
92      keys = new long[hashSize];
93      Arrays.fill(keys, NULL);
94  
95      @SuppressWarnings("unchecked")
96      V[] theValues = (V[]) new Object[hashSize];
97      values = theValues;
98    }

100   /**
101    * @see #findForAdd(long)
102    */
103   private int find(long key) {
104     int theHashCode = (int) key & 0x7FFFFFFF; // make sure it's positive
105     long[] keys = this.keys;
106     int hashSize = keys.length;
107     int jump = 1 + theHashCode % (hashSize - 2);
108     int index = theHashCode % hashSize;
109     long currentKey = keys[index];
110     while (currentKey != NULL && key != currentKey) {
111       index -= index < jump ? jump - hashSize : jump;
112       currentKey = keys[index];
113     }
114     return index;
115   }

117   /**
118    * @see #find(long)
119    */
120   private int findForAdd(long key) {
121     int theHashCode = (int) key & 0x7FFFFFFF; // make sure it's positive
122     long[] keys = this.keys;
123     int hashSize = keys.length;
124     int jump = 1 + theHashCode % (hashSize - 2);
125     int index = theHashCode % hashSize;
126     long currentKey = keys[index];
127     while (currentKey != NULL && currentKey != REMOVED && key != currentKey) {
128       index -= index < jump ? jump - hashSize : jump;
129       currentKey = keys[index];
130     }
131     if (currentKey != REMOVED) {
132       return index;
133     }
134     // If we're adding, it's here, but, the key might have a value already later
135     int addIndex = index;
136     while (currentKey != NULL && key != currentKey) {
137       index -= index < jump ? jump - hashSize : jump;
138       currentKey = keys[index];
139     }
140     return key == currentKey ? index : addIndex;
141   }

193   /**
194    * @param key key to map
195    * @param value value that the key maps to
196    * @return previous value for the key, if any, or {@code null} if there was no previous mapping.
197    */
198   public V put(long key, V value) {
199     Preconditions.checkArgument(key != NULL && key != REMOVED && value != null);
200     // If many slots are used, let's clear it up
201     if (numSlotsUsed * LOAD_FACTOR >= keys.length) {
202       // If over half the slots used are actual entries, let's grow
203       if (numEntries * LOAD_FACTOR >= numSlotsUsed) {
204         growAndRehash();
205       } else {
206         // Otherwise just rehash to clear REMOVED entries and don't grow
207         rehash();
208       }
209     }
210     // Here we may later consider implementing Brent's variation described on page 532
211     int index = findForAdd(key);
212     long keyIndex = keys[index];
213     if (keyIndex == key) {
214       V oldValue = values[index];
215       values[index] = value;
216       return oldValue;
217     }
218     keys[index] = key;
219     values[index] = value;
220     numEntries++;
221     if (keyIndex == NULL) {
222       numSlotsUsed++;
223     }
224     return null;
225   }

227   /**
228    * @param key key for which mapping should be removed
229    * @return previous value for the key, or {@code null} if no such value existed
230    */
231   public V remove(long key) {
232     if (key == NULL || key == REMOVED) {
233       return null;
234     }
235     int index = find(key);
236     if (keys[index] == NULL) {
237       return null;
238     } else {
239       keys[index] = REMOVED;
240       numEntries--;
241       V oldValue = values[index];
242       values[index] = null;
243       // don't decrement numSlotsUsed
244       return oldValue;
245     }
246     // Could un-set recentlyAccessed's bit but doesn't matter
247   }

280   /**
281    * Optimizes internal storage of keys by removing spaces held by previously removed keys.
282    */
283   public void rehash() {
284     rehash(RandomUtils.nextTwinPrime((int) (LOAD_FACTOR * numEntries) + 1));
285   }
286   
287   private void growAndRehash() {
288     Preconditions.checkState(keys.length * LOAD_FACTOR < RandomUtils.MAX_INT_SMALLER_TWIN_PRIME,
289                              "Can't grow any more");
290     rehash(RandomUtils.nextTwinPrime((int) (LOAD_FACTOR * keys.length) + 1));
291   }
292   
293   private void rehash(int newHashSize) {
294     long[] oldKeys = keys;
295     V[] oldValues = values;
296     numEntries = 0;
297     numSlotsUsed = 0;
298     keys = new long[newHashSize];
299     Arrays.fill(keys, NULL);
300 
301     @SuppressWarnings("unchecked")
302     V[] theValues = (V[]) new Object[newHashSize];
303     values = theValues;
304 
305     int length = oldKeys.length;
306     for (int i = 0; i < length; i++) {
307       long key = oldKeys[i];
308       if (key != NULL && key != REMOVED) {
309         put(key, oldValues[i]);
310       }
311     }
312   }
313   
314   void iteratorRemove(int lastNext) {
315     if (lastNext >= values.length) {
316       throw new NoSuchElementException();
317     }
318     Preconditions.checkState(lastNext >= 0);
319     values[lastNext] = null;
320     keys[lastNext] = REMOVED;
321     numEntries--;
322   }

339   @Override
340   public String toString() {
341     if (isEmpty()) {
342       return "{}";
343     }
344     StringBuilder result = new StringBuilder();
345     result.append('{');
346     for (int i = 0; i < keys.length; i++) {
347       long key = keys[i];
348       if (key != NULL && key != REMOVED) {
349         result.append(key).append('=').append(values[i]).append(',');
350       }
351     }
352     result.setCharAt(result.length() - 1, '}');
353     return result.toString();
354   }

356   @Override
357   public int hashCode() {
358     int hash = 0;
359     long[] keys = this.keys;
360     int max = keys.length;
361     for (int i = 0; i < max; i++) {
362       long key = keys[i];
363       if (key != NULL && key != REMOVED) {
364         hash = 31 * hash + ((int) (key >> 32) ^ (int) key);
365         hash = 31 * hash + values[i].hashCode();
366       }
367     }
368     return hash;
369   }

371   @Override
372   public boolean equals(Object other) {
373     if (!(other instanceof LongObjectMap)) {
374       return false;
375     }
376     @SuppressWarnings("unchecked")
377     LongObjectMap<V> otherMap = (LongObjectMap<V>) other;
378     long[] otherKeys = otherMap.keys;
379     V[] otherValues = otherMap.values;
380     int length = keys.length;
381     int otherLength = otherKeys.length;
382     int max = FastMath.min(length, otherLength);
383 
384     int i = 0;
385     while (i < max) {
386       long key = keys[i];
387       long otherKey = otherKeys[i];
388       if (key == NULL || key == REMOVED) {
389         if (otherKey != NULL && otherKey != REMOVED) {
390           return false;
391         }
392       } else {
393         if (key != otherKey || !values[i].equals(otherValues[i])) {
394           return false;
395         }
396       }
397       i++;
398     }
399     while (i < length) {
400       long key = keys[i];
401       if (key != NULL && key != REMOVED) {
402         return false;
403       }
404       i++;
405     }
406     while (i < otherLength) {
407       long key = otherKeys[i];
408       if (key != NULL && key != REMOVED) {
409         return false;
410       }
411       i++;
412     }
413     return true;
414   }
