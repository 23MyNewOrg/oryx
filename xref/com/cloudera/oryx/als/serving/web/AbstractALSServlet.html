28  import java.io.Writer;

30  import com.cloudera.oryx.als.common.IDValue;
31  import com.cloudera.oryx.als.common.OryxRecommender;
32  import com.cloudera.oryx.als.common.rescorer.RescorerProvider;
33  import com.cloudera.oryx.common.io.DelimitedDataUtils;
34  import com.cloudera.oryx.serving.web.AbstractOryxServlet;
35  
36  /**
37   * Superclass of all servletes used in the ALS recommender API.
38   *
39   * @author Sean Owen
40   */
41  public abstract class AbstractALSServlet extends AbstractOryxServlet {
42  
43    private static final int DEFAULT_HOW_MANY = 10;
44  
45    private static final String KEY_PREFIX = AbstractALSServlet.class.getName();
46    public static final String RECOMMENDER_KEY = KEY_PREFIX + ".RECOMMENDER";
47    public static final String RESCORER_PROVIDER_KEY = KEY_PREFIX + ".RESCORER_PROVIDER";
48    private static final String[] NO_PARAMS = new String[0];
49  
50    private OryxRecommender recommender;
51    private RescorerProvider rescorerProvider;
52  
53    @Override
54    public void init(ServletConfig config) throws ServletException {
55      super.init(config);
56      ServletContext context = config.getServletContext();
57      recommender = (OryxRecommender) context.getAttribute(RECOMMENDER_KEY);
58      rescorerProvider = (RescorerProvider) context.getAttribute(RESCORER_PROVIDER_KEY);
59    }
60  
61    final OryxRecommender getRecommender() {
62      return recommender;
63    }
64  
65    final RescorerProvider getRescorerProvider() {
66      return rescorerProvider;
67    }
68  
69    static int getHowMany(ServletRequest request) {
70      String howManyString = request.getParameter("howMany");
71      if (howManyString == null) {
72        return DEFAULT_HOW_MANY;
73      }
74      int howMany = Integer.parseInt(howManyString);
75      Preconditions.checkArgument(howMany > 0, "howMany must be positive");
76      return howMany;
77    }
78  
79    static String[] getRescorerParams(ServletRequest request) {
80      String[] rescorerParams = request.getParameterValues("rescorerParams");
81      return rescorerParams == null ? NO_PARAMS : rescorerParams;
82    }
83  
84    static boolean getConsiderKnownItems(ServletRequest request) {
85      return Boolean.valueOf(request.getParameter("considerKnownItems"));
86    }
87  
88    /**
89     * <p>CSV output contains one recommendation per line, and each line is of the form {@code itemID,strength},
90     * like {@code "ABC",0.53}. Strength is an opaque indicator of the relative quality of the recommendation.</p>
91     */
92    final void output(HttpServletRequest request, ServletResponse response, Iterable<IDValue> items) throws IOException {
93      Writer writer = response.getWriter();
94      switch (determineResponseType(request)) {
95        case JSON:
96          writer.write('[');
97          boolean first = true;
98          for (IDValue item : items) {
99            if (first) {
100             first = false;
101           } else {
102             writer.write(',');
103           }
104           writer.write("[\"");
105           writer.write(item.getID());
106           // Not using DelimitedDataUtils as JSON needs comma
107           writer.write("\",");
108           writer.write(Float.toString(item.getValue()));
109           writer.write(']');
110         }
111         writer.write("]\n");
112         break;
113       case DELIMITED:
114         for (IDValue item : items) {
115           writer.write(DelimitedDataUtils.encode(',', item.getID(), Float.toString(item.getValue())));
116           writer.write('\n');
117         }
118         break;
119       default:
120         throw new IllegalStateException("Unknown response type");
121     }
122   }

124 }
