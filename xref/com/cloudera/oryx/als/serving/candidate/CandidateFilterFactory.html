16  package com.cloudera.oryx.als.serving.candidate;
17  
18  import java.util.concurrent.locks.Lock;
19  
20  import com.cloudera.oryx.common.ClassUtils;
21  import com.google.common.base.Preconditions;
22  import com.typesafe.config.Config;
23  
24  import com.cloudera.oryx.common.collection.LongObjectMap;
25  import com.cloudera.oryx.common.settings.ConfigUtils;
26  
27  /**
28   * <p>This class helps choose which {@link CandidateFilter} to apply to the recommendation process.
29   * If the "model.candidateFilter.customClass" system property is set, then this class will be loaded and used.
30   * See notes in {@link CandidateFilter} about how the class must be implemented.</p>
31   * 
32   * <p>Otherwise, if "model.lsh.sample-ratio" is set to a value less than 1, then {@link LocationSensitiveHashFilter}
33   * will be used. It is a somewhat special case, a built-in type of filter.</p>
34   * 
35   * <p>Otherwise an implementation that does no filtering will be returned.</p>
36   * 
37   * @author Sean Owen
38   */
39  public final class CandidateFilterFactory {
40  
41    private final double lshSampleRatio;
42    private final int numHashes;
43    private final String candidateFilterClassName;
44  
45    public CandidateFilterFactory() {
46      Config config = ConfigUtils.getDefaultConfig();
47      lshSampleRatio = config.getDouble("model.lsh.sample-ratio");
48      numHashes = config.getInt("model.lsh.num-hashes");
49      candidateFilterClassName =
50          config.hasPath("serving-layer.candidate-filter-class") ?
51          config.getString("serving-layer.candidate-filter-class") : null;
52    }
53  
54    /**
55     * @return an implementation of {@link CandidateFilter} chosen per above. It will be non-null.
56     * 
57     * @param Y item-feature matrix
58     * @param yReadLock read lock that should be acquired to access {@code Y}
59     */
60    public CandidateFilter buildCandidateFilter(LongObjectMap<float[]> Y, Lock yReadLock) {
61      Preconditions.checkNotNull(Y);
62      if (!Y.isEmpty()) {
63        yReadLock.lock();
64        try {
65          if (candidateFilterClassName != null) {
66            return ClassUtils.loadInstanceOf(candidateFilterClassName,
67                                             CandidateFilter.class,
68                                             new Class<?>[]{LongObjectMap.class},
69                                             new Object[]{Y});
70          }
71          // LSH is a bit of a special case, handled here
72          if (lshSampleRatio < 1.0) {
73            return new LocationSensitiveHashFilter(Y, lshSampleRatio, numHashes);
74          }
75        } finally {
76          yReadLock.unlock();
77        }
78      }
79      return new IdentityCandidateFilter(Y);    
80    }
81    
82  }
