58    GenerationLoader(String instanceDir,
59                     LongSet recentlyActiveUsers,
60                     LongSet recentlyActiveItems,
61                     Object lockForRecent) {
62      this.instanceDir = instanceDir;
63      this.recentlyActiveUsers = recentlyActiveUsers;
64      this.recentlyActiveItems = recentlyActiveItems;
65      // This must be acquired to access the 'recent' fields above:
66      this.lockForRecent = lockForRecent;
67    }

69    void loadModel(long generationID, Generation currentGeneration) throws IOException {
70  
71      File modelPMMLFile = File.createTempFile("oryx-model", ".pmml.gz");
72      modelPMMLFile.deleteOnExit();
73      IOUtils.delete(modelPMMLFile);
74  
75      String generationPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
76      String modelPMMLKey = generationPrefix + "model.pmml.gz";
77      Store.get().download(modelPMMLKey, modelPMMLFile);
78      log.info("Loading model description from {}", modelPMMLKey);
79  
80      ALSModelDescription modelDescription = ALSModelDescription.read(modelPMMLFile);
81      IOUtils.delete(modelPMMLFile);
82  
83      Collection<Future<Object>> futures = Lists.newArrayList();
84      // Limit this fairly sharply to 2 so as to not saturate the network link
85      ExecutorService executor = Executors.newFixedThreadPool(
86          2,
87          new ThreadFactoryBuilder().setDaemon(true).setNameFormat("LoadModel-%d").build());
88  
89      LongSet loadedUserIDs;
90      LongSet loadedItemIDs;
91      LongSet loadedUserIDsForKnownItems;
92      try {
93        loadedUserIDs = loadXOrY(generationPrefix, modelDescription, true, currentGeneration, futures, executor);
94        loadedItemIDs = loadXOrY(generationPrefix, modelDescription, false, currentGeneration, futures, executor);
95  
96        if (currentGeneration.getKnownItemIDs() == null) {
97          loadedUserIDsForKnownItems = null;
98        } else {
99          loadedUserIDsForKnownItems =
100             loadKnownItemIDs(generationPrefix,
101                              modelDescription,
102                              currentGeneration,
103                              futures,
104                              executor);
105       }
106 
107       loadIDMapping(generationPrefix, modelDescription, currentGeneration, futures, executor);
108 
109       ExecutorUtils.checkExceptions(futures);
110 
111       log.info("Finished all load tasks");
112       
113     } finally {
114       ExecutorUtils.shutdownNowAndAwait(executor);
115     }
116 
117     synchronized (lockForRecent) {
118       log.info("Pruning old entries...");
119       removeNotUpdated(currentGeneration.getX().keySetIterator(),
120                        loadedUserIDs,
121                        recentlyActiveUsers,
122                        currentGeneration.getXLock().writeLock());
123       removeNotUpdated(currentGeneration.getY().keySetIterator(),
124                        loadedItemIDs,
125                        recentlyActiveItems,
126                        currentGeneration.getYLock().writeLock());
127       if (loadedUserIDsForKnownItems != null && currentGeneration.getKnownItemIDs() != null) {
128         removeNotUpdated(currentGeneration.getKnownItemIDs().keySetIterator(),
129                          loadedUserIDsForKnownItems,
130                          recentlyActiveUsers,
131                          currentGeneration.getKnownItemLock().writeLock());
132       }
133       this.recentlyActiveItems.clear();
134       this.recentlyActiveUsers.clear();
135     }
136 
137     log.info("Recomputing generation state...");
138     currentGeneration.recomputeState();
139 
140     log.info("All model elements loaded, {} users and {} items", 
141              currentGeneration.getNumUsers(), currentGeneration.getNumItems());
142   }

144   private static LongSet loadXOrY(String generationPrefix,
145                                     ALSModelDescription modelDescription,
146                                     boolean isX,
147                                     Generation generation,
148                                     Collection<Future<Object>> futures,
149                                     ExecutorService executor) throws IOException {
150 
151     String xOrYPrefix = generationPrefix + (isX ? modelDescription.getXPath() : modelDescription.getYPath());
152     final LongSet loadedIDs = new LongSet();
153     
154     final Lock writeLock = isX ? generation.getXLock().writeLock() : generation.getYLock().writeLock();
155     final LongObjectMap<float[]> xOrYMatrix = isX ? generation.getX() : generation.getY();
156 
157     for (final String xOrYFilePrefix : Store.get().list(xOrYPrefix, true)) {
158       futures.add(executor.submit(new Callable<Object>() {
159         @Override
160         public Void call() throws IOException {
161           for (String line : new FileLineIterable(Store.get().readFrom(xOrYFilePrefix))) {
162 
163             int tab = line.indexOf('\t');
164             Preconditions.checkArgument(tab >= 0, "Bad input line in %s: %s", xOrYFilePrefix, line);
165             long id = Long.parseLong(line.substring(0, tab));
166 
167             float[] elements = DataUtils.readFeatureVector(line.substring(tab + 1));
168 
169             writeLock.lock();
170             try {
171               xOrYMatrix.put(id, elements);
172               loadedIDs.add(id);
173             } finally {
174               writeLock.unlock();
175             }
176           }
177           log.info("Loaded feature vectors from {}", xOrYFilePrefix);
178           return null;
179         }
180       }));
181     }
182 
183     return loadedIDs;
184   }

187   private static LongSet loadKnownItemIDs(String generationPrefix,
188                                             ALSModelDescription modelDescription,
189                                             final Generation generation,
190                                             Collection<Future<Object>> futures,
191                                             ExecutorService executor) throws IOException {
192     final LongSet loadedIDs = new LongSet();
193     String knownItemsPrefix = generationPrefix + modelDescription.getKnownItemsPath();
194     for (final String knownItemFilePrefix : Store.get().list(knownItemsPrefix, true)) {
195       futures.add(executor.submit(new Callable<Object>() {
196         @Override
197         public Void call() throws IOException {
198           for (String line : new FileLineIterable(Store.get().readFrom(knownItemFilePrefix))) {
199             int tab = line.indexOf('\t');
200             Preconditions.checkArgument(tab >= 0, "Bad input line in %s: %s", knownItemFilePrefix, line);
201             long userID = Long.parseLong(line.substring(0, tab));
202             LongSet itemIDs = stringToSet(line.substring(tab + 1));
203             Lock writeLock = generation.getKnownItemLock().writeLock();
204             LongObjectMap<LongSet> knownItems = generation.getKnownItemIDs();
205             writeLock.lock();
206             try {
207               knownItems.put(userID, itemIDs);
208               loadedIDs.add(userID);
209             } finally {
210               writeLock.unlock();
211             }
212           }
213           log.info("Loaded known items from {}", knownItemFilePrefix);
214           return null;
215         }
216       }));
217     }
218     return loadedIDs;
219   }

221   private static LongSet stringToSet(CharSequence values) {
222     LongSet result = new LongSet();
223     for (String valueString : DelimitedDataUtils.decode(values)) {
224       result.add(Long.parseLong(valueString));
225     }
226     return result;
227   }

229   private static void removeNotUpdated(LongPrimitiveIterator it,
230                                        LongSet updated,
231                                        LongSet recentlyActive,
232                                        Lock writeLock) {
233     writeLock.lock();
234     try {
235       while (it.hasNext()) {
236         long id = it.nextLong();
237         if (!updated.contains(id) && !recentlyActive.contains(id)) {
238           it.remove();
239         }
240       }
241     } finally {
242       writeLock.unlock();
243     }
244   }

246   private static void loadIDMapping(String generationPrefix,
247                                     ALSModelDescription modelDescription,
248                                     final Generation generation,
249                                     Collection<Future<Object>> futures,
250                                     ExecutorService executor) throws IOException {
251     String idMappingPrefix = generationPrefix + modelDescription.getIDMappingPath();
252     for (final String prefix : Store.get().list(idMappingPrefix, true)) {
253       futures.add(executor.submit(new Callable<Object>() {
254         @Override
255         public Void call() throws IOException {
256           for (CharSequence line : new FileLineIterable(Store.get().readFrom(prefix))) {
257             String[] columns = DelimitedDataUtils.decode(line);
258             long numericID = Long.parseLong(columns[0]);
259             String id = columns[1];
260             Lock writeLock = generation.getKnownItemLock().writeLock();
261             StringLongMapping idMapping = generation.getIDMapping();
262             writeLock.lock();
263             try {
264               idMapping.addMapping(id, numericID);
265             } finally {
266               writeLock.unlock();
267             }
268           }
269           return null;
270         }
271       }));
272     }
