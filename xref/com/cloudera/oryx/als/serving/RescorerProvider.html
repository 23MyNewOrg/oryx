22  /**
23   * <p>Implementations of this interface provide, optionally, objects that can be used to modify and influence
24   * the results of:</p>
25   *
26   * <ul>
27   *  <li>{@link ServerRecommender#recommend(String, int)}</li>
28   *  <li>{@link ServerRecommender#recommendToMany(String[], int, boolean, Rescorer)}</li>
29   *  <li>{@link ServerRecommender#recommendToAnonymous(String[], int, Rescorer)}</li>
30   *  <li>{@link ServerRecommender#mostSimilarItems(String, int, PairRescorer)}</li>
31   * </ul>
32   *
33   * <p>It is a means to inject business logic into the results of {@link ServerRecommender}.</p>
34   *
35   * <p>Implementations of this class are factories. An implementation creates and configures an {@link Rescorer}
36   * rescoring object and returns it for use in the context of one
37   * {@link ServerRecommender#recommend(String, int, Rescorer)} method call. (A {@code Rescorer&lt;LongPair&gt;}
38   * is used for {@link ServerRecommender#mostSimilarItems(String, int, PairRescorer)} since it operates on item ID
39   * <em>pairs</em>, but is otherwise analogous.) The {@link Rescorer} then filters the candidates
40   * recommendations or most similar items by item ID ({@link Rescorer#isFiltered(String)})
41   * or modifies the scores of item candidates that are not filtered ({@link Rescorer#rescore(String, double)})
42   * based on the item ID and original score.</p>
43   *
44   * <p>The factory methods, like {@link #getRecommendRescorer(String[], OryxRecommender, String...)}, take optional
45   * {@code String} arguments. These are passed from the REST API, as a {@code String}, from URL parameter
46   * {@code rescorerParams}. The implementation may need this information to initialize its rescoring
47   * logic for the request.  For example, the argument may be the user's current location, used to filter
48   * results by location.</p>
49   *
50   * <p>For example, a request containing {@code ...?rescorerParams=xxx,yyy,zzz} will result in an {@code args}
51   * parameter with <em>one</em> elements, {@code xxx,yyy,zzz}. A request containing 
52   * {@code ...?rescorerParams=xxx&...rescorerParams=yyy&...rescorerParams=zzz...} will result in an
53   * {@code args} parameter with 3 elements, {@code xxx}, {@code yyy}, {@code zzz}.</p>
54   *
55   * @author Sean Owen
56   * @see MultiRescorer
57   * @see com.cloudera.oryx.als.serving.candidate.CandidateFilter
58   */
59  public interface RescorerProvider {
60  
61    /**
62     * @param userIDs user(s) for which recommendations are being made, which may be needed in the rescoring logic.
63     * @param recommender the recommender instance that is rescoring results
64     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
65     *  information from the request that may be necessary to its logic, like current location. What it means
66     *  is up to the implementation.
67     * @return {@link Rescorer} to use with {@link ServerRecommender#recommend(String, int, Rescorer)}
68     *  or {@code null} if none should be used. The resulting {@link Rescorer} will be passed each candidate
69     *  item ID to {@link Rescorer#isFiltered(String)}, and each non-filtered candidate with its original score
70     *  to {@link Rescorer#rescore(String, double)}
71     */
72    Rescorer getRecommendRescorer(String[] userIDs, OryxRecommender recommender, String... args);
73  
74    /**
75     * @param itemIDs items that the anonymous user is associated to
76     * @param recommender the recommender instance that is rescoring results
77     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
78     *  information from the request that may be necessary to its logic, like current location. What it means
79     *  is up to the implementation.
80     * @return {@link Rescorer} to use with {@link ServerRecommender#recommendToAnonymous(String[], int, Rescorer)}
81     *  or {@code null} if none should be used. The resulting {@link Rescorer} will be passed each candidate
82     *  item ID to {@link Rescorer#isFiltered(String)}, and each non-filtered candidate with its original score
83     *  to {@link Rescorer#rescore(String, double)}
84     */
85    Rescorer getRecommendToAnonymousRescorer(String[] itemIDs, OryxRecommender recommender, String... args);
86  
87    /**
88     * @param recommender the recommender instance that is rescoring results
89     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
90     *  information from the request that may be necessary to its logic, like current location. What it means
91     *  is up to the implementation.
92     * @return {@link Rescorer} to use with {@link ServerRecommender#mostPopularItems(int, Rescorer)}
93     *  or {@code null} if none should be used.
94     */
95    Rescorer getMostPopularItemsRescorer(OryxRecommender recommender, String... args);
96  
97    /**
98     * @param recommender the recommender instance that is rescoring results
99     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
100    *  information from the request that may be necessary to its logic, like current location. What it means
101    *  is up to the implementation.
102    * @return {@link PairRescorer} to use with {@link ServerRecommender#mostSimilarItems(String[], int, PairRescorer)}
103    *  or {@code null} if none should be used. The {@link PairRescorer} will be passed, to its
104    *  {@link PairRescorer#isFiltered(String, String)} method, the candidate item ID
105    *  and item ID passed in the user query as its second element.
106    *  Each non-filtered pair is passed with its original score to
107    *  {@link PairRescorer#rescore(String, String, double)}
108    */
109   PairRescorer getMostSimilarItemsRescorer(OryxRecommender recommender, String... args);

111 }
