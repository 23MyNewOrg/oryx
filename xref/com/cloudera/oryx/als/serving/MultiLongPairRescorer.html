16  package com.cloudera.oryx.als.serving;
17  
18  import java.util.List;
19  
20  import com.google.common.base.Preconditions;
21  
22  import com.cloudera.oryx.als.common.PairRescorer;
23  
24  /**
25   * Convenience implementation that will aggregate the behavior of multiple {@link PairRescorer}s.
26   * It will filter an item if any of the given instances filter it, and will rescore
27   * by applying the rescorings in the given order.
28   *
29   * @author Sean Owen
30   * @see MultiRescorer
31   * @see MultiRescorerProvider
32   */
33  public final class MultiLongPairRescorer implements PairRescorer {
34  
35    private final PairRescorer[] rescorers;
36  
37    public MultiLongPairRescorer(List<PairRescorer> rescorers) {
38      Preconditions.checkNotNull(rescorers);
39      Preconditions.checkState(!rescorers.isEmpty());
40      PairRescorer[] newArray = new PairRescorer[rescorers.size()];
41      this.rescorers = rescorers.toArray(newArray);
42    }
43  
44    @Override
45    public double rescore(String fromID, String toID, double value) {
46      for (PairRescorer rescorer : rescorers) {
47        value = rescorer.rescore(fromID, toID, value);
48        if (Double.isNaN(value)) {
49          return Double.NaN;
50        }
51      }
52      return value;
53    }
54  
55    @Override
56    public boolean isFiltered(String fromID, String toID) {
57      for (PairRescorer rescorer : rescorers) {
58        if (rescorer.isFiltered(fromID, toID)) {
59          return true;
60        }
61      }
62      return false;
63    }
64  
65  }
