16  package com.cloudera.oryx.als.common.rescorer;
17  
18  import com.cloudera.oryx.als.common.OryxRecommender;
19  
20  /**
21   * <p>Implementations of this interface provide, optionally, objects that can be used to modify and influence
22   * the results of:</p>
23   *
24   * <ul>
25   *  <li>{@code ServerRecommender#recommend(String, int)}</li>
26   *  <li>{@code ServerRecommender#recommendToMany(String[], int, boolean, Rescorer)}</li>
27   *  <li>{@code ServerRecommender#recommendToAnonymous(String[], int, Rescorer)}</li>
28   *  <li>{@code ServerRecommender#mostSimilarItems(String, int, PairRescorer)}</li>
29   * </ul>
30   *
31   * <p>It is a means to inject business logic into the results of {@code ServerRecommender}.</p>
32   *
33   * <p>Implementations of this class are factories. An implementation creates and configures an {@link Rescorer}
34   * rescoring object and returns it for use in the context of one
35   * {@code ServerRecommender#recommend(String, int, Rescorer)} method call. (A {@code Rescorer&lt;LongPair&gt;}
36   * is used for {@code ServerRecommender#mostSimilarItems(String, int, PairRescorer)} since it operates on item ID
37   * <em>pairs</em>, but is otherwise analogous.) The {@link Rescorer} then filters the candidates
38   * recommendations or most similar items by item ID ({@link Rescorer#isFiltered(String)})
39   * or modifies the scores of item candidates that are not filtered ({@link Rescorer#rescore(String, double)})
40   * based on the item ID and original score.</p>
41   *
42   * <p>The factory methods, like {@link #getRecommendRescorer(String[], OryxRecommender, String...)}, take optional
43   * {@code String} arguments. These are passed from the REST API, as a {@code String}, from URL parameter
44   * {@code rescorerParams}. The implementation may need this information to initialize its rescoring
45   * logic for the request.  For example, the argument may be the user's current location, used to filter
46   * results by location.</p>
47   *
48   * <p>For example, a request containing {@code ...?rescorerParams=xxx,yyy,zzz} will result in an {@code args}
49   * parameter with <em>one</em> elements, {@code xxx,yyy,zzz}. A request containing 
50   * {@code ...?rescorerParams=xxx&...rescorerParams=yyy&...rescorerParams=zzz...} will result in an
51   * {@code args} parameter with 3 elements, {@code xxx}, {@code yyy}, {@code zzz}.</p>
52   *
53   * @author Sean Owen
54   * @see MultiRescorer
55   * @see com.cloudera.oryx.als.common.candidate.CandidateFilter
56   */
57  public interface RescorerProvider {
58  
59    /**
60     * @param userIDs user(s) for which recommendations are being made, which may be needed in the rescoring logic.
61     * @param recommender the recommender instance that is rescoring results
62     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
63     *  information from the request that may be necessary to its logic, like current location. What it means
64     *  is up to the implementation.
65     * @return {@link Rescorer} to use with {@code ServerRecommender#recommend(String, int, Rescorer)}
66     *  or {@code null} if none should be used. The resulting {@link Rescorer} will be passed each candidate
67     *  item ID to {@link Rescorer#isFiltered(String)}, and each non-filtered candidate with its original score
68     *  to {@link Rescorer#rescore(String, double)}
69     */
70    Rescorer getRecommendRescorer(String[] userIDs, OryxRecommender recommender, String... args);
71  
72    /**
73     * @param itemIDs items that the anonymous user is associated to
74     * @param recommender the recommender instance that is rescoring results
75     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
76     *  information from the request that may be necessary to its logic, like current location. What it means
77     *  is up to the implementation.
78     * @return {@link Rescorer} to use with {@code ServerRecommender#recommendToAnonymous(String[], int, Rescorer)}
79     *  or {@code null} if none should be used. The resulting {@link Rescorer} will be passed each candidate
80     *  item ID to {@link Rescorer#isFiltered(String)}, and each non-filtered candidate with its original score
81     *  to {@link Rescorer#rescore(String, double)}
82     */
83    Rescorer getRecommendToAnonymousRescorer(String[] itemIDs, OryxRecommender recommender, String... args);
84  
85    /**
86     * @param recommender the recommender instance that is rescoring results
87     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
88     *  information from the request that may be necessary to its logic, like current location. What it means
89     *  is up to the implementation.
90     * @return {@link Rescorer} to use with {@code ServerRecommender#mostPopularItems(int, Rescorer)}
91     *  or {@code null} if none should be used.
92     */
93    Rescorer getMostPopularItemsRescorer(OryxRecommender recommender, String... args);
94  
95    /**
96     * @param recommender the recommender instance that is rescoring results
97     * @param args arguments, if any, that should be used when making the {@link Rescorer}. This is additional
98     *  information from the request that may be necessary to its logic, like current location. What it means
99     *  is up to the implementation.
100    * @return {@link PairRescorer} to use with {@code ServerRecommender#mostSimilarItems(String[], int, PairRescorer)}
101    *  or {@code null} if none should be used. The {@link PairRescorer} will be passed, to its
102    *  {@link PairRescorer#isFiltered(String, String)} method, the candidate item ID
103    *  and item ID passed in the user query as its second element.
104    *  Each non-filtered pair is passed with its original score to
105    *  {@link PairRescorer#rescore(String, String, double)}
106    */
107   PairRescorer getMostSimilarItemsRescorer(OryxRecommender recommender, String... args);
108 
109 }
