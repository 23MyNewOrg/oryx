52  /**
53   * <p>Implements the Alternating Least Squares algorithm described in
54   * <a href="http://www2.research.att.com/~yifanhu/PUB/cf.pdf">"Collaborative Filtering for Implicit Feedback Datasets"</a>
55   * by Yifan Hu, Yehuda Koren, and Chris Volinsky.</p>
56   *
57   * <p>This implementation varies in some small details; it does not use the same mechanism for explaining ratings
58   * for example and seeds the initial Y differently.</p>
59   *
60   * <p>Note that in this implementation, matrices are sparse and are implemented with a {@link com.cloudera.oryx.common.collection.LongObjectMap} of
61   * {@link com.cloudera.oryx.common.collection.LongFloatMap} so as to be able to use {@code long} keys. In many cases, a tall, skinny matrix is
62   * needed (sparse rows, dense columns). This is represented with {@link com.cloudera.oryx.common.collection.LongObjectMap} of {@code float[]}.</p>
63   *
64   * <p>This implementation implements essentially this function, expressed in Octave/Matlab:</p>
65   *
66   * <p><pre>
67   * {@code
68   * function [X, Y] = als(R, f, iterations)
69   *
70   *   alpha = 1;
71   *   lambda = 0.1 * alpha;
72   *   [m,n] = size(R);
73   *
74   *   P = R > 0;
75   *   Pnonzero = R != 0;
76   *   C = 1 + alpha * abs(R);
77   *
78   *   X = zeros(m,f);
79   *   Y = stdnormal_rnd(n,f);
80   *   for i = 1:n
81   *     Y(i,:) /= norm(Y(i,:));
82   *   endfor
83   *
84   *   If = eye(f);
85   *   Im = eye(m);
86   *   In = eye(n);
87   *
88   *   for iter = 1:iterations
89   *
90   *     YTY = Y' * Y;
91   *     for u = 1:m
92   *       Cu = zeros(n,n);
93   *       for i = 1:n
94   *         Cu(i,i) = C(u,i);
95   *       endfor
96   *       count = sum(Pnonzero(u,:));
97   *       YTCupu = Y' * Cu * P(u,:)';
98   *       YTCuY = YTY + Y' * (Cu - In) * Y;
99   *       Wu = pinv(YTCuY + lambda * count * If);
100  *       X(u,:) = (Wu * YTCupu)';
101  *
102  *     endfor
103  *
104  *     XTX = X' * X;
105  *     for i = 1:n
106  *       Ci = zeros(m,m);
107  *       for u = 1:m
108  *         Ci(u,u) = C(u,i);
109  *       endfor
110  *       count = sum(Pnonzero(:,i));
111  *       XTCipi = X' * Ci * P(:,i);
112  *       XTCiX = XTX + X' * (Ci - Im) * X;
113  *       Wi = pinv(XTCiX + lambda * count * If);
114  *       Y(i,:) = (Wi * XTCipi)';
115  *
116  *     endfor
117  *
118  *   endfor
119  *
120  * end}
121  * </pre></p>
122  *
123  * @author Sean Owen
124  */

144   /**
145    * @param RbyRow the input R matrix, indexed by row
146    * @param RbyColumn the input R matrix, indexed by column
147    * @param features number of features, must be positive
148    * @param estimateErrorConvergenceThreshold when the average absolute difference in estimated user-item
149    *   scores falls below this threshold between iterations, iterations will stop
150    * @param maxIterations caps the number of iterations run. If non-positive, there is no cap.

152   public AlternatingLeastSquares(LongObjectMap<LongFloatMap> RbyRow,
153                                  LongObjectMap<LongFloatMap> RbyColumn,
154                                  int features,
155                                  double estimateErrorConvergenceThreshold,
156                                  int maxIterations) {
157     Preconditions.checkNotNull(RbyRow);
158     Preconditions.checkNotNull(RbyColumn);
159     Preconditions.checkArgument(features > 0, "features must be positive: %s", features);
160     Preconditions.checkArgument(estimateErrorConvergenceThreshold > 0.0 && estimateErrorConvergenceThreshold < 1.0,
161                                 "threshold must be in (0,1): %s", estimateErrorConvergenceThreshold);
162     this.RbyRow = RbyRow;
163     this.RbyColumn = RbyColumn;
164     this.features = features;
165     this.estimateErrorConvergenceThreshold = estimateErrorConvergenceThreshold;
166     this.maxIterations = maxIterations;
167   }

188   @Override
189   public Void call() throws ExecutionException, InterruptedException {
190 
191     X = new LongObjectMap<float[]>(RbyRow.size());
192 
193     boolean randomY = previousY == null || previousY.isEmpty();
194     Y = constructInitialY(previousY);
195 
196     // This will be used to compute rows/columns in parallel during iteration
197 
198     int numThreads = ExecutorUtils.getParallelism();
199     ExecutorService executor =
200         Executors.newFixedThreadPool(numThreads,
201                                      new ThreadFactoryBuilder().setNameFormat("ALS-%d").setDaemon(true).build());
202 
203     log.info("Iterating using {} threads", numThreads);
204 
205     RandomGenerator random = RandomManager.getRandom();
206     long[] testUserIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
207                                                             RbyRow.keySetIterator(), 
208                                                             RbyRow.size(), 
209                                                             random);
210     long[] testItemIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
211                                                             RbyColumn.keySetIterator(), 
212                                                             RbyColumn.size(), 
213                                                             random);
214     double[][] estimates = new double[testUserIDs.length][testItemIDs.length];
215     if (!X.isEmpty()) {
216       for (int i = 0; i < testUserIDs.length; i++) {
217         for (int j = 0; j < testItemIDs.length; j++) {
218           estimates[i][j] = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j])); 
219         }
220       }
221     }
222     // Otherwise X is empty because it's the first ever iteration. Estimates can be left at initial 0 value
223 
224     try {
225       int iterationNumber = 0;
226       while (true) {
227         iterateXFromY(executor);
228         iterateYFromX(executor);
229         DoubleWeightedMean averageAbsoluteEstimateDiff = new DoubleWeightedMean();
230         for (int i = 0; i < testUserIDs.length; i++) {
231           for (int j = 0; j < testItemIDs.length; j++) {
232             double newValue = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j]));            
233             double oldValue = estimates[i][j];
234             estimates[i][j] = newValue;
235             // Weight, simplistically, by newValue to emphasize effect of good recommendations.
236             // But that only makes sense where newValue > 0
237             if (newValue > 0.0f) {
238               averageAbsoluteEstimateDiff.increment(FastMath.abs(newValue - oldValue), newValue);
239             }
240           }
241         }
242       
243         iterationNumber++;
244         log.info("Finished iteration {}", iterationNumber);
245         if (maxIterations > 0 && iterationNumber >= maxIterations) {
246           log.info("Reached iteration limit");
247           break;
248         }
249         log.info("Avg absolute difference in estimate vs prior iteration: {}", averageAbsoluteEstimateDiff);
250         double convergenceValue = averageAbsoluteEstimateDiff.getResult();
251         if (!Doubles.isFinite(convergenceValue)) {
252           log.warn("Invalid convergence value, aborting iteration! {}", convergenceValue);
253           break;
254         }
255         // Don't converge after 1 iteration if starting from a random point
256         if (!(randomY && iterationNumber == 1) && convergenceValue < estimateErrorConvergenceThreshold) {
257           log.info("Converged");          
258           break;
259         }
260       }
261     } finally {
262       ExecutorUtils.shutdownNowAndAwait(executor);
263     }
264     return null;
265   }

267   private LongObjectMap<float[]> constructInitialY(LongObjectMap<float[]> previousY) {
268 
269     RandomGenerator random = RandomManager.getRandom();
270     
271     LongObjectMap<float[]> randomY;
272     if (previousY == null || previousY.isEmpty()) {
273       // Common case: have to start from scratch
274       log.info("Starting from new, random Y matrix");      
275       randomY = new LongObjectMap<float[]>(RbyColumn.size());
276       
277     } else {
278       
279       int oldFeatureCount = previousY.entrySet().iterator().next().getValue().length;
280       if (oldFeatureCount > features) {
281         // Fewer features, use some dimensions from prior larger number of features as-is
282         log.info("Feature count has decreased to {}, projecting down previous generation's Y matrix", features);                
283         randomY = new LongObjectMap<float[]>(previousY.size());
284         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
285           float[] oldLargerVector = entry.getValue();
286           float[] newSmallerVector = new float[features];
287           System.arraycopy(oldLargerVector, 0, newSmallerVector, 0, newSmallerVector.length);
288           SimpleVectorMath.normalize(newSmallerVector);
289           randomY.put(entry.getKey(), newSmallerVector);
290         }
291         
292       } else if (oldFeatureCount < features) {
293         log.info("Feature count has increased to {}, using previous generation's Y matrix as subspace", features);        
294         randomY = new LongObjectMap<float[]>(previousY.size());
295         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
296           float[] oldSmallerVector = entry.getValue();
297           float[] newLargerVector = new float[features];
298           System.arraycopy(oldSmallerVector, 0, newLargerVector, 0, oldSmallerVector.length);
299           // Fill in new dimensions with random values
300           for (int i = oldSmallerVector.length; i < newLargerVector.length; i++) {
301             newLargerVector[i] = (float) random.nextGaussian();
302           }
303           SimpleVectorMath.normalize(newLargerVector);          
304           randomY.put(entry.getKey(), newLargerVector);
305         }
306         
307       } else {
308         // Common case: previous generation is same number of features
309         log.info("Starting from previous generation's Y matrix");        
310         randomY = previousY;
311       }
312     }
313     
314     List<float[]> recentVectors = Lists.newArrayList();
315     for (LongObjectMap.MapEntry<float[]> entry : randomY.entrySet()) {
316       if (recentVectors.size() >= MAX_FAR_FROM_VECTORS) {
317         break;
318       }
319       recentVectors.add(entry.getValue());
320     }
321     LongPrimitiveIterator it = RbyColumn.keySetIterator();
322     long count = 0;
323     while (it.hasNext()) {
324       long id = it.nextLong();
325       if (!randomY.containsKey(id)) {
326         float[] vector = RandomUtils.randomUnitVectorFarFrom(features, recentVectors, random);
327         randomY.put(id, vector);
328         if (recentVectors.size() < MAX_FAR_FROM_VECTORS) {
329           recentVectors.add(vector);
330         }
331       }
332       if (++count % LOG_INTERVAL == 0) {
333         log.info("Computed {} initial Y rows", count);
334       }
335     }
336     log.info("Constructed initial Y");
337     return randomY;
338   }

340   /**
341    * Runs one iteration to compute X from Y.
342    */
343   private void iterateXFromY(ExecutorService executor) throws ExecutionException, InterruptedException {
344 
345     RealMatrix YTY = MatrixUtils.transposeTimesSelf(Y);
346     Collection<Future<?>> futures = Lists.newArrayList();
347     addWorkers(RbyRow, Y, YTY, X, executor, futures);
348 
349     int count = 0;
350     long total = 0;
351     for (Future<?> f : futures) {
352       f.get();
353       count += WORK_UNIT_SIZE;
354       if (count >= LOG_INTERVAL) {
355         total += count;
356         JVMEnvironment env = new JVMEnvironment();
357         log.info("{} X/tag rows computed ({}MB heap)", total, env.getUsedMemoryMB());
358         if (env.getPercentUsedMemory() > 95) {
359           log.warn("Memory is low. Increase heap size with -Xmx, decrease new generation size with larger " +
360                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
361         }
362         count = 0;
363       }
364     }
365   }

394   private void addWorkers(LongObjectMap<LongFloatMap> R,
395                           LongObjectMap<float[]> M,
396                           RealMatrix MTM, 
397                           LongObjectMap<float[]> MTags,
398                           ExecutorService executor,                          
399                           Collection<Future<?>> futures) {
400     if (R != null) {
401       List<Pair<Long, LongFloatMap>> workUnit = Lists.newArrayListWithCapacity(WORK_UNIT_SIZE);
402       for (LongObjectMap.MapEntry<LongFloatMap> entry : R.entrySet()) {
403         workUnit.add(new Pair<Long,LongFloatMap>(entry.getKey(), entry.getValue()));
404         if (workUnit.size() == WORK_UNIT_SIZE) {
405           futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
406           workUnit = Lists.newArrayListWithCapacity(WORK_UNIT_SIZE);
407         }
408       }
409       if (!workUnit.isEmpty()) {
410         futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
411       }
412     }

415   private static final class Worker implements Callable<Object> {
416 
417     private final int features;
418     private final LongObjectMap<float[]> Y;
419     private final RealMatrix YTY;
420     private final LongObjectMap<float[]> X;
421     private final Iterable<Pair<Long, LongFloatMap>> workUnit;
422 
423     private Worker(int features,
424                    LongObjectMap<float[]> Y,
425                    RealMatrix YTY,
426                    LongObjectMap<float[]> X,
427                    Iterable<Pair<Long, LongFloatMap>> workUnit) {
428       this.features = features;
429       this.Y = Y;
430       this.YTY = YTY;
431       this.X = X;
432       this.workUnit = workUnit;
433     }
434 
435     @Override
436     public Void call() {
437 
438       Config config = ConfigUtils.getDefaultConfig();
439       double alpha = config.getDouble("model.alpha");
440       double lambda = config.getDouble("model.lambda") * alpha;
441       // This will cause the ALS algorithm to reconstruction the input matrix R, rather than the
442       // matrix P = R > 0 . Don't use this unless you understand it!
443       boolean reconstructRMatrix = config.getBoolean("model.reconstruct-r-matrix");
444       // Causes the loss function to exclude entries for any input pairs that do not appear in the
445       // input and are implicitly 0
446       // Likewise, don't touch this for now unless you know what it does.
447       boolean lossIgnoresUnspecified = config.getBoolean("model.loss-ignores-unspecified");
448 
449       int features = this.features;
450       // Each worker has a batch of rows to compute:
451       for (Pair<Long,LongFloatMap> work : workUnit) {
452 
453         // Row (column) in original R matrix containing total association value. For simplicity we will
454         // talk about users and rows only in the comments and variables. It's symmetric for columns / items.
455         // This is Ru:
456         LongFloatMap ru = work.getSecond();
457 
458         // Start computing Wu = (YT*Cu*Y + lambda*I) = (YT*Y + YT*(Cu-I)*Y + lambda*I),
459         // by first starting with a copy of YT * Y. Or, a variant on YT * Y, if LOSS_IGNORES_UNSPECIFIED is set
460         RealMatrix Wu =
461             lossIgnoresUnspecified ?
462             partialTransposeTimesSelf(Y, YTY.getRowDimension(), ru.keySetIterator()) : 
463             YTY.copy();
464 
465         double[][] WuData = MatrixUtils.accessMatrixDataDirectly(Wu);
466         double[] YTCupu = new double[features];
467 
468         for (LongFloatMap.MapEntry entry : ru.entrySet()) {
469 
470           double xu = entry.getValue();
471 
472           float[] vector = Y.get(entry.getKey());
473           if (vector == null) {
474             log.warn("No vector for {}. This should not happen. Continuing...", entry.getKey());
475             continue;
476           }
477 
478           // Wu and YTCupu
479           if (reconstructRMatrix) {
480             for (int row = 0; row < features; row++) {
481               YTCupu[row] += xu * vector[row];
482             }
483           } else {
484             double cu = 1.0 + alpha * FastMath.abs(xu);            
485             for (int row = 0; row < features; row++) {
486               float vectorAtRow = vector[row];
487               double rowValue = vectorAtRow * (cu - 1.0);
488               double[] WuDataRow = WuData[row];              
489               for (int col = 0; col < features; col++) {
490                 WuDataRow[col] += rowValue * vector[col];
491                 //Wu.addToEntry(row, col, rowValue * vector[col]);
492               }
493               if (xu > 0.0) {
494                 YTCupu[row] += vectorAtRow * cu;
495               }
496             }
497           }
498 
499         }
500 
501         double lambdaTimesCount = lambda * ru.size();
502         for (int x = 0; x < features; x++) {
503           WuData[x][x] += lambdaTimesCount;          
504           //Wu.addToEntry(x, x, lambdaTimesCount);
505         }
506 
507         float[] xu = MatrixUtils.getSolver(Wu).solveDToF(YTCupu);
508 
509         // Store result:
510         synchronized (X) {
511           X.put(work.getFirst(), xu);
512         }
513 
514         // Process is identical for computing Y from X. Swap X in for Y, Y for X, i for u, etc.
515       }
516       return null;
517     }

519     /**
520      * Like {@link MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)}, but instead of computing MT * M,
521      * it computes MT * C * M, where C is a diagonal matrix of 1s and 0s. This is like pretending some
522      * rows of M are 0.
523      * 
524      * @see MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)
525      */
526     private static RealMatrix partialTransposeTimesSelf(LongObjectMap<float[]> M,
527                                                         int dimension, 
528                                                         LongPrimitiveIterator keys) {
529       RealMatrix result = new Array2DRowRealMatrix(dimension, dimension);
530       while (keys.hasNext()) {
531         long key = keys.next();
532         float[] vector = M.get(key);
533         for (int row = 0; row < dimension; row++) {
534           float rowValue = vector[row];
535           for (int col = 0; col < dimension; col++) {
536             result.addToEntry(row, col, rowValue * vector[col]);
537           }
538         }
539       }
540       return result;
541     }
