16  package com.cloudera.oryx.als.common.candidate;
17  
18  import java.util.Collection;
19  import java.util.Iterator;
20  
21  import com.cloudera.oryx.common.collection.LongObjectMap;
22  
23  /**
24   * <p>Implementations of this interface speed up the recommendation process by pre-selecting a set of items
25   * that are the only possible candidates for recommendation, or, are the most likely to be top recommendations.</p>
26   * 
27   * <p>For example, an implementation might know that most items in the model, while useful as data, 
28   * are not recommendable since they are old -- out of date, or at least, undesirable enough as recommendations 
29   * to be excluded. It could pre-compute which items are eligible and </p>
30   * 
31   * <p>This is a form of filtering, but, differs from the filtering provided by 
32   * {@link com.cloudera.oryx.als.common.rescorer.RescorerProvider}. That is a run-time, per-request filter;
33   * this class represents a more global, precomputed filtering that is not parameterized by the request.</p>
34   *
35   * <p><em>This is quite a low-level API and should be considered "advanced" usage; use a
36   * {@link com.cloudera.oryx.als.common.rescorer.RescorerProvider} unless it's clear that it is not fast
37   * enough.</em></p>
38   *
39   * <p>Implementations should define a constructor that accepts a parameter of type {@link LongObjectMap}.
40   * This is a reference to the "Y" matrix in the model -- item-feature matrix.
41   * Access to Y is protected by a lock, but, the implementation can assume that it is locked for
42   * reading (not writing) during the constructor call, and is locked for reading (not writing) during
43   * a call to {@link #getCandidateIterator(float[][])} and while the result of that method is used.
44   * So, implementations may save and use a reference to Y if it is only used in the context of these
45   * methods and only for reading.</p>
46   * 
47   * @author Sean Owen
48   * @see com.cloudera.oryx.als.common.rescorer.RescorerProvider
49   */
50  public interface CandidateFilter {
51    
52    // Note that your implementation will need a constructor matching the following, which is how it
53    // gets a reference to the set of items:
54    
55    // public YourCandidateFilter(FastByIDMap<float[]> Y) {
56    //   ...
57    // }
58  
59    /**
60     * @param userVectors user feature vector(s) for which recommendations are being made. This may or may not
61     *  influence which items are returned. Use {@link com.cloudera.oryx.als.common.StringLongMapping#toLong(String)}
62     *  to find the numeric internal ID for a given string ID.
63     * @return a set of items most likely to be a good recommendation for the given users. These are returned
64     *  as item ID / vector pairs ({@link LongObjectMap}'s {@code MapEntry}). They are returned as an {@link Iterator},
65     *  and not just one, but potentially many. If several are returned, then the caller can process the
66     *  {@link Iterator}s in parallel for speed.
67     */
68    Collection<Iterator<LongObjectMap.MapEntry<float[]>>> getCandidateIterator(float[][] userVectors);
69  
70    /**
71     * Note a new item has appeared at run-time.
72     *
73     * @param itemID ID of new item
74     */
75    void addItem(String itemID);
76  
77  }
