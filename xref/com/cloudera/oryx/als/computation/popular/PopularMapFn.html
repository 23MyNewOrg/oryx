16  package com.cloudera.oryx.als.computation.popular;
17  
18  import com.cloudera.oryx.common.collection.LongFloatMap;
19  import com.cloudera.oryx.common.collection.LongSet;
20  import com.cloudera.oryx.common.iterator.LongPrimitiveIterator;
21  import com.cloudera.oryx.computation.common.fn.OryxMapFn;
22  import com.google.common.base.Preconditions;
23  import org.apache.crunch.Emitter;
24  import org.apache.crunch.Pair;
25  import org.apache.hadoop.mapreduce.Partitioner;
26  import org.apache.hadoop.mapreduce.lib.partition.HashPartitioner;
27  
28  public final class PopularMapFn extends OryxMapFn<Pair<Long, LongFloatMap>, Pair<Integer, LongSet>> {
29  
30    private static final Partitioner<Long,?> HASHER = new HashPartitioner<Long, Object>();
31    private static final int CHUNK_SIZE = 4096;
32  
33    private int numReducers;
34  
35    @Override
36    public void initialize() {
37      super.initialize();
38      this.numReducers = getContext().getNumReduceTasks();
39    }
40  
41    @Override
42    public void process(Pair<Long, LongFloatMap> input, Emitter<Pair<Integer, LongSet>> emitter) {
43      LongFloatMap vector = input.second();
44      Preconditions.checkState(!vector.isEmpty());
45      // vector is already large to have in memory, and a copy of its key set, copied several times through
46      // Crunch / Avro, can exhaust memory. Output in chunks.
47      LongSet targetIDs = new LongSet();
48      // Make sure we use exactly the same hash:
49      int partition = HASHER.getPartition(input.first(), null, numReducers);
50      LongPrimitiveIterator it = vector.keySetIterator();
51      while (it.hasNext()) {
52        targetIDs.add(it.nextLong());
53        if (targetIDs.size() >= CHUNK_SIZE) {
54          emitter.emit(Pair.of(partition, targetIDs));
55          targetIDs = new LongSet();
56        }
57      }
58      if (!targetIDs.isEmpty()) {
59        emitter.emit(Pair.of(partition, targetIDs));
60      }
61    }
62  
63    @Override
64    public Pair<Integer, LongSet> map(Pair<Long, LongFloatMap> input) {
65      throw new UnsupportedOperationException("process() should have been called!");
66    }
67  
68  }
