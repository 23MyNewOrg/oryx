39   * @author Sean Owen

41  public final class RowStep extends IterationStep {
42  
43    private static final Logger log = LoggerFactory.getLogger(RowStep.class);
44  
45    public static final String Y_KEY_KEY = "Y_KEY";
46    public static final String POPULAR_KEY = "POPULAR";
47    public static final String CONVERGENCE_SAMPLING_MODULUS_KEY = "CONVERGENCE_SAMPLING_MODULUS";
48  
49    @Override
50    protected MRPipeline createPipeline() throws IOException {
51  
52      IterationState iterationState = getIterationState();
53      String iterationKey = iterationState.getIterationKey();
54      boolean x = iterationState.isComputingX();
55      int lastIteration = iterationState.getIteration() - 1;
56      Store store = Store.get();
57  
58      JobStepConfig config = getConfig();
59      String instanceDir = config.getInstanceDir();
60      int generationID = config.getGenerationID();
61  
62      if (store.exists(Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "X/", false)) {
63        // Actually, looks like whole computation of X/Y finished -- just proceed
64        return null;
65      }
66  
67      // Take the opportunity to clean out iteration before last, if computing X
68      if (x) {
69        String lastLastIterationKey =
70            Namespaces.getIterationsPrefix(instanceDir, generationID) + (lastIteration - 1) + '/';
71        if (store.exists(lastLastIterationKey, false)) {
72          log.info("Deleting old iteration data from {}", lastLastIterationKey);
73          store.recursiveDelete(lastLastIterationKey);
74        }
75      }
76  
77      String yKey;
78      if (x) {
79        yKey = Namespaces.getIterationsPrefix(instanceDir, generationID) + lastIteration + "/Y/";
80      } else {
81        yKey = iterationKey + "X/";
82      }
83  
84      String xKey = iterationKey + (x ? "X/" : "Y/");
85      String rKey = Namespaces.getTempPrefix(instanceDir, generationID) + (x ? "userVectors/" : "itemVectors/");
86  
87      if (!validOutputPath(xKey)) {
88        return null;
89      }
90  
91      MRPipeline p = createBasicPipeline(RowReduceFn.class);
92      Configuration conf = p.getConfiguration();
93      conf.set(Y_KEY_KEY, yKey);
94  
95      String tempKey = Namespaces.getTempPrefix(instanceDir, generationID);
96      String popularKey = tempKey + (x ? "popularItemsByUserPartition/" : "popularUsersByItemPartition/");
97      conf.set(POPULAR_KEY, popularKey);
98  
99      YState yState = new YState(ALSTypes.DENSE_ROW_MATRIX); // Shared Y-Matrix state
100 
101     GroupingOptions opts = groupingOptions();
102     PCollection<MatrixRow> matrix = PTables.asPTable(p.read(input(rKey, ALSTypes.SPARSE_ROW_MATRIX)))
103         .groupByKey(opts)
104         .parallelDo("rowReduce", new RowReduceFn(yState), ALSTypes.DENSE_ROW_MATRIX)
105         .write(output(xKey));
106 
107     if (!x) {
108       // Configure and perform convergence sampling
109       int modulus = chooseConvergenceSamplingModulus(opts);
110       conf.setInt(CONVERGENCE_SAMPLING_MODULUS_KEY, modulus);
111 
112       matrix
113           .parallelDo("asPair", MatrixRow.AS_PAIR, Avros.tableOf(Avros.longs(), ALSTypes.FLOAT_ARRAY))
114           .parallelDo("convergenceSample", new ConvergenceSampleFn(yState), Avros.strings())
115           .write(compressedTextOutput(p.getConfiguration(), iterationKey + "Yconvergence"));
116     }
117     return p;
118   }
119   
120   private static int chooseConvergenceSamplingModulus(GroupingOptions opts) {
121     // Kind of arbitrary formula, determined empirically.
122     int modulus = RandomUtils.nextTwinPrime(16 * opts.getNumReducers() * opts.getNumReducers());
123     log.info("Using convergence sampling modulus {} to sample about {}% of all user-item pairs for convergence",
124              modulus, 100.0f / modulus / modulus);
125     return modulus;
126   }
127 
128   @Override
129   protected boolean isHighMemoryStep() {
130     return true;
131   }
132 
133   public static void main(String[] args) throws Exception {
134     run(new RowStep(), args);
135   }
