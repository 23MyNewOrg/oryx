16  package com.cloudera.oryx.als.computation.merge;
17  
18  import com.cloudera.oryx.als.common.NumericIDValue;
19  import com.cloudera.oryx.common.collection.LongFloatMap;
20  import com.cloudera.oryx.common.collection.LongSet;
21  import com.cloudera.oryx.common.settings.ConfigUtils;
22  import com.cloudera.oryx.computation.common.fn.OryxReduceDoFn;
23  import com.google.common.base.Preconditions;
24  import com.typesafe.config.Config;
25  import org.apache.commons.math3.util.FastMath;
26  import org.apache.crunch.Emitter;
27  import org.apache.crunch.Pair;
28  
29  public final class MergeNewOldValuesFn extends OryxReduceDoFn<Pair<Long, Integer>, Iterable<NumericIDValue>,
30      Pair<Long, NumericIDValue>> {
31  
32    private static final int BEFORE = 0;
33    private static final int AFTER = 1;
34  
35    private boolean doDecay;
36    private float decayFactor;
37    private float zeroThreshold;
38    private Long previousUserID;
39    private LongFloatMap previousUserPrefs;
40  
41    @Override
42    public void initialize() {
43      super.initialize();
44      Config config = ConfigUtils.getDefaultConfig();
45      decayFactor = (float) config.getDouble("model.decay.factor");
46      Preconditions.checkArgument(decayFactor > 0.0f && decayFactor <= 1.0f,
47                                  "Decay factor must be in (0,1]: %s", zeroThreshold);
48      zeroThreshold = (float) config.getDouble("model.decay.zeroThreshold");
49      Preconditions.checkArgument(zeroThreshold >= 0.0f,
50                                  "Zero threshold must be nonnegative: %s", zeroThreshold);
51      doDecay = decayFactor < 1.0f;
52    }
53  
54    @Override
55    public void process(Pair<Pair<Long, Integer>, Iterable<NumericIDValue>> input,
56                        Emitter<Pair<Long, NumericIDValue>> emitter) {
57      Pair<Long, Integer> key = input.first();
58      long currentUserID = key.first();
59  
60      if (key.second() == BEFORE) {
61  
62        // Last old data had no match, just output it
63        if (previousUserPrefs != null) {
64          Preconditions.checkNotNull(previousUserID);
65          output(previousUserID, previousUserPrefs, null, null, emitter);
66          previousUserPrefs = null;
67          previousUserID = null;
68        }
69  
70        LongFloatMap oldPrefs = new LongFloatMap();
71        for (NumericIDValue itemPref : input.second()) {
72          float oldPrefValue = itemPref.getValue();
73          Preconditions.checkState(!Float.isNaN(oldPrefValue), "No prior pref value?");
74          // Apply decay factor here, if applicable:
75          oldPrefs.increment(itemPref.getID(), doDecay ? oldPrefValue * decayFactor : oldPrefValue);
76        }
77  
78        previousUserPrefs = oldPrefs;
79        previousUserID = currentUserID;
80  
81      } else {
82        // Last old data had no match, just output it
83        if (previousUserPrefs != null && currentUserID != previousUserID) {
84          Preconditions.checkNotNull(previousUserID);
85          output(previousUserID, previousUserPrefs, null, null, emitter);
86          previousUserPrefs = null;
87          previousUserID = null;
88        }
89  
90        LongFloatMap newPrefs = new LongFloatMap();
91        LongSet removedItemIDs = new LongSet();
92        for (NumericIDValue itemPref : input.second()) {
93          long itemID = itemPref.getID();
94          float newPrefValue = itemPref.getValue();
95          if (Float.isNaN(newPrefValue)) {
96            removedItemIDs.add(itemID);
97          } else {
98            newPrefs.increment(itemID, newPrefValue);
99          }
100       }
101 
102       output(currentUserID, previousUserPrefs, newPrefs, removedItemIDs, emitter);
103 
104       previousUserPrefs = null;
105       previousUserID = null;
106     }
107   }
108 
109   @Override
110   public void cleanup(Emitter<Pair<Long, NumericIDValue>> emitter) {
111     if (previousUserPrefs != null) {
112       Preconditions.checkNotNull(previousUserID);
113       output(previousUserID, previousUserPrefs, null, null, emitter);
114     }
115     super.cleanup(emitter);
116   }
117 
118   private void output(long userID,
119                       LongFloatMap oldPrefs,
120                       LongFloatMap newPrefs,
121                       LongSet removedItemIDs,
122                       Emitter<Pair<Long, NumericIDValue>> emitter) {
123     // Old prefs may be null when there is no previous generation, for example, or the user is new.
124     // First, write out existing prefs, possibly updated by new values
125     if (oldPrefs != null && !oldPrefs.isEmpty()) {
126       for (LongFloatMap.MapEntry entry : oldPrefs.entrySet()) {
127         long itemID = entry.getKey();
128         float oldPrefValue = entry.getValue();
129         Preconditions.checkState(!Float.isNaN(oldPrefValue), "No prior pref value?");
130 
131         // May be NaN if no new data at all, or new data has no update:
132         float sum = oldPrefValue;
133         if (newPrefs != null) {
134           float newPrefValue = newPrefs.get(itemID);
135           if (!Float.isNaN(newPrefValue)) {
136             sum += newPrefValue;
137           }
138         }
139 
140         boolean remove = false;
141         if (removedItemIDs != null && removedItemIDs.contains(itemID)) {
142           remove = true;
143         } else if (FastMath.abs(sum) <= zeroThreshold) {
144           remove = true;
145         }
146 
147         if (!remove) {
148           emitter.emit(Pair.of(userID, new NumericIDValue(itemID, sum)));
149         }
150       }
151     }
152 
153     // Now output new data, that didn't exist in old prefs
154     if (newPrefs != null && !newPrefs.isEmpty()) {
155       for (LongFloatMap.MapEntry entry : newPrefs.entrySet()) {
156         long itemID = entry.getKey();
157         if (oldPrefs == null || !oldPrefs.containsKey(itemID)) {
158           // It wasn't already written. If it exists in newPrefs, it's also not removed
159           float newPrefValue = entry.getValue();
160           if (FastMath.abs(newPrefValue) > zeroThreshold) {
161             emitter.emit(Pair.of(userID, new NumericIDValue(itemID, newPrefValue)));
162           }
163         }
164       }
165     }
166   }
