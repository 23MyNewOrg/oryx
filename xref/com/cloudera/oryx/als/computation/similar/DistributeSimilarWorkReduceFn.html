16  package com.cloudera.oryx.als.computation.similar;

18  import com.cloudera.oryx.als.common.NumericIDValue;
19  import com.cloudera.oryx.als.common.TopN;

25  import com.google.common.base.Preconditions;
26  import org.apache.crunch.CrunchRuntimeException;
27  import org.apache.crunch.Emitter;
28  import org.apache.crunch.Pair;
29  import org.apache.hadoop.conf.Configuration;
30  
31  import java.io.IOException;
32  
33  public final class DistributeSimilarWorkReduceFn extends
34      OryxReduceDoFn<Integer, Iterable<MatrixRow>, Pair<Long, NumericIDValue>> {
35  
36    private LongObjectMap<float[]> partialY;
37    private int numSimilar;
38  
39    @Override
40    public void initialize() {
41      super.initialize();
42      Configuration conf = getConfiguration();
43      String yKey = conf.get(DistributeSimilarWorkStep.Y_KEY_KEY);
44      try {
45        partialY = ComputationDataUtils.loadPartialY(getPartition(), getNumPartitions(), yKey, conf);
46      } catch (IOException e) {
47        throw new CrunchRuntimeException(e);
48      }
49      numSimilar = ConfigUtils.getDefaultConfig().getInt("model.item-similarity.how-many");
50      Preconditions.checkArgument(numSimilar > 0, "# similar must be positive: %s", numSimilar);
51    }
52  
53    @Override
54    public void process(Pair<Integer, Iterable<MatrixRow>> input, Emitter<Pair<Long, NumericIDValue>> emitter) {
55      Preconditions.checkState(input.first() == getPartition(),
56          "Key must match partition: %s != %s", input.first(), getPartition());
57      for (MatrixRow value : input.second()) {
58        long itemID = value.getRowId();
59        float[] itemFeatures = value.getValues();
60        Iterable<NumericIDValue> mostSimilar = TopN.selectTopN(
61            new MostSimilarItemIterator(partialY.entrySet().iterator(), itemID, itemFeatures), numSimilar);
62        for (NumericIDValue similar : mostSimilar) {
63          emitter.emit(Pair.of(itemID, similar));
64        }
65      }
66    }
67  }
