39  final class WriteOutputs implements Callable<Object> {
40  
41    private static final Logger log = LoggerFactory.getLogger(WriteOutputs.class);
42  
43    private static final char KEY_VALUE_DELIMITER = '\t'; // Matches Hadoop TextOutputFormat
44    private static final String SINGLE_OUT_FILENAME = "0.csv.gz";
45  
46    private final File modelDir;
47    private final LongObjectMap<LongFloatMap> RbyRow;
48    private final LongObjectMap<LongSet> knownItemIDs;
49    private final LongObjectMap<float[]> X;
50    private final LongObjectMap<float[]> Y;
51    private final StringLongMapping idMapping;
52  
53    WriteOutputs(File modelDir,
54                 LongObjectMap<LongFloatMap> RbyRow,
55                 LongObjectMap<LongSet> knownItemIDs,
56                 LongObjectMap<float[]> X,
57                 LongObjectMap<float[]> Y,
58                 StringLongMapping idMapping) {
59      this.modelDir = modelDir;
60      this.RbyRow = RbyRow;
61      this.knownItemIDs = knownItemIDs;
62      this.X = X;
63      this.Y = Y;
64      this.idMapping = idMapping;
65    }
66  
67    @Override
68    public Void call() throws IOException {
69      log.info("Writing current input");
70      writeCombinedInput(RbyRow, new File(modelDir, "input"));
71      log.info("Writing known items");
72      writeIDIDsMap(knownItemIDs, new File(modelDir, "knownItems"));
73      log.info("Writing X");
74      writeIDFloatMap(X, new File(modelDir, "X"));
75      log.info("Writing Y");
76      writeIDFloatMap(Y, new File(modelDir, "Y"));
77      log.info("Writing ID mapping");
78      writeMapping(idMapping, new File(modelDir, "idMapping"));
79      log.info("Writing model");
80      File modelDescriptionFile = new File(modelDir, "model.pmml.gz");
81      ALSModelDescription modelDescription = new ALSModelDescription();
82      modelDescription.setKnownItemsPath("knownItems");
83      modelDescription.setXPath("X");
84      modelDescription.setYPath("Y");
85      modelDescription.setIDMappingPath("idMapping");
86      ALSModelDescription.write(modelDescriptionFile, modelDescription);
87      return null;
88    }
89  
90    private static void writeCombinedInput(LongObjectMap<LongFloatMap> RbyRow, File inputDir) throws IOException {
91      File outFile = new File(inputDir, SINGLE_OUT_FILENAME);
92      Files.createParentDirs(outFile);
93      Writer out = IOUtils.buildGZIPWriter(outFile);
94      try {
95        for (LongObjectMap.MapEntry<LongFloatMap> row : RbyRow.entrySet()) {
96          long rowID = row.getKey();
97          for (LongFloatMap.MapEntry entry : row.getValue().entrySet()) {
98            long colID = entry.getKey();
99            float value = entry.getValue();
100           out.write(DelimitedDataUtils.encode(Long.toString(rowID), Long.toString(colID), Float.toString(value)));
101           out.write('\n');
102         }
103       }
104     } finally {
105       out.close();
106     }
107   }

109   private static void writeIDIDsMap(LongObjectMap<LongSet> idIDs, File idIDsDir) throws IOException {
110     if (idIDs.isEmpty()) {
111       return;
112     }
113     File outFile = new File(idIDsDir, SINGLE_OUT_FILENAME);
114     Files.createParentDirs(outFile);
115     Writer out = IOUtils.buildGZIPWriter(outFile);
116     try {
117       for (LongObjectMap.MapEntry<LongSet> entry : idIDs.entrySet()) {
118         out.write(String.valueOf(entry.getKey()));
119         out.write(KEY_VALUE_DELIMITER);
120         LongSet ids = entry.getValue();
121         LongPrimitiveIterator it = ids.iterator();
122         Collection<String> keyStrings = Lists.newArrayListWithCapacity(ids.size());
123         while (it.hasNext()) {
124           keyStrings.add(Long.toString(it.nextLong()));
125         }
126         out.write(DelimitedDataUtils.encode(keyStrings));
127         out.write('\n');
128       }
129     } finally {
130       out.close();
131     }
132   }
133 
134   private static void writeIDFloatMap(LongObjectMap<float[]> idFloatMap, File idFloatDir) throws IOException {
135     if (idFloatMap.isEmpty()) {
136       return;
137     }
138     File outFile = new File(idFloatDir, SINGLE_OUT_FILENAME);
139     Files.createParentDirs(outFile);
140     Writer out = IOUtils.buildGZIPWriter(outFile);
141     try {
142       for (LongObjectMap.MapEntry<float[]> entry : idFloatMap.entrySet()) {
143         out.write(String.valueOf(entry.getKey()));
144         out.write(KEY_VALUE_DELIMITER);
145         float[] f = entry.getValue();
146         String[] floatStrings = new String[f.length];
147         for (int i = 0; i < floatStrings.length; i++) {
148           floatStrings[i] = Float.toString(f[i]);
149         }
150         out.write(DelimitedDataUtils.encode(floatStrings));
151         out.write('\n');
152       }
153     } finally {
154       out.close();
155     }
156   }

158   private static void writeMapping(StringLongMapping idMapping, File idMappingDir) throws IOException {
159     File outFile = new File(idMappingDir, SINGLE_OUT_FILENAME);
160     Files.createParentDirs(outFile);
161     Writer out = IOUtils.buildGZIPWriter(outFile);
162     try {
163       Lock lock = idMapping.getLock().readLock();
164       lock.lock();
165       try {
166         for (LongObjectMap.MapEntry<String> entry : idMapping.getReverseMapping().entrySet()) {
167           out.write(DelimitedDataUtils.encode(Long.toString(entry.getKey()), entry.getValue()));
168           out.write('\n');
169         }
170       } finally {
171         lock.unlock();
172       }
173     } finally {
174       out.close();
175     }
176   }
