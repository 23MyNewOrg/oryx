36  public final class ALSLocalGenerationRunner extends LocalGenerationRunner {
37  
38    @Override
39    protected void runSteps() throws IOException, InterruptedException, JobException {
40  
41      String instanceDir = getInstanceDir();
42      long generationID = getGenerationID();
43      String generationPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
44      long lastGenerationID = generationID - 1;
45  
46      File currentInboundDir = Files.createTempDir();
47      currentInboundDir.deleteOnExit();
48      File tempOutDir = Files.createTempDir();
49      tempOutDir.deleteOnExit();
50      File lastInputDir = null;
51      File lastMappingDir = null;
52      if (lastGenerationID >= 0) {
53        lastInputDir = Files.createTempDir();
54        lastInputDir.deleteOnExit();
55        lastMappingDir = Files.createTempDir();
56        lastMappingDir.deleteOnExit();
57      }
58  
59      try {
60  
61        Store store = Store.get();
62        store.downloadDirectory(generationPrefix + "inbound/", currentInboundDir);
63        if (lastGenerationID >= 0) {
64          store.downloadDirectory(Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "input/",
65                                  lastInputDir);
66          store.downloadDirectory(Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "idMapping/",
67                                  lastMappingDir);
68        }
69  
70        Config config = ConfigUtils.getDefaultConfig();
71  
72        boolean noKnownItems = config.getBoolean("model.no-known-items");
73        LongObjectMap<LongSet> knownItemIDs = noKnownItems ? null : new LongObjectMap<LongSet>();
74        LongObjectMap<LongFloatMap> RbyRow = new LongObjectMap<LongFloatMap>();
75        LongObjectMap<LongFloatMap> RbyColumn = new LongObjectMap<LongFloatMap>();
76        StringLongMapping idMapping = new StringLongMapping();
77  
78        if (lastGenerationID >= 0) {
79          new ReadInputs(lastInputDir, false, knownItemIDs, RbyRow, RbyColumn, idMapping).call();
80          new ReadMapping(lastMappingDir, idMapping).call();
81        }
82        new ReadInputs(currentInboundDir, true, knownItemIDs, RbyRow, RbyColumn, idMapping).call();
83  
84        if (RbyRow.isEmpty() || RbyColumn.isEmpty()) {
85          return;
86        }
87  
88        MatrixFactorizer als = new FactorMatrix(RbyRow, RbyColumn).call();
89  
90        new WriteOutputs(tempOutDir, RbyRow, knownItemIDs, als.getX(), als.getY(), idMapping).call();
91  
92        if (config.getBoolean("model.recommend.compute")) {
93          new MakeRecommendations(tempOutDir, knownItemIDs, als.getX(), als.getY(), idMapping).call();
94        }
95  
96        if (config.getBoolean("model.item-similarity.compute")) {
97          new MakeItemSimilarity(tempOutDir, als.getY(), idMapping).call();
98        }
99  
100       store.uploadDirectory(generationPrefix, tempOutDir, false);
101 
102     } finally {
103       IOUtils.deleteRecursively(currentInboundDir);
104       IOUtils.deleteRecursively(tempOutDir);
105       IOUtils.deleteRecursively(lastInputDir);
106     }
107   }
