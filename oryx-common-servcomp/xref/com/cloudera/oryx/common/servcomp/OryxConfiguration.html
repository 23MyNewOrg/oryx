35  /**
36   * {@link Configuration} subclass which manages app-specific settings. Subclasses ensure that settings
37   * specific to the underlying datastore are set as well.
38   * 
39   * @author Sean Owen
40   */
41  public final class OryxConfiguration extends Configuration {
42  
43    private static final String HADOOP_CONF_DIR_KEY = "HADOOP_CONF_DIR";
44    private static final String DEFAULT_HADOOP_CONF_DIR = "/etc/hadoop/conf";
45  
46    public OryxConfiguration() {
47      init();
48    }
49  
50    public OryxConfiguration(Configuration configuration) {
51      super(configuration);
52      init();
53    }
54  
55    private void init() {
56      if (!ConfigUtils.getDefaultConfig().getBoolean("model.local")) {
57        File hadoopConfDir = findHadoopConfDir();
58        addResource(hadoopConfDir, "hdfs-default.xml");
59        addResource(hadoopConfDir, "hdfs-site.xml");
60        addResource(hadoopConfDir, "mapred-default.xml");
61        addResource(hadoopConfDir, "mapred-site.xml");
62        addResource(hadoopConfDir, "yarn-default.xml");
63        addResource(hadoopConfDir, "yarn-site.xml");
64        addResource(hadoopConfDir, "core-site.xml");
65  
66        // Can we do away with these? the standard config generated by CDH4 still refers to old prop names
67        setFromEnvOrConf(Collections.singleton(Arrays.asList(CommonConfigurationKeysPublic.FS_DEFAULT_NAME_KEY,
68                                                             "fs.default.name")));
69  
70        fixLzoCodecIssue();
71      }
72    }
73  
74    private void addResource(File hadoopConfDir, String fileName) {
75      File file = new File(hadoopConfDir, fileName);
76      if (!file.exists()) {
77        return;
78      }
79      try {
80        addResource(file.toURI().toURL());
81      } catch (MalformedURLException e) {
82        throw new IllegalStateException(e);
83      }
84    }
85  
86    /**
87     * @param keySynonymGroupsToSet several groups of keys whose values should be set. Within each group, if a
88     *  value is found for any key, it is set to the first keys in the group, which is considered the definitive
89     *  value. This is repeated for each group.
90     */
91    private void setFromEnvOrConf(Collection<? extends Collection<String>> keySynonymGroupsToSet) {
92      Preconditions.checkArgument(!keySynonymGroupsToSet.isEmpty());
93      for (Collection<String> synonymGroup : keySynonymGroupsToSet) {
94        String value = null;
95        for (String synonym : synonymGroup) {
96          value = get(synonym);
97          if (value != null) {
98            break;
99          }
100       }
101       Preconditions.checkNotNull(value, "None of %s specified, and not set", synonymGroup);
102       set(synonymGroup.iterator().next(), value);
103     }
104   }
105 
106   private static File findHadoopConfDir() {
107     String hadoopConfPath = System.getenv(HADOOP_CONF_DIR_KEY);
108     if (hadoopConfPath == null) {
109       hadoopConfPath = DEFAULT_HADOOP_CONF_DIR;
110     }
111     File hadoopConfDir = new File(hadoopConfPath);
112     Preconditions.checkState(hadoopConfDir.exists() && hadoopConfDir.isDirectory(),
113                              "Not a directory: %s", hadoopConfDir);
114     return hadoopConfDir;
115   }

117   /**
118    * Removes LzoCodec and LzopCodec from io.compression.codecs. Implementations aren't shipped
119    * with Hadoop, but are in some cases instantiated anyway even when unused. So, try
120    * to erase them.
121    */
122   private void fixLzoCodecIssue() {
123     String codecsProperty = get("io.compression.codecs");
124     if (codecsProperty != null && !codecsProperty.isEmpty()) {
125       List<String> codecs = Lists.newArrayList(Splitter.on(',').split(codecsProperty));
126       for (Iterator<String> it = codecs.iterator(); it.hasNext(); ) {
127         if (it.next().contains(".lzo.Lzo")) {
128           it.remove();
129         }
130       }
131       set("io.compression.codecs", Joiner.on(',').join(codecs));
132     }
133   }
