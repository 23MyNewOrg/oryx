35  public final class SummaryStep extends KMeansJobStep {
36  
37    @Override
38    protected MRPipeline createPipeline() throws IOException {
39      JobStepConfig stepConfig = getConfig();
40  
41      String instanceDir = stepConfig.getInstanceDir();
42      long generationID = stepConfig.getGenerationID();
43      String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "summary/";
44      if (!validOutputPath(outputKey)) {
45        return null;
46      }
47  
48      String inboundKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "inbound/";
49  
50      InboundSettings settings = InboundSettings.create(ConfigUtils.getDefaultConfig());
51      MRPipeline p = createBasicPipeline(StringSplitFn.class);
52      PCollection<Record> records = toRecords(p.read(textInput(inboundKey)));
53      PCollection<String> jsonSummary = getSummarizer(settings).buildJson(records);
54      jsonSummary.write(compressedTextOutput(p.getConfiguration(), outputKey));
55      return p;
56    }
57  
58    public static Summarizer getSummarizer(InboundSettings settings) {
59      return new Summarizer()
60          .spec(getSpec(settings))
61          .categoricalColumns(settings.getCategoricalColumns())
62          .ignoreColumns(settings.getIgnoredColumns())
63          .ignoreColumns(settings.getIdColumns());
64  
65    }
66  
67    private static Spec getSpec(InboundSettings settings) {
68      List<String> columnNames = settings.getColumnNames();
69      if (columnNames.isEmpty()) {
70        return null;
71      }
72      RecordSpec.Builder rsb = RecordSpec.builder();
73      for (int i = 0; i < columnNames.size(); i++) {
74        String columnName = columnNames.get(i);
75        if (settings.isNumeric(i)) {
76          rsb.add(columnName, DataType.DOUBLE);
77        } else {
78          rsb.add(columnName, DataType.STRING);
79        }
80      }
81      return rsb.build();
82    }
