56    @Override
57    protected MRPipeline createPipeline() throws IOException {
58      JobStepConfig stepConfig = getConfig();
59      Config config = ConfigUtils.getDefaultConfig();
60      EvaluationSettings settings = EvaluationSettings.create(config);
61  
62      String instanceDir = stepConfig.getInstanceDir();
63      long generationID = stepConfig.getGenerationID();
64      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
65      String outputKey = prefix + "eval/";
66      if (!validOutputPath(outputKey)) {
67        return null;
68      }
69      String inputKey = prefix + "weighted/weightedKSketchVectors/";
70      MRPipeline p = createBasicPipeline(ClosestSketchVectorFn.class);
71  
72      PType<Pair<Integer, WeightedRealVector>> inputType = KMeansTypes.FOLD_WEIGHTED_VECTOR;
73      PCollection<Pair<Integer, WeightedRealVector>> weightedSketchVectors = p.read(avroInput(inputKey, inputType));
74  
75      PCollection<KMeansEvaluationData> evaluation = weightedSketchVectors
76          .parallelDo("replicate",
77              new ReplicateValuesFn<Pair<Integer, WeightedRealVector>>(settings.getKValues(), settings.getReplications()),
78              Avros.tableOf(Avros.pairs(Avros.ints(), Avros.ints()), Avros.pairs(Avros.ints(), MLAvros.weightedVector())))
79          .groupByKey(settings.getParallelism())
80          .parallelDo("cluster",
81              new KMeansClusteringFn(settings),
82              Serializables.avro(KMeansEvaluationData.class));
83  
84      // Write out the centers themselves to a text file
85      evaluation.parallelDo("replicaCenters", new CentersOutputFn(prefix), Avros.strings())
86          .write(compressedTextOutput(p.getConfiguration(), outputKey + "replicaCenters/"));
87  
88      // Write out the per-replica stats
89      evaluation.parallelDo("replicaStats", new StatsOutputFn(), Avros.strings())
90          .write(compressedTextOutput(p.getConfiguration(), outputKey + "replicaStats/"));
91  
92      return p;
93    }

95    @Override
96    protected void postRun() throws IOException {
97      JobStepConfig stepConfig = getConfig();
98      String instanceDir = stepConfig.getInstanceDir();
99      long generationID = stepConfig.getGenerationID();
100     String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
101     String replicaStatsKey = prefix + "eval/replicaStats/";
102     String replicaCentersKey = prefix + "eval/replicaCenters/";
103 
104     boolean succeeded = false;
105     try {
106       List<ClusterValidityStatistics> stats = findBest(replicaStatsKey);
107       uploadCenters(prefix, replicaCentersKey, stats);
108       succeeded = true;
109     } finally {
110       if (!succeeded) {
111         // This causes a re-run of the M/R next time if failed
112         Store.get().delete(prefix + "_SUCCESS");
113       }
114     }
115   }

117   private static List<ClusterValidityStatistics> findBest(String replicaStatsKey) throws IOException {
118     KMeansEvalStrategy evalStrategy = EvaluationSettings.create(ConfigUtils.getDefaultConfig()).getEvalStrategy();
119     Store store = Store.get();
120     List<ClusterValidityStatistics> stats = Lists.newArrayList();
121     for (String statsKey : store.list(replicaStatsKey, true)) {
122       for (String line : new FileLineIterable(store.readFrom(statsKey))) {
123         ClusterValidityStatistics cvs = ClusterValidityStatistics.parse(line);
124         log.info("PS: {}", cvs);
125         stats.add(cvs);
126       }
127     }
128     return evalStrategy == null ? stats : evalStrategy.evaluate(stats);
129   }

131   private static void uploadCenters(String prefix,
132                                     String replicaCentersKey,
133                                     List<ClusterValidityStatistics> bestStats) throws IOException {
134     Store store = Store.get();
135     List<Model> models = Lists.newArrayList();
136     String bestName = null;
137     if (bestStats.size() == 1) {
138       ClusterValidityStatistics cvs = bestStats.get(0);
139       bestName = prefix + ':' + cvs.getK() + ':' + cvs.getReplica();
140       log.info("Best scoring model = {}", bestName);
141     }
142     log.info("Reading model centers from key = {}", replicaCentersKey);
143     DataDictionary dictionary = null;
144     for (String modelKey : store.list(replicaCentersKey, true)) {
145       try {
146         PMML pmml = KMeansPMML.read(store.streamFrom(modelKey));
147         log.info("Read {} from key = {}", pmml.getModels().size(), modelKey);
148         if (dictionary == null) {
149           dictionary = pmml.getDataDictionary();
150         }
151         if (bestName == null) {
152           models.addAll(pmml.getModels());
153         } else {
154           for (Model m : pmml.getModels()) {
155             if (bestName.equals(m.getModelName())) {
156               models = ImmutableList.of(m);
157               break;
158             }
159           }
160         }
161       } catch (JAXBException e) {
162         log.error("Serialization error", e);
163       } catch (SAXException e) {
164         log.error("Serialization error", e);
165       }
166     }
167 
168     File keptModels = File.createTempFile("model", ".pmml.gz");
169     keptModels.deleteOnExit();
170     log.info("Writing {} models to model.pmml.gz...", models.size());
171     KMeansPMML.write(keptModels, dictionary, models);
172     store.upload(prefix + "model.pmml.gz", keptModels, true);
173     IOUtils.delete(keptModels);
174   }
