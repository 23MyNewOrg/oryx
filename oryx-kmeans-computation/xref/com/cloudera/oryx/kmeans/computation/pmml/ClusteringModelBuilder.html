51  public class ClusteringModelBuilder {

53    private final MiningSchema miningSchema;
54    private final DataDictionary dictionary;
55    private LocalTransformations transforms;
56    private List<ClusteringField> clusteringFields;

58    public ClusteringModelBuilder(Summary summary) {
59      InboundSettings settings = InboundSettings.create(ConfigUtils.getDefaultConfig());
60      this.miningSchema = PMMLUtils.buildMiningSchema(settings);
61      this.dictionary = PMMLUtils.buildDataDictionary(settings, summary.getCategoryLevelsMapping());
62      buildSummaryInfo(summary, settings);
63    }
64  
65    public DataDictionary getDictionary() {
66      return dictionary;
67    }

69    public ClusteringModel build(String modelName, Centers centers) {
70      ClusteringModel model = new ClusteringModel(
71          miningSchema,
72          new ComparisonMeasure(ComparisonMeasure.Kind.DISTANCE).withMeasure(new SquaredEuclidean()),
73          MiningFunctionType.CLUSTERING,
74          ClusteringModel.ModelClass.CENTER_BASED,
75          centers.size());
76      model.setModelName(modelName);
77      model.setAlgorithmName("K-Means||");
78      model.setLocalTransformations(transforms);
79      model.getClusteringFields().addAll(clusteringFields);
80      for (int i = 0; i < centers.size(); i++) {
81        model.getClusters().add(toCluster(centers.get(i), i));
82      }
83      return model;
84    }

86    private void buildSummaryInfo(Summary summary, InboundSettings settings) {
87      this.transforms = new LocalTransformations();
88      this.clusteringFields = Lists.newArrayList();
89      NormalizeSettings normalize = NormalizeSettings.create(ConfigUtils.getDefaultConfig());
90      List<String> columnNames = settings.getColumnNames();
91      int offset = 0;
92      for (int i = 0; i < columnNames.size(); i++) {
93        if (settings.isIgnored(i)) {
94          // Skip processing
95        } else {
96          SummaryStats ss = summary.getStats(i);
97          FieldName baseName = new FieldName(columnNames.get(i));
98          if (settings.isNumeric(i)) {
99            FieldName fn = baseName;
100           Transform t = normalize.getTransform(i);
101           if (t != Transform.NONE) {
102             Expression e = null;
103             if (t == Transform.LINEAR) {
104               e = new NormContinuous(baseName).withLinearNorms(
105                   new LinearNorm(ss.min(), 0.0),
106                   new LinearNorm(ss.max(), 1.0));
107             } else if (t == Transform.LOG) {
108               e = new Apply("ln").withExpressions(new FieldRef(baseName));
109             } else if (t == Transform.Z) {
110               e = new NormContinuous(baseName).withLinearNorms(
111                   new LinearNorm(0.0, -ss.mean()/ss.stdDev()),
112                   new LinearNorm(ss.mean(), 0.0));
113             }
114             fn = new FieldName(columnNames.get(i) + "_normed");
115             DerivedField df = new DerivedField(OpType.CONTINUOUS, DataType.DOUBLE);
116             df.setName(fn);
117             df.setExpression(e);
118             transforms.getDerivedFields().add(df);
119             offset++;
120           }
121           ClusteringField cf = new ClusteringField(fn);
122           if (normalize.getScale(i) != 1.0) {
123             cf.setFieldWeight(normalize.getScale(i));
124           }
125           clusteringFields.add(cf);
126         } else if (settings.isCategorical(i)) {
127           List<String> levels = summary.getStats(i).getLevels();
128           for (String level : levels) {
129             DerivedField df = new DerivedField(OpType.CONTINUOUS, DataType.DOUBLE);
130             FieldName fn = new FieldName(columnNames.get(i) + ':' + level);
131             df.setName(fn);
132             df.setExpression(new NormDiscrete(baseName, level));
133             transforms.getDerivedFields().add(df);
134 
135             ClusteringField cf = new ClusteringField(fn);
136             if (normalize.getScale(i) != 1.0) {
137               cf.setFieldWeight(normalize.getScale(i));
138             }
139             clusteringFields.add(cf);
140             offset++;
141           }
142         }
143       }
144     }
145   }

147   private static Cluster toCluster(RealVector point, int pointId) {
148     Cluster cluster = new Cluster();
149     cluster.setId(String.valueOf(pointId));
150     Array array = new Array(toString(point), Array.Type.REAL);
151     array.setN(point.getDimension());
152     cluster.setArray(array);
153     return cluster;
154   }
155 
156   private static String toString(RealVector vec) {
157     StringBuilder sb = new StringBuilder();
158     for (int i = 0; i < vec.getDimension(); i++) {
159       if (i > 0) {
160         sb.append(' ');
161       }
162       sb.append(vec.getEntry(i));
163     }
164     return sb.toString();
165   }
