34  /**
35   * An internal data structure that manages the locations of the current centers during
36   * k-means|| processing.
37   */
38  public final class KSketchIndex implements Serializable {

40    private final int[] pointsPerFold;
41    private final List<List<BitSet>> indices;
42    private final List<List<RealVector>> points;
43    private final List<List<Double>> lengthSquared;
44    private final int dimensions;
45    private final int projectionBits;
46    private final int projectionSamples;
47    private final long seed;
48    private double[] projection;
49    private boolean updated;

51    public KSketchIndex(int numFolds, int dimensions, int projectionBits, int projectionSamples, long seed) {
52      this.pointsPerFold = new int[numFolds];
53      this.indices = Lists.newArrayList();
54      this.points = Lists.newArrayList();
55      this.lengthSquared = Lists.newArrayList();
56      for (int i = 0; i < numFolds; i++) {
57        points.add(Lists.<RealVector>newArrayList());
58        lengthSquared.add(Lists.<Double>newArrayList());
59      }
60      this.dimensions = dimensions;
61      this.projectionBits = projectionBits;
62      this.projectionSamples = projectionSamples;
63      this.seed = seed;
64    }

66    public KSketchIndex(List<Centers> centers, int projectionBits, int projectionSamples, long seed) {
67      this(centers.size(), centers.get(0).get(0).getDimension(), projectionBits, projectionSamples, seed);
68      for (int centerId = 0; centerId < centers.size(); centerId++) {
69        for (RealVector v : centers.get(centerId)) {
70          add(v, centerId);
71        }
72      }
73    }

87    public void rebuildIndices() {
88      if (projection == null) {
89        RandomGenerator r = RandomManager.getSeededRandom(seed);
90        this.projection = new double[dimensions * projectionBits];
91        for (int i = 0; i < projection.length; i++) {
92          projection[i] = r.nextGaussian();
93        }
94      }
95      indices.clear();
96      for (List<RealVector> px : points) {
97        List<BitSet> indx = Lists.newArrayList();
98        for (RealVector aPx : px) {
99          indx.add(index(aPx));
100       }
101       indices.add(indx);
102     }
103     updated = false;
104   }
105   
106   public void add(RealVector vec, int centerId) {
107     points.get(centerId).add(vec);
108     double length = vec.getNorm();
109     lengthSquared.get(centerId).add(length * length);
110     pointsPerFold[centerId]++;
111     updated = true;
112   }

114   private BitSet index(RealVector vec) {
115     final double[] prod = new double[projectionBits];
116 
117     vec.walkInDefaultOrder(new AbstractRealVectorPreservingVisitor() {
118       @Override
119       public void visit(int index, double value) {
120         for (int j = 0; j < projectionBits; j++) {
121           prod[j] += value * projection[index + j * dimensions];
122         }
123       }
124     });
125 
126     BitSet bitset = new BitSet(projectionBits);
127     for (int i = 0; i < projectionBits; i++) {
128       if (prod[i] > 0.0) {
129         bitset.set(i);
130       }
131     }
132     return bitset;
133   }

143   public Distance getDistance(RealVector vec, int id, boolean approx) {
144     double distance = Double.POSITIVE_INFINITY;
145     int closestPoint = -1;
146     if (approx) {
147       if (updated) {
148         rebuildIndices();
149       }
150 
151       BitSet q = index(vec);
152       List<BitSet> index = indices.get(id);
153       SortedSet<Idx> lookup = Sets.newTreeSet();
154       for (int j = 0; j < index.size(); j++) {
155         Idx idx = new Idx(hammingDistance(q, index.get(j)), j);
156         if (lookup.size() < projectionSamples) {
157           lookup.add(idx);
158         } else if (idx.compareTo(lookup.last()) < 0) {
159           lookup.add(idx);
160           lookup.remove(lookup.last());
161         }
162       }
163 
164       List<RealVector> p = points.get(id);
165       List<Double> lsq = lengthSquared.get(id);
166       for (Idx idx : lookup) {
167         double lenSq = lsq.get(idx.getIndex());
168         double length = vec.getNorm();
169         double d = length * length + lenSq - 2 * vec.dotProduct(p.get(idx.getIndex()));
170         if (d < distance) {
171           distance = d;
172           closestPoint = idx.getIndex();
173         }
174       }
175     } else { // More expensive exact computation
176       List<RealVector> px = points.get(id);
177       List<Double> lsq = lengthSquared.get(id);
178       for (int j = 0; j < px.size(); j++) {
179         RealVector p = px.get(j);
180         double lenSq = lsq.get(j);
181         double length = vec.getNorm();
182         double d = length * length + lenSq - 2 * vec.dotProduct(p);
183         if (d < distance) {
184           distance = d;
185           closestPoint = j;
186         }
187       }
188     }
189     
190     return new Distance(distance, closestPoint);
191   }

193   static final class Idx implements Comparable<Idx> {
194     private final int distance;
195     private final int index;
196     
197     Idx(int distance, int index) {
198       this.distance = distance;
199       this.index = index;
200     }
201 
202     int getIndex() {
203       return index;
204     }
205 
206     @Override
207     public int compareTo(Idx idx) {
208       if (distance < idx.distance) {
209         return -1;
210       }
211       if (distance > idx.distance) {
212         return 1;
213       }
214       return 0;
215     }
216     
217     @Override
218     public boolean equals(Object o) {
219       if (!(o instanceof Idx)) {
220         return false;
221       }
222       Idx other = (Idx) o;
223       return distance == other.distance && index == other.index;
224     }
225     
226     @Override
227     public int hashCode() {
228       return distance ^ index;
229     }
230   }

232   private static int hammingDistance(BitSet q, BitSet idx) {
233     BitSet x = new BitSet(q.size());
234     x.or(q);
235     x.xor(idx);
236     return x.cardinality();
237   }

239   public List<WeightedRealVector> getWeightedVectorsForFold(int foldId, long[] weights) {
240     List<WeightedRealVector> ret = Lists.newArrayList();
241     int i = 0;
242     for (RealVector vec : points.get(foldId)) {
243       ret.add(new WeightedRealVector(vec, weights[i]));
244       i++;
245     }
246     return ret;
247   }

249   public List<List<WeightedRealVector>> getWeightedVectors(ClosestSketchVectorData data) {
250     List<List<WeightedRealVector>> ret = Lists.newArrayList();
251     for (int i = 0; i < data.getNumFolds(); i++) {
252       List<RealVector> p = points.get(i);
253       List<WeightedRealVector> weighted = Lists.newArrayList();
254       for (int j = 0; j < p.size(); j++) { // TODO: Assume static? Or fold specific? Or something?
255         weighted.add(new WeightedRealVector(p.get(j), data.get(i, j)));
256       }
257       ret.add(weighted);
258     }
259     return ret;
260   }
