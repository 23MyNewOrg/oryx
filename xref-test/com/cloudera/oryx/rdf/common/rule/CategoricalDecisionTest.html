32  public final class CategoricalDecisionTest extends OryxTest {
33  
34    @Test
35    public void testDecisionBasics() {
36      CategoricalDecision decision = new CategoricalDecision(0, new BitSet(1), false);
37      assertEquals(0, decision.getFeatureNumber());
38      assertEquals(new BitSet(1), decision.getCategoryIDs());
39      assertFalse(decision.getDefaultDecision());
40      assertEquals(FeatureType.CATEGORICAL, decision.getType());
41    }
42  
43    @Test
44    public void testDecision() {
45      BitSet activeCategories = new BitSet(10);
46      activeCategories.set(2);
47      activeCategories.set(5);
48      Decision decision = new CategoricalDecision(0, activeCategories, true);
49      for (int i = 0; i < 10; i++) {
50        assertEquals(activeCategories.get(i), decision.isPositive(new Example(null, CategoricalFeature.forValue(i))));
51      }
52      assertTrue(decision.isPositive(new Example(null, new Feature[] {null})));
53    }
