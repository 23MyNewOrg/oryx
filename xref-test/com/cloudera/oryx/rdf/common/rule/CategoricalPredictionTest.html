16  package com.cloudera.oryx.rdf.common.rule;
17  
18  import org.junit.Test;
19  
20  import com.cloudera.oryx.common.OryxTest;
21  import com.cloudera.oryx.rdf.common.example.CategoricalFeature;
22  import com.cloudera.oryx.rdf.common.example.Example;
23  import com.cloudera.oryx.rdf.common.example.FeatureType;
24  
25  /**
26   * Tests {@link CategoricalPrediction}.
27   *
28   * @author Sean Owen
29   */
30  public final class CategoricalPredictionTest extends OryxTest {
31  
32    @Test
33    public void testConstruct() {
34      int[] counts = { 0, 1, 3, 0, 4, 0};
35      CategoricalPrediction prediction = new CategoricalPrediction(counts);
36      assertEquals(FeatureType.CATEGORICAL, prediction.getFeatureType());
37      assertEquals(4, prediction.getMostProbableCategoryID());
38      assertArrayEquals(counts, prediction.getCategoryCounts());
39      assertArrayEquals(new float[] {0.0f, 0.125f, 0.375f, 0.0f, 0.5f, 0.0f}, prediction.getCategoryProbabilities());
40    }
41  
42    @Test
43    public void testConstructFromProbability() {
44      float[] probability = {0.0f, 0.125f, 0.375f, 0.0f, 0.5f, 0.0f};
45      CategoricalPrediction prediction = new CategoricalPrediction(probability);
46      assertEquals(FeatureType.CATEGORICAL, prediction.getFeatureType());
47      assertEquals(4, prediction.getMostProbableCategoryID());
48      assertNull(prediction.getCategoryCounts());
49      assertArrayEquals(probability, prediction.getCategoryProbabilities());
50    }
51  
52    @Test
53    public void testUpdate() {
54      int[] counts = { 0, 1, 3, 0, 4, 0};
55      CategoricalPrediction prediction = new CategoricalPrediction(counts);
56      Example example = new Example(CategoricalFeature.forValue(2));
57      prediction.update(example);
58      prediction.update(example);
59      assertEquals(2, prediction.getMostProbableCategoryID());
60      counts[2] += 2;
61      assertArrayEquals(counts, prediction.getCategoryCounts());
62      assertArrayEquals(new float[] {0.0f, 0.1f, 0.5f, 0.0f, 0.4f, 0.0f}, prediction.getCategoryProbabilities());
63    }
64  
65  }
