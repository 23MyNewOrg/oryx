16  package com.cloudera.oryx.common.stats;
17  
18  import org.junit.Test;
19  
20  import com.cloudera.oryx.common.OryxTest;
21  
22  /**
23   * Tests {@link DoubleWeightedMean}.
24   *
25   * @author Sean Owen
26   */
27  public final class DoubleWeightedMeanTest extends OryxTest {
28  
29    @Test
30    public void testNone() {
31      DoubleWeightedMean mean = new DoubleWeightedMean();
32      assertEquals(0, mean.getN());
33      assertTrue(Double.isNaN(mean.getResult()));
34    }
35  
36    @Test
37    public void testOne() {
38      DoubleWeightedMean mean = new DoubleWeightedMean();
39      mean.increment(1.5);
40      assertEquals(1, mean.getN());
41      assertEquals(1.5, mean.getResult());
42    }
43  
44    @Test
45    public void testWeighted() {
46      DoubleWeightedMean mean = new DoubleWeightedMean();
47      mean.increment(0.2, 4.0);
48      mean.increment(-0.1, 2.0);
49      assertEquals(2, mean.getN());
50      assertEquals(0.1, mean.getResult());
51    }
52  
53    @Test
54    public void testNegative() {
55      DoubleWeightedMean mean = new DoubleWeightedMean();
56      mean.increment(-0.1, 2.1);
57      mean.increment(0.1, 2.1);
58      assertEquals(2, mean.getN());
59      assertEquals(0.0, mean.getResult());
60    }
61  
62    @Test
63    public void testComplex() {
64      DoubleWeightedMean mean = new DoubleWeightedMean();
65      for (int i = 1; i <= 5; i++) {
66        mean.increment(1.0 / (i + 1), i);
67      }
68      assertEquals(5, mean.getN());
69      assertEquals((1.0/2.0 + 2.0/3.0 + 3.0/4.0 + 4.0/5.0 + 5.0/6.0) / (15.0), mean.getResult());
70    }
71  
72  }
