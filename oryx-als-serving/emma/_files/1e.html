677  private static void updateFeatures(float[] userFeatures, float[] itemFeatures, float value, Generation generation) {
678    if (userFeatures == null || itemFeatures == null) {
679      return;
680    }
681    double signedFoldInWeight = foldInWeight(SimpleVectorMath.dot(userFeatures, itemFeatures), value);
682    if (signedFoldInWeight == 0.0) {
683      return;
684    }
685    // Here, we are using userFeatures, which is a row of X, as if it were a column of X'.
686    // This is multiplied on the left by (X'*X)^-1. That's our left-inverse of X or at least the one
687    // column we need. Which is what the new data point is multiplied on the left by. The result is a column;
688    // we scale to complete the multiplication of the fold-in and add it in.
689    Solver xtxSolver = generation.getXTXSolver();
690    double[] itemFoldIn = xtxSolver == null ? null : xtxSolver.solveFToD(userFeatures);
691 
692    // Same, but reversed. Multiply itemFeatures, which is a row of Y, on the right by (Y'*Y)^-1.
693    // This is the right-inverse for Y', or at least the row we need. Because of the symmetries we can use
694    // the same method above to carry out the multiply; the result is conceptually a row vector.
695    // The result is scaled and added in.
696    Solver ytySolver = generation.getYTYSolver();
697    double[] userFoldIn = ytySolver == null ? null : ytySolver.solveFToD(itemFeatures);
698 
699    if (itemFoldIn != null) {
700      for (int i = 0; i < itemFeatures.length; i++) {
701        double delta = signedFoldInWeight * itemFoldIn[i];
702        Preconditions.checkState(Doubles.isFinite(delta));
703        itemFeatures[i] += (float) delta;
704      }
705    }
706    if (userFoldIn != null) {
707      for (int i = 0; i < userFeatures.length; i++) {
708        double delta = signedFoldInWeight * userFoldIn[i];
709        Preconditions.checkState(Doubles.isFinite(delta));
710        userFeatures[i] += (float) delta;
711      }
712    }
713  }

715  private static int countFeatures(LongObjectMap<float[]> M) {
716    // assumes the read lock is held
717    return M.isEmpty() ? 0 : M.entrySet().iterator().next().getValue().length;
718  }
719 
720  /**
721   * This function decides how much of a folded-in user or item vector should be added to a target item or user
722   * vector, respectively, on a new action. The idea is that a positive value should push the current value towards
723   * 1, but not further, and a negative value should push towards 0, but not further. How much to move should be
724   * mostly proportional to the size of the value. 0 should move the result not at all; 2 ought to move twice as
725   * much as 1, etc. This isn't quite possible but can be approximated by moving a fraction 1-1/(1+value) of the
726   * distance towards 1, or 0.
727   */
728  private static double foldInWeight(double estimate, float value) {
729    Preconditions.checkState(Doubles.isFinite(estimate));
730    double signedFoldInWeight;
731    if (value > 0.0f && estimate < 1.0) {
732      double multiplier = 1.0 - FastMath.max(0.0, estimate);
733      signedFoldInWeight = (1.0 - 1.0 / (1.0 + value)) * multiplier;
734    } else if (value < 0.0f && estimate > 0.0) {
735      double multiplier = -FastMath.min(1.0, estimate);
736      signedFoldInWeight = (1.0 - 1.0 / (1.0 - value)) * multiplier;
737    } else {
738      signedFoldInWeight = 0.0;
739    }
740    return signedFoldInWeight;
741  }

955          similarities[i] = (float) (SimpleVectorMath.dot(features, toFeatures) / (featuresNorm * toFeaturesNorm));
956        }
957      }
958      if (!anyFound) {

960      }
961 
962    } finally {

964    }
