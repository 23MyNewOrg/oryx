24  import javax.servlet.ServletRequest;
25  import javax.servlet.ServletResponse;

27  import java.io.IOException;
28  import java.io.Writer;

30  import com.cloudera.oryx.als.common.IDValue;
31  import com.cloudera.oryx.als.common.OryxRecommender;
32  import com.cloudera.oryx.als.serving.RescorerProvider;
33  import com.cloudera.oryx.common.io.DelimitedDataUtils;
34  import com.cloudera.oryx.serving.web.AbstractOryxServlet;

36  /**
37   * Superclass of all servletes used in the ALS recommender API.
38   *
39   * @author Sean Owen
40   */
41  public abstract class AbstractALSServlet extends AbstractOryxServlet {
42  
43    private static final int DEFAULT_HOW_MANY = 10;
44    static final Splitter SLASH = Splitter.on('/').omitEmptyStrings();
45  
46    private static final String KEY_PREFIX = AbstractALSServlet.class.getName();
47    public static final String RECOMMENDER_KEY = KEY_PREFIX + ".RECOMMENDER";
48    private static final String RESCORER_PROVIDER_KEY = KEY_PREFIX + ".RESCORER_PROVIDER";
49    private static final String[] NO_PARAMS = new String[0];
50  
51    private OryxRecommender recommender;
52    private RescorerProvider rescorerProvider;
53  
54    @Override
55    public void init(ServletConfig config) throws ServletException {
56      super.init(config);
57      ServletContext context = config.getServletContext();
58      recommender = (OryxRecommender) context.getAttribute(RECOMMENDER_KEY);
59      rescorerProvider = (RescorerProvider) context.getAttribute(RESCORER_PROVIDER_KEY);
60    }
61  
62    final OryxRecommender getRecommender() {
63      return recommender;
64    }
65  
66    final RescorerProvider getRescorerProvider() {
67      return rescorerProvider;
68    }
69  
70    static int getHowMany(ServletRequest request) {
71      String howManyString = request.getParameter("howMany");
72      if (howManyString == null) {
73        return DEFAULT_HOW_MANY;
74      }
75      int howMany = Integer.parseInt(howManyString);
76      Preconditions.checkArgument(howMany > 0, "howMany must be positive");
77      return howMany;
78    }
79  
80    static String[] getRescorerParams(ServletRequest request) {
81      String[] rescorerParams = request.getParameterValues("rescorerParams");
82      return rescorerParams == null ? NO_PARAMS : rescorerParams;
83    }
84  
85    static boolean getConsiderKnownItems(ServletRequest request) {
86      return Boolean.valueOf(request.getParameter("considerKnownItems"));
87    }
88  
89    /**
90     * <p>CSV output contains one recommendation per line, and each line is of the form {@code itemID,strength},
91     * like {@code "ABC",0.53}. Strength is an opaque indicator of the relative quality of the recommendation.</p>
92     */
93    final void output(ServletResponse response, Iterable<IDValue> items) throws IOException {
94      Writer writer = response.getWriter();
95      for (IDValue item : items) {
96        writer.write(DelimitedDataUtils.encode(item.getID(), Float.toString(item.getValue())));
97        writer.write('\n');
98      }
99    }
