232  private static List<Decision> sortAndGetDecisionsOverSubset(int featureNumber,
233                                                              int categoryCount,
234                                                              List<Pair<Double,Integer>> sorted,
235                                                              int maxCategory,
236                                                              int suggestedMaxSplitCandidates) {
237    if (sorted.isEmpty()) {
238      return Collections.emptyList();
239    }
240    Collections.sort(sorted, PairComparator.INSTANCE);
241    // The vital condition here is that if decision n decides an example is positive, then all subsequent
242    // decisions in the list will also find it positive. Here we create decisions with sets of categories that
243    // are always supersets of previous ones.
244    int numDecisions = FastMath.min(sorted.size(), suggestedMaxSplitCandidates);
245    // Going to take the decisions based on earlier splits, if we need to cut it down.
246    // These are decisions that partition the data into a low-entropy subset, and everything else.
247    // These are likely to be the best.
248    List<Decision> decisions = Lists.newArrayListWithCapacity(numDecisions - 1);
249    for (int i = 1; i < numDecisions; i++) {
250      decisions.add(new CategoricalDecision(featureNumber, categoryCount, sorted, 0, i, maxCategory));
251    }
252    return decisions;
253  }
