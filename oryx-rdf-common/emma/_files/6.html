352  private static TreeNode translateFromPMML(Node root,

362      Collection<ScoreDistribution> scoreDistributions = root.getScoreDistributions();
363      Prediction prediction;
364      if (scoreDistributions != null && !scoreDistributions.isEmpty()) {
365        // Categorical target
366        Map<String,Integer> valueToID = columnToCategoryNameToIDMapping.get(targetColumn);
367        int[] categoryCounts = new int[valueToID.size()];
368        for (ScoreDistribution dist : scoreDistributions) {
369          int valueID = valueToID.get(dist.getValue());
370          categoryCounts[valueID] = (int) Math.round(dist.getRecordCount());
371        }
372        prediction = new CategoricalPrediction(categoryCounts);
373      } else {
374        prediction = new NumericPrediction(Float.parseFloat(root.getScore()), (int) Math.round(root.getRecordCount()));
375      }
376      return new TerminalNode(prediction);
377    }
378 
379    Preconditions.checkArgument(children.size() == 2);
380    // Decision
381    Node child1 = children.get(0);
382    Node child2 = children.get(1);
383    Node negativeLeftChild;
384    Node positiveRightChild;
385    if (child1.getPredicate().getClass().equals(True.class)) {
386      negativeLeftChild = child1;
387      positiveRightChild = child2;
388    } else {

392    }

401      Preconditions.checkArgument(simplePredicate.getOperator() == SimplePredicate.Operator.GREATER_OR_EQUAL);

403      int featureNumber = columnNames.indexOf(simplePredicate.getField().getValue());
