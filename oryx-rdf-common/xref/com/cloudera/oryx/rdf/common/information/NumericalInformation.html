29  /**
30   * <p>Utility methods for computing information, measured in nats, over a numeric feature. This
31   * includes information gain induced by a {@link Decision} over that feature. Here, information
32   * is measured by evaluating <a href="http://en.wikipedia.org/wiki/Differential_entropy">differential entropy</a>
33   * over a set of values of that numeric feature. The decision tree model supposes that the values are normally
34   * distributed around the mean, so differential entropy of ln(stdev * sqrt(2*pi*e)) is used.</p>
35   *
36   * <p>It would be more correct, I believe, to use
37   * <a href="http://en.wikipedia.org/wiki/Limiting_density_of_discrete_points">relative entropy</a>.
38   * This is computed as something like -integral(p log p/m)  =  -integral(p log p - p log m)  =
39   * -integral(p log p) + integral(p log m). m(x) here is a constant and equal to the range of the values
40   * in the set, which is a constant. The first term is differential entropy, and the second is log(m) * integral(p),
41   * which is log(m) since the integral is 1. This is just a constant too, log(m). This will end up falling
42   * out too so has no impact on which split is chosen to have the best gain.</p>
43   *
44   * @author Sean Owen
45   * @see CategoricalInformation
46   */
47  final class NumericalInformation {
48    
49    //private static final double HALF_LN_2_PI_E = 0.5 * Math.log(2.0 * Math.PI * Math.E);
50  
51    private NumericalInformation() {
52    }

54    static Pair<Decision,Double> bestGain(Iterable<Decision> decisions, ExampleSet examples) {
55  
56      UpdatableVariance varianceNegativeStat = new UpdatableVariance();
57  
58      // Start with everything considered a negative example:
59      for (Example example : examples) {
60        float value = ((NumericFeature) example.getTarget()).getValue();
61        varianceNegativeStat.increment(value);
62      }
63  
64      // Save this off
65      double varianceAll = varianceNegativeStat.getResult();
66      if (Double.isNaN(varianceAll) || varianceAll <= 0.0) {
67        // Weird case, no information at all
68        return null;
69      }
70      // Entropy in nats is ln (stdev * sqrt(2*pi*e)) = ln(stdev) + 0.5*ln(2*pi*e) = 0.5*ln(variance) + ...
71      // Actually to compute gain, we only need to know the ln(variance) since the additive constants will
72      // fall out when we take the difference in entropies.
73      //double entropyAll = 0.5 * Math.log(varianceAll) + HALF_LN_2_PI_E;
74      double logVarianceAll = Math.log(varianceAll);
75  
76      List<Example> exampleList = examples.getExamples();
77      BitSet notYetPositiveExamples = new BitSet(exampleList.size());
78      notYetPositiveExamples.set(0, exampleList.size());
79      Decision bestDecision = null;
80      double bestGain = Double.NEGATIVE_INFINITY;
81      UpdatableVariance variancePositiveStat = new UpdatableVariance();
82  
83      for (Decision decision : decisions) {
84        boolean noChange = true;
85        int nextNotYetPositive = -1;
86        while ((nextNotYetPositive = notYetPositiveExamples.nextSetBit(nextNotYetPositive + 1)) >= 0) {
87          Example example = exampleList.get(nextNotYetPositive);
88          if (decision.isPositive(example)) {
89            noChange = false;
90            float value = ((NumericFeature) example.getTarget()).getValue();
91            varianceNegativeStat.decrement(value);
92            variancePositiveStat.increment(value);
93            notYetPositiveExamples.clear(nextNotYetPositive);
94          }
95        }
96        if (noChange) {
97          continue;
98        }
99  
100       double variancePositive = variancePositiveStat.getResult();
101       double varianceNegative = varianceNegativeStat.getResult();
102       if (Double.isNaN(variancePositive) || variancePositive <= 0.0 ||
103           Double.isNaN(varianceNegative) || varianceNegative <= 0.0) {
104         continue;
105       }
106 
107       //double entropyNegative = 0.5 * Math.log(varianceNegative) + HALF_LN_2_PI_E;
108       //double entropyPositive = 0.5 * Math.log(variancePositive) + HALF_LN_2_PI_E;
109       double logVarianceNegative = Math.log(varianceNegative);
110       double logVariancePositive = Math.log(variancePositive);
111       long numNegative = varianceNegativeStat.getN();
112       long numPositive = variancePositiveStat.getN();
113       //double oldgain = entropyAll -
114       //    (numNegative * entropyNegative + numPositive * entropyPositive) / (numNegative + numPositive);
115       double gain =
116           0.5 * (logVarianceAll - (numNegative * logVarianceNegative + numPositive * logVariancePositive) /
117                                   (numNegative + numPositive));
118       if (gain > bestGain) {
119         bestGain = gain;
120         bestDecision = decision;
121       }
122     }
123 
124     return bestDecision == null ? null : new Pair<Decision,Double>(bestDecision, bestGain);
125   }
