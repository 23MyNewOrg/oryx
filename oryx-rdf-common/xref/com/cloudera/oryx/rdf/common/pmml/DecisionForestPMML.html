92    /**
93     * @param pmmlFile file to write PMML representation to
94     * @param forest {@link DecisionForest} to encode as PMML
95     * @param columnToCategoryNameToIDMapping {@link Map} from column number in the input, to a {@link BiMap}
96     *  mapping between category value names and category value IDs (for categorical feature columns only). This
97     *  is necessary because the {@link DecisionForest} operates in terms of value IDs, but the PMML encoding
98     *  should encode the names of these category values -- {@code female}, not {@code 2}
99     */
100   public static void write(File pmmlFile,
101                            DecisionForest forest,
102                            Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping) throws IOException {
103 
104     InboundSettings inboundSettings = InboundSettings.create(ConfigUtils.getDefaultConfig());
105     List<String> columnNames = inboundSettings.getColumnNames();
106     int targetColumn = inboundSettings.getTargetColumn();
107     boolean classificationTask = inboundSettings.isCategorical(targetColumn);
108 
109     MiningFunctionType miningFunctionType =
110         classificationTask ? MiningFunctionType.CLASSIFICATION : MiningFunctionType.REGRESSION;
111     MiningSchema miningSchema = PMMLUtils.buildMiningSchema(inboundSettings, columnNames, targetColumn);
112     MiningModel miningModel = new MiningModel(miningSchema, miningFunctionType);
113     MultipleModelMethodType multipleModelMethodType = classificationTask ?
114         MultipleModelMethodType.WEIGHTED_MAJORITY_VOTE :
115         MultipleModelMethodType.WEIGHTED_AVERAGE;
116     Segmentation segmentation = new Segmentation(multipleModelMethodType);
117     miningModel.setSegmentation(segmentation);
118 
119     int treeID = 0;
120     for (DecisionTree tree : forest) {
121       Segment segment = buildTreeModel(forest,
122                                        columnToCategoryNameToIDMapping,
123                                        miningFunctionType,
124                                        miningSchema,
125                                        treeID,
126                                        tree,
127                                        inboundSettings);
128       segmentation.getSegments().add(segment);
129       treeID++;
130     }
131 
132     DataDictionary dictionary = PMMLUtils.buildDataDictionary(inboundSettings, columnToCategoryNameToIDMapping);
133     PMML pmml = new PMML(null, dictionary, "4.1");
134     pmml.getModels().add(miningModel);
135 
136     OutputStream out = IOUtils.buildGZIPOutputStream(new FileOutputStream(pmmlFile));
137     try {
138       IOUtil.marshal(pmml, out);
139     } catch (JAXBException jaxbe) {
140       throw new IOException(jaxbe);
141     } finally {
142       out.close();
143     }
144   }

146   private static Segment buildTreeModel(DecisionForest forest,
147                                         Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping,
148                                         MiningFunctionType miningFunctionType,
149                                         MiningSchema miningSchema,
150                                         int treeID,
151                                         DecisionTree tree,
152                                         InboundSettings settings) {
153 
154     List<String> columnNames = settings.getColumnNames();
155     int targetColumn = settings.getTargetColumn();
156 
157     Node root = new Node();
158     root.setId("r");
159 
160     // Queue<Node> modelNodes = Queues.newArrayDeque();
161     Queue<Node> modelNodes = new ArrayDeque<Node>();
162     modelNodes.add(root);
163 
164     Queue<Pair<TreeNode, Decision>> treeNodes = new ArrayDeque<Pair<TreeNode, Decision>>();
165     treeNodes.add(new Pair<TreeNode,Decision>(tree.getRoot(), null));
166 
167     while (!treeNodes.isEmpty()) {
168 
169       Pair<TreeNode,Decision> treeNodePredicate = treeNodes.remove();
170       Node modelNode = modelNodes.remove();
171 
172       // This is the decision that got us here from the parent, if any; not the predicate at this node
173       Predicate predicate = buildPredicate(treeNodePredicate.getSecond(),
174                                            columnNames,
175                                            columnToCategoryNameToIDMapping);
176       modelNode.setPredicate(predicate);
177 
178       TreeNode treeNode = treeNodePredicate.getFirst();
179       if (treeNode.isTerminal()) {
180 
181         TerminalNode terminalNode = (TerminalNode) treeNode;
182         modelNode.setRecordCount((double) terminalNode.getCount());
183 
184         Prediction prediction = terminalNode.getPrediction();
185 
186         if (prediction.getFeatureType() == FeatureType.CATEGORICAL) {
187 
188           Map<Integer,String> categoryIDToName =
189               columnToCategoryNameToIDMapping.get(targetColumn).inverse();
190           CategoricalPrediction categoricalPrediction = (CategoricalPrediction) prediction;
191           int[] categoryCounts = categoricalPrediction.getCategoryCounts();
192           float[] categoryProbabilities = categoricalPrediction.getCategoryProbabilities();
193           for (int categoryID = 0; categoryID < categoryProbabilities.length; categoryID++) {
194             int categoryCount = categoryCounts[categoryID];
195             float probability = categoryProbabilities[categoryID];
196             if (categoryCount > 0 && probability > 0.0f) {
197               String categoryName = categoryIDToName.get(categoryID);
198               ScoreDistribution distribution = new ScoreDistribution(categoryName, categoryCount);
199               distribution.setProbability((double) probability);
200               modelNode.getScoreDistributions().add(distribution);
201             }
202           }
203 
204         } else {
205 
206           NumericPrediction numericPrediction = (NumericPrediction) prediction;
207           modelNode.setScore(Float.toString(numericPrediction.getPrediction()));
208         }
209 
210       } else {
211 
212         DecisionNode decisionNode = (DecisionNode) treeNode;
213         Decision decision = decisionNode.getDecision();
214 
215         Node positiveModelNode = new Node();
216         positiveModelNode.setId(modelNode.getId() + '+');
217         modelNode.getNodes().add(positiveModelNode);
218         Node negativeModelNode = new Node();
219         negativeModelNode.setId(modelNode.getId() + '-');
220         modelNode.getNodes().add(negativeModelNode);
221         modelNode.setDefaultChild(
222             decision.getDefaultDecision() ? positiveModelNode.getId() : negativeModelNode.getId());
223         modelNodes.add(positiveModelNode);
224         modelNodes.add(negativeModelNode);
225         treeNodes.add(new Pair<TreeNode,Decision>(decisionNode.getRight(), decision));
226         treeNodes.add(new Pair<TreeNode,Decision>(decisionNode.getLeft(), null));
227 
228       }
229 
230     }
231 
232     TreeModel treeModel = new TreeModel(miningSchema, root, miningFunctionType);
233     treeModel.setSplitCharacteristic(TreeModel.SplitCharacteristic.BINARY_SPLIT);
234     treeModel.setMissingValueStrategy(MissingValueStrategyType.DEFAULT_CHILD);
235 
236     Segment segment = new Segment();
237     segment.setId(Integer.toString(treeID));
238     segment.setPredicate(new True());
239     segment.setModel(treeModel);
240     segment.setWeight(forest.getWeights()[treeID]);
241 
242     return segment;
243   }

245   private static Predicate buildPredicate(Decision decision,
246                                           List<String> columnNames,
247                                           Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping) {
248     Predicate predicate;
249     if (decision == null) {
250       predicate = new True();
251 
252     } else {
253       int columnNumber = decision.getFeatureNumber();
254       FieldName fieldName = new FieldName(columnNames.get(columnNumber));
255 
256       if (decision.getType() == FeatureType.CATEGORICAL) {
257         CategoricalDecision categoricalDecision = (CategoricalDecision) decision;
258         Map<Integer,String> categoryIDToName = columnToCategoryNameToIDMapping.get(columnNumber).inverse();
259         BitSet includedCategoryIDs = categoricalDecision.getCategoryIDs();
260         List<String> categoryNames = Lists.newArrayList();
261         int categoryID = -1;
262         while ((categoryID = includedCategoryIDs.nextSetBit(categoryID + 1)) >= 0) {
263           categoryNames.add(categoryIDToName.get(categoryID));
264         }
265         Array categories = new Array(DelimitedDataUtils.encode(categoryNames, ' '), Array.Type.STRING);
266         predicate = new SimpleSetPredicate(categories, fieldName, SimpleSetPredicate.BooleanOperator.IS_IN);
267 
268       } else {
269         NumericDecision numericDecision = (NumericDecision) decision;
270         SimplePredicate numericPredicate = new SimplePredicate(fieldName, SimplePredicate.Operator.GREATER_OR_EQUAL);
271         numericPredicate.setValue(Float.toString(numericDecision.getThreshold()));
272         predicate = numericPredicate;
273       }
274     }
275     return predicate;
276   }

281   /**
282    * @param pmmlFile file to read PMML encoding from
283    * @return a {@link DecisionForest} representation of the PMML encoded model
284    */
285   public static Pair<DecisionForest, Map<Integer,BiMap<String,Integer>>> read(File pmmlFile) throws IOException {
286 
287     PMML pmml;
288     InputStream in = IOUtils.openMaybeDecompressing(pmmlFile);
289     try {
290       pmml = IOUtil.unmarshal(in);
291     } catch (SAXException e) {
292       throw new IOException(e);
293     } catch (JAXBException e) {
294       throw new IOException(e);
295     } finally {
296       in.close();
297     }
298 
299     List<Model> models = pmml.getModels();
300     Preconditions.checkNotNull(models);
301     Preconditions.checkArgument(!models.isEmpty());
302     Preconditions.checkArgument(models.get(0) instanceof MiningModel);
303     MiningModel miningModel = (MiningModel) models.get(0);
304 
305     Segmentation segmentation = miningModel.getSegmentation();
306     Preconditions.checkNotNull(segmentation);
307 
308     List<Segment> segments = segmentation.getSegments();
309     Preconditions.checkNotNull(segments);
310     Preconditions.checkArgument(!segments.isEmpty());
311 
312     Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping = PMMLUtils.buildColumnCategoryMapping(
313         pmml.getDataDictionary());
314     InboundSettings settings = InboundSettings.create(ConfigUtils.getDefaultConfig());
315     DecisionTree[] trees = new DecisionTree[segments.size()];
316     double[] weights = new double[trees.length];
317     for (int i = 0; i < trees.length; i++) {
318       Segment segment = segments.get(i);
319       weights[i] = segment.getWeight();
320       TreeModel treeModel = (TreeModel) segment.getModel();
321       TreeNode root = translateFromPMML(treeModel.getNode(), columnToCategoryNameToIDMapping, settings);
322       trees[i] = new DecisionTree(root);
323     }
324 
325     return new Pair<DecisionForest, Map<Integer, BiMap<String, Integer>>>(
326         new DecisionForest(trees, weights),
327         columnToCategoryNameToIDMapping);
328   }

330   private static TreeNode translateFromPMML(Node root,
331                                             Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping,
332                                             InboundSettings settings) {
333 
334     List<String> columnNames = settings.getColumnNames();
335     int targetColumn = settings.getTargetColumn();
336 
337     List<Node> children = root.getNodes();
338     if (children.isEmpty()) {
339       // Terminal
340       Collection<ScoreDistribution> scoreDistributions = root.getScoreDistributions();
341       Prediction prediction;
342       if (scoreDistributions != null && !scoreDistributions.isEmpty()) {
343         // Categorical target
344         Map<String,Integer> valueToID = columnToCategoryNameToIDMapping.get(targetColumn);
345         int[] categoryCounts = new int[valueToID.size()];
346         for (ScoreDistribution dist : scoreDistributions) {
347           int valueID = valueToID.get(dist.getValue());
348           categoryCounts[valueID] = (int) Math.round(dist.getRecordCount());
349         }
350         prediction = new CategoricalPrediction(categoryCounts);
351       } else {
352         prediction = new NumericPrediction(Float.parseFloat(root.getScore()), (int) Math.round(root.getRecordCount()));
353       }
354       return new TerminalNode(prediction);
355     }
356 
357     Preconditions.checkArgument(children.size() == 2);
358     // Decision
359     Node child1 = children.get(0);
360     Node child2 = children.get(1);
361     Node negativeLeftChild;
362     Node positiveRightChild;
363     if (child1.getPredicate().getClass().equals(True.class)) {
364       negativeLeftChild = child1;
365       positiveRightChild = child2;
366     } else {
367       Preconditions.checkArgument(child2.getPredicate().getClass().equals(True.class));
368       negativeLeftChild = child2;
369       positiveRightChild = child1;
370     }
371 
372     Decision decision;
373     Predicate predicate = positiveRightChild.getPredicate();
374     boolean defaultDecision = positiveRightChild.getId().equals(root.getDefaultChild());
375 
376     if (predicate instanceof SimplePredicate) {
377       // Numeric decision
378       SimplePredicate simplePredicate = (SimplePredicate) predicate;
379       Preconditions.checkArgument(simplePredicate.getOperator() == SimplePredicate.Operator.GREATER_OR_EQUAL);
380       float threshold = Float.parseFloat(simplePredicate.getValue());
381       int featureNumber = columnNames.indexOf(simplePredicate.getField().getValue());
382       decision = new NumericDecision(featureNumber, threshold, defaultDecision);
383 
384     } else {
385       // Cateogrical decision
386       Preconditions.checkArgument(predicate instanceof SimpleSetPredicate);
387       SimpleSetPredicate simpleSetPredicate = (SimpleSetPredicate) predicate;
388       Preconditions.checkArgument(simpleSetPredicate.getBooleanOperator() == SimpleSetPredicate.BooleanOperator.IS_IN);
389       int featureNumber = columnNames.indexOf(simpleSetPredicate.getField().getValue());
390       Map<String,Integer> categoryNameToID = columnToCategoryNameToIDMapping.get(featureNumber);
391       String[] categories = DelimitedDataUtils.decode(simpleSetPredicate.getArray().getValue(), ' ');
392       BitSet activeCategories = new BitSet(categoryNameToID.size());
393       for (String category : categories) {
394         int categoryID = categoryNameToID.get(category);
395         activeCategories.set(categoryID);
396       }
397       decision = new CategoricalDecision(featureNumber, activeCategories, defaultDecision);
398     }
399 
400     return new DecisionNode(decision,
401                             translateFromPMML(negativeLeftChild, columnToCategoryNameToIDMapping, settings),
402                             translateFromPMML(positiveRightChild, columnToCategoryNameToIDMapping, settings));
403   }
